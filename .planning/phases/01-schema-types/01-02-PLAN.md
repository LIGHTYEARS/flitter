## JSON-RPC 2.0 消息格式

```typescript
// Request
{ jsonrpc: "2.0"; id: string | number; method: string; params?: unknown }

// Response (success)
{ jsonrpc: "2.0"; id: string | number; result: unknown }

// Response (error)
{ jsonrpc: "2.0"; id: string | number; error: { code: number; message: string; data?: unknown } }

// Notification (no id)
{ jsonrpc: "2.0"; method: string; params?: unknown }
```

## MCP 连接状态 (type 判别，8 种)

```typescript
type MCPConnectionStatus =
  | { type: "connecting" }
  | { type: "authenticating" }
  | { type: "reconnecting"; attempt: number; nextRetryMs: number }
  | { type: "connected"; capabilities: Record<string, unknown>; serverInfo: Record<string, unknown> }
  | { type: "denied" }
  | { type: "awaiting-approval" }
  | { type: "failed"; error: MCPConnectionError }
  | { type: "blocked-by-registry"; registryUrl: string };

interface MCPConnectionError {
  code: "timeout" | "network" | "server-error" | "auth-failed" | "spawn-failed" | "permission-denied";
  message: string;
  stderr?: string;
}
```

## MCP 服务器规范

```typescript
// Command 类型
interface MCPCommandServerSpec {
  command: string;
  args?: string[];
  env?: Record<string, string>;
  _target?: string;
}

// URL 类型
interface MCPURLServerSpec {
  url: string;
  headers?: Record<string, string>;
  transport?: string;
  _target?: string;
}
```

## MCP 工具定义

```typescript
interface MCPToolSpec {
  name: string;
  description: string;
  inputSchema: object;
  source: { mcp: string; target?: string };
  meta?: object;
}
```

## MCP 传输类型

```typescript
type MCPTransportType = "StdioClientTransport" | "SSEClientTransport" | "StreamableHTTPClientTransport";
```

## JSON-RPC 错误码

```typescript
type JSONRPCErrorCode = "RequestTimeout" | "ConnectionClosed" | "InvalidRequest" |
  "InvalidParams" | "ParseError" | "MethodNotFound" | "InternalError";
```

## Skill 系统

```typescript
interface SkillFrontmatter {
  name: string;
  description: string;
  compatibility?: string;
  isolatedContext?: boolean;
  "disable-model-invocation"?: boolean;
  "argument-hint"?: string;
  "builtin-tools"?: string[];
}

interface Skill {
  name: string;
  description: string;
  frontmatter: SkillFrontmatter;
  content: string;
  baseDir: string;
  mcpServers?: Record<string, MCPServerSpec>;
  builtinTools?: string[];
  files?: string[];
}
```

```typescript
type GuidanceFileType = "project" | "user" | "parent" | "subtree" | "mentioned";

interface GuidanceFile {

  type: GuidanceFileType;
  priority: number;
  globs?: string[];
}
```
