## 配置键分类

```typescript
// Admin 可覆盖的键
type AdminOverrideKey =
  | "agent.deepReasoningEffort" | "agent.skipTitleGenerationIfMessageContains"
  | "anthropic.effort" | "anthropic.interleavedThinking.enabled"
  | "anthropic.provider" | "anthropic.speed" | "anthropic.temperature"
  | "anthropic.thinking.enabled" | "gemini.thinkingLevel"
  | "internal.compactionThresholdPercent" | "internal.model"
  | "internal.oracleReasoningEffort" | "openai.speed"
  | "tools.disable" | "tools.enable";

// 数组合并键 (concat + dedup)
type MergedArrayKey =
  | "guardedFiles.allowlist" | "mcpPermissions"
  | "tools.disable" | "tools.enable" | "permissions";

// 仅全局键
type GlobalOnlyKey = "mcpServers" | "mcpPermissions" | "url";

// 密钥键（独立存储）
type SecretKey =
  | "apiKey" | "buildkite-access-token" | "github-access-token"
  | "gitlab-access-token" | "gitlab-instance-url"
  | "mcp-oauth-client-secret" | "mcp-oauth-token";
```

## Settings 完整结构
```typescript
interface Settings {
  // 认证 & API
  url?: string;
  proxy?: string;

  // 模型配置
  "anthropic.speed"?: string;
  "anthropic.temperature"?: number;
  "anthropic.thinking.enabled"?: boolean;
  "anthropic.interleavedThinking.enabled"?: boolean;
  "anthropic.effort"?: string;
  "anthropic.provider"?: string;
  "openai.speed"?: string;
  "gemini.thinkingLevel"?: string;
  "internal.model"?: string;
  "internal.compactionThresholdPercent"?: number;
  "internal.oracleReasoningEffort"?: string;
  "internal.scaffoldCustomizationFile"?: string;
  "internal.fireworks.directRouting"?: boolean;
  "internal.kimi.reasoning"?: string;

  // Agent 配置
  "agent.deepReasoningEffort"?: string;
  "agent.skipTitleGenerationIfMessageContains"?: string;
  "agent.showUsageDebugInfo"?: boolean;

  // 工具配置
  "tools.disable"?: string[];
  "tools.enable"?: string[];
  "tools.inactivityTimeout"?: number;   // 默认 300
  "tools.stopTimeout"?: number;         // 默认 300
  "network.timeout"?: number;           // 默认 30

  // MCP (引用 01-02 的 MCPServerSpec 类型)
  mcpServers?: Record<string, MCPServerSpec>;
  mcpPermissions?: MCPPermissionEntry[];
  mcpTrustedServers?: string[];

  // 权限 (引用 01-05 的 PermissionEntry 类型，此处用 z.any() 占位)
  permissions?: unknown[];

  // Skills
  "skills.path"?: string;
  "skills.disableClaudeCodeSkills"?: boolean;
  "toolbox.path"?: string;

  // 终端
  "terminal.animation"?: string;
  "terminal.theme"?: string;
  "terminal.commands.nodeSpawn.loadProfile"?: boolean;

  // 其他
  systemPrompt?: string;
  hooks?: Record<string, unknown>;
  workspaces?: Record<string, unknown>;
  "notifications.system.enabled"?: boolean;
  "fuzzy.alwaysIncludePaths"?: string[];
  "experimental.autoSnapshot"?: boolean;
  "experimental.agentMode"?: string;
  "experimental.cli.commandTelemetry.enabled"?: boolean;
  "git.commit.ampThread.enabled"?: boolean;
  "git.commit.coauthor.enabled"?: boolean;
  "guardedFiles.allowlist"?: string[];
}
```
