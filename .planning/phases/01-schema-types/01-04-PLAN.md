## ThreadSnapshot (持久化根对象)
```typescript
interface ThreadSnapshot {
  id: string;
  v: number;
  title?: string;
  agentMode?: string;
  messages: ThreadMessage[];
  queuedMessages?: QueuedMessageEntry[];
  relationships?: ThreadRelationship[];
  nextMessageId?: number;
  meta?: ThreadMeta;
  env?: ThreadEnvironment;
}
```

## ThreadMessage (role 判别)
```typescript
interface UserThreadMessage {
  role: "user";
  content: ContentBlock[];
  messageId: number;
  dtwMessageID?: string;
  meta?: { sentAt?: number };
  userState?: Record<string, unknown>;
  readAt?: number;
  agentMode?: string;
  discoveredGuidanceFiles?: GuidanceFileRef[];
  parentToolUseId?: string;
}

interface AssistantThreadMessage {
  role: "assistant";
  content: ContentBlock[];
  messageId: number;
  dtwMessageID?: string;
  meta?: Record<string, unknown>;
  usage?: Usage;
  readAt?: number;
  state: AssistantMessageState;
  cancelled?: boolean;
  parentToolUseId?: string;
}

interface InfoThreadMessage {
  role: "info";
  content: ContentBlock[];
  messageId: number;
  dtwMessageID?: string;
  parentToolUseId?: string;
}
```

## GuidanceFileRef
```typescript
interface GuidanceFileRef {
  uri: string;
  lineCount: number;
  content?: string;
}
```

## ThreadRelationship
```typescript
interface ThreadRelationship {
  threadID: string;
  type: "handoff";
  messageIndex?: number;
  blockIndex?: number;
  comment?: string;
}
```

## QueuedMessageEntry
```typescript
interface QueuedMessageEntry {
  id: string;
  interrupt: boolean;
  queuedMessage: UserThreadMessage;
}
```

## ThreadMeta / ThreadEnvironment
```typescript
interface ThreadMeta {
  status?: string;
  executorType?: string;
}

interface ThreadEnvironment {
  initial?: { trees?: Array<{ repository?: { url?: string } }> };
  trees: unknown[];
  platform: string;
}
```

## ConnectionState (5 种)
```typescript
type ConnectionState = "connected" | "reconnecting" | "disconnected" | "connecting" | "authenticating";
```

## ConnectionInfo
```typescript
interface ConnectionInfo {
  state: ConnectionState;
  role?: "executor" | "observer";
  clientId?: string;
  threadId?: string;
  reconnectCause?: ReconnectCause;
}

interface ReconnectCause {
  type: string;
  at: number;
  code?: number;
  reason?: string;
  error?: string;
}
```

```typescript
type CompactionState = "idle" | "compacting";
```

## AgentLoopState

```typescript
type AgentLoopState = "idle" | string;
```

## ConnectionMode
```typescript
type ConnectionMode = "observer" | "executor+observer";
```

```typescript
interface ThreadSummary {

  updatedAt: string;

  workspaceURI?: string;
```

## Delta 事件类型 (24+ 种)
```typescript
type ConversationDelta =
  | { type: "cancelled" }
  | { type: "user:message"; message: UserMessagePayload; index?: number }
  | { type: "user:message:interrupt"; messageIndex: number }
  | { type: "user:message:append-content"; messageId: number; content: unknown }
  | { type: "user:message-queue:enqueue"; message: UserMessagePayload }
  | { type: "user:message-queue:dequeue" }
  | { type: "user:message-queue:discard"; id?: string }
  | { type: "user:tool-input"; toolUse: string; value: unknown }
  | { type: "tool:data"; toolUse: string; data: unknown }
  | { type: "tool:processed"; toolUse: string; newArgs: unknown }
  | { type: "assistant:message"; message: AssistantMessagePayload }
  | { type: "assistant:message-update"; message: AssistantMessagePayload }
  | { type: "inference:completed"; usage?: Usage; model?: string }
  | { type: "title"; value: string }
  | { type: "max-tokens"; value: string }
  | { type: "main-thread"; value: string }
  | { type: "agent-mode"; mode: string }
  | { type: "environment"; env: unknown }
  | { type: "thread:truncate"; fromIndex: number }
  | { type: "info:manual-bash-invocation"; args: unknown; toolRun: unknown; hidden?: boolean }
  | { type: "clearPendingNavigation" }
  | { type: "setPendingNavigation"; threadID: string }
  | { type: "relationship"; relationship: ThreadRelationship }
  | { type: "draft"; content: unknown; autoSubmit?: boolean }
  | { type: "trace:start"; span: unknown }
  | { type: "trace:end"; span: unknown }
  | { type: "trace:event"; span: unknown; event: unknown }
  | { type: "trace:attributes"; span: unknown; attributes: unknown };
```
