## Observable (from vendor/esm/observable-impl.js)

```typescript
class Observable<T> {
  constructor(subscriber: (observer: Observer<T>) => (() => void) | void);
  subscribe(observer: Observer<T>): Subscription;
  subscribe(next?: (value: T) => void, error?: (err: any) => void, complete?: () => void): Subscription;
  pipe<R>(...operators: OperatorFunction<any, any>[]): Observable<R>;
  static from<T>(input: ObservableInput<T>): Observable<T>;
  static of<T>(...values: T[]): Observable<T>;
  [Symbol.observable](): Observable<T>;
}

interface Observer<T> {
  next(value: T): void;
  error(err: any): void;
  complete(): void;
}

interface Subscription {
  unsubscribe(): void;
  closed: boolean;
}

type OperatorFunction<T, R> = (source: Observable<T>) => Observable<R>;
type ObservableInput<T> = Observable<T> | Iterable<T> | Promise<T>;
```

## Subject (from module-gs-constructor-y0T.js — W0)

```typescript
class Subject<T> extends Observable<T> {
  next(value: T): void;
  error(err: any): void;
  complete(): void;
}
```

## BehaviorSubject (from module-gs-constructor-y0T.js — f0)

```typescript
class BehaviorSubject<T> extends Subject<T> {
  constructor(initialValue: T);
  getValue(): T;
  get value(): T;  // alias for getValue
}
```

## IDisposable

```typescript
interface IDisposable {
  dispose(): void;
}

class DisposableCollection implements IDisposable {
  add(disposable: IDisposable): void;
  add(fn: () => void): void;
  dispose(): void;
  get disposed(): boolean;
}
```

## ObservableSet (from module-gs-y0t-CnR.js — EnR)

```typescript
class ObservableSet<T> extends Set<T> implements IDisposable {
```

   - `get value(): T` — getValue 的别名
