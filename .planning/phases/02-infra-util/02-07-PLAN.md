```typescript
// === error.ts ===

class TimeoutError extends Error {
  constructor(message: string, timeout: number);
  readonly timeout: number;
}

async function callWithTimeout<T>(promise: Promise<T>, timeout: number, message?: string): Promise<T>;

// === logger.ts ===

interface Logger {
  debug(message: string, ...args: unknown[]): void;
  info(message: string, ...args: unknown[]): void;
  warn(message: string, ...args: unknown[]): void;
  error(message: string, ...args: unknown[]): void;
  wsMessage(direction: "SEND" | "RECV", clientId: string, data: unknown): void;
  child(context: Record<string, unknown>): Logger;
}

function createLogger(name: string): Logger;

// === assert.ts ===

function assert(condition: unknown, message?: string): asserts condition;
function assertNever(value: never, message?: string): never;
function assertDefined<T>(value: T | undefined | null, message?: string): T;

// === process.ts ===

interface SpawnOptions {
  cwd?: string;
  env?: Record<string, string>;
  timeout?: number;
  maxBuffer?: number;
  signal?: AbortSignal;
}

interface SpawnResult {
  stdout: string;
  stderr: string;
  exitCode: number | null;
}

async function spawn(command: string, args: string[], options?: SpawnOptions): Promise<SpawnResult>;
```
</interfaces>

<tasks>
1. **Create `packages/util/src/error.ts`** — TimeoutError + callWithTimeout + Result type
   - Implement `TimeoutError` extending `Error`:
     - Constructor sets `this.name = 'TimeoutError'` and stores `timeout` as readonly
     - Proper prototype chain for `instanceof` checks
   - Implement `callWithTimeout<T>(promise, timeout, message?)`:
     - Race the input promise against a `setTimeout` that rejects with `TimeoutError`
     - Clear the timer on resolution to avoid leaks
     - If `message` not provided, use a default like `"Operation timed out after ${timeout}ms"`
   - Optionally export a `Result<T, E>` type alias (`{ ok: true; value: T } | { ok: false; error: E }`)

2. **Create `packages/util/src/error.test.ts`** — error module tests
   - `TimeoutError` is an instance of both `Error` and `TimeoutError`
   - `TimeoutError.timeout` returns the configured value
   - `TimeoutError.name` is `'TimeoutError'`
   - `callWithTimeout` resolves normally when promise completes before timeout
   - `callWithTimeout` rejects with `TimeoutError` when promise exceeds timeout
   - `callWithTimeout` uses custom message when provided
   - `callWithTimeout` uses default message when message not provided
   - Timer is cleaned up after resolution (no dangling timers)
   - Works with immediately resolving promises (timeout=0 edge case handled)

3. **Create `packages/util/src/logger.ts`** — Logger interface + createLogger
   - Implement `Logger` interface with structured JSON output:
     - Each log line is a JSON object: `{ timestamp, level, name, message, ...context, ...extra }`
     - `timestamp` in ISO 8601 format
     - `level` is the string `"debug"`, `"info"`, `"warn"`, or `"error"`
     - Extra args are merged into the JSON output
   - Implement `wsMessage(direction, clientId, data)`:
     - Log at debug level with `direction`, `clientId`, and serialized `data`
   - Implement `child(context)`:
     - Return a new Logger instance whose context is merged with parent context
     - Child inherits parent `name` and context; new context overrides on conflict
   - `createLogger(name)` — factory that returns a Logger with the given name
   - Output to `process.stderr` (or configurable stream)

4. **Create `packages/util/src/logger.test.ts`** — logger tests
   - `createLogger` returns a Logger with the given name
   - `info()` outputs JSON with correct timestamp, level, name, message
   - `debug/warn/error` set the correct level string
   - Extra args appear in the output JSON
   - `child()` creates logger with merged context
   - `child()` context overrides parent on key conflict
   - Nested `child().child()` accumulates context
   - `wsMessage` logs direction, clientId, and data at debug level
   - Timestamp is valid ISO 8601

5. **Create `packages/util/src/assert.ts`** — assert/assertNever/assertDefined
   - `assert(condition, message?)`:
     - If `condition` is falsy, throw `Error` with `message` (default: `"Assertion failed"`)
     - TypeScript `asserts condition` return type narrows types
   - `assertNever(value: never, message?)`:
     - Always throws — used in switch/if exhaustiveness checks
     - Default message includes the unexpected value for debugging
   - `assertDefined<T>(value: T | undefined | null, message?): T`:
     - If `value` is `null` or `undefined`, throw `Error`
     - Otherwise return `value` (typed as `T`)

6. **Create `packages/util/src/assert.test.ts`** — assert tests
   - `assert(true)` does not throw
   - `assert(false)` throws with default message
   - `assert(false, 'custom')` throws with custom message
   - `assert` accepts truthy values: non-empty string, number, object
   - `assert` rejects falsy values: `0`, `""`, `null`, `undefined`, `NaN`
   - `assertNever` always throws (test with `as never` cast)
   - `assertNever` includes the unexpected value in the error message
   - `assertDefined(value)` returns value when not null/undefined
   - `assertDefined(null)` throws
   - `assertDefined(undefined)` throws
   - `assertDefined` preserves the type (TypeScript compile check)

7. **Create `packages/util/src/process.ts`** — spawn wrapper
   - Implement `spawn(command, args, options?)`:
     - Use `child_process.execFile` (not `exec`) for safety
     - Return `{ stdout, stderr, exitCode }` on completion
     - On non-zero exit, still resolve (not reject) with the exitCode
     - On spawn error (e.g. ENOENT), reject with a descriptive error
     - Respect `options.timeout` — kill process after timeout ms
     - Respect `options.signal` (AbortSignal) — kill process on abort
     - Respect `options.cwd` and `options.env`
     - Respect `options.maxBuffer` (pass to execFile)

8. **Create `packages/util/src/process.test.ts`** — process tests
   - Executes a simple command (`echo hello`) and returns stdout
   - Returns stderr from a command that writes to stderr
   - Returns correct exitCode for successful command (0)
   - Returns non-zero exitCode for failing command (does not reject)
   - Rejects on command not found (ENOENT)
   - `cwd` option changes working directory
   - `env` option sets environment variables
   - `timeout` option kills long-running process and rejects/returns timeout indication
   - `signal` option: aborting the signal kills the process
   - `maxBuffer` option: exceeding buffer causes error
   - Handles commands with spaces in arguments

9. **Update `packages/util/src/index.ts` barrel export**
   - Export all public types, classes, and functions from error, logger, assert, process modules
