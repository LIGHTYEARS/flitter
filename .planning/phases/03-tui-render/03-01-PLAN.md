## Color (from tui-render-pipeline.js — LT 类)

```typescript
/** 终端颜色，支持 4 种模式 */
class Color {
  /** 颜色种类: "default" | "named" | "index" | "rgb" */
  readonly kind: ColorKind;
  /** named/index 时的索引值 (0-255) */
  readonly index: number;
  /** rgb 时的 r/g/b 分量 (0-255) */
  readonly r: number;
  readonly g: number;
  readonly b: number;

  static default(): Color;
  static black(): Color;       // index 0
  static red(): Color;         // index 1
  static green(): Color;       // index 2
  static yellow(): Color;      // index 3
  static blue(): Color;        // index 4
  static magenta(): Color;     // index 5
  static cyan(): Color;        // index 6
  static white(): Color;       // index 7
  static brightBlack(): Color;   // index 8
  static brightRed(): Color;     // index 9
  static brightGreen(): Color;   // index 10
  static brightYellow(): Color;  // index 11
  static brightBlue(): Color;    // index 12
  static brightMagenta(): Color; // index 13
  static brightCyan(): Color;    // index 14
  static brightWhite(): Color;   // index 15
  static indexed(n: number): Color;  // 256 色 (0-255)
  static rgb(r: number, g: number, b: number): Color;

  equals(other: Color): boolean;
  toAnsi(isForeground: boolean): string;  // SGR 参数序列
}

type ColorKind = "default" | "named" | "index" | "rgb";
```

## TextStyle (from tui-render-pipeline.js — cT 类)

```typescript
class TextStyle {
  readonly foreground: Color;
  readonly background: Color;
  readonly bold: boolean;
  readonly italic: boolean;
  readonly underline: boolean;
  readonly strikethrough: boolean;
  readonly dim: boolean;

  static readonly NORMAL: TextStyle;  // 全默认

  constructor(options?: Partial<TextStyleOptions>);
  copyWith(options: Partial<TextStyleOptions>): TextStyle;
  merge(other: TextStyle): TextStyle;
  equals(other: TextStyle): boolean;
  toSgr(): string;   // 完整 SGR 参数序列
  diffSgr(previous: TextStyle): string;  // 差分 SGR（仅变化的属性）
}

interface TextStyleOptions {
  foreground: Color;
  background: Color;
  bold: boolean;
  italic: boolean;
  underline: boolean;
  strikethrough: boolean;
  dim: boolean;
}
```

## Cell (from tui-render-pipeline.js — 单元格概念)

```typescript
class Cell {
  /** 单个 grapheme cluster（可能为空字符串表示宽字符续位） */
  readonly char: string;
  /** 文本样式 */
  readonly style: TextStyle;
  /** 显示宽度 (1=普通, 2=CJK/Emoji 宽字符, 0=续位) */
  readonly width: number;

  static readonly EMPTY: Cell;  // 空格 + NORMAL 样式 + width 1

  constructor(char: string, style: TextStyle, width?: number);
  equals(other: Cell): boolean;
}
```
