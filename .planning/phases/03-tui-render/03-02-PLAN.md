## VT Low-Level Events (parser output)

```typescript
/** VT 解析器输出的低层事件 */
type VtEvent =
  | VtPrintEvent
  | VtCsiEvent
  | VtEscapeEvent
  | VtOscEvent
  | VtDcsEvent
  | VtApcEvent;

interface VtPrintEvent {
  type: "print";
  grapheme: string;         // Unicode grapheme cluster
}

interface VtCsiEvent {
  type: "csi";
  params: CsiParam[];       // 参数列表（含可选 subparams）
  intermediates: string;     // 中间字节 (如 "?" 或 " ")
  private_marker: string;    // 私有标记 (如 "?" "<")
  final: string;             // 终止字节 (如 "A" "m" "h")
}

interface CsiParam {
  value: number;             // 参数值 (-1 表示缺省)
  subparams?: number[];      // 子参数 (冒号分隔)
}

interface VtEscapeEvent {
  type: "escape";
  intermediates: string;
  final: string;
}

interface VtOscEvent {
  type: "osc";
  data: string;              // OSC 数据 (如 "52;c;base64data")
}

interface VtDcsEvent {
  type: "dcs";
  params: CsiParam[];
  intermediates: string;
  private_marker: string;
  final: string;
  data: string;              // DCS 数据
}

interface VtApcEvent {
  type: "apc";
  data: string;
}
```

## Input High-Level Events (for widgets)

```typescript
type InputEvent =
  | KeyEvent
  | MouseEvent
  | PasteEvent
  | FocusEvent
  | ResizeEvent;

interface KeyEvent {
  type: "key";
  key: string;               // 逻辑键名 ("Enter", "ArrowUp", "a", "A", "F1")
  modifiers: Modifiers;
}

interface MouseEvent {
  type: "mouse";
  x: number;                 // 列位置 (0-based)
  y: number;                 // 行位置 (0-based)
  button: MouseButton;
  action: MouseAction;
  modifiers: Modifiers;
}

interface PasteEvent {
  type: "paste";
  text: string;
}

interface FocusEvent {
  type: "focus";
  focused: boolean;
}

interface ResizeEvent {
  type: "resize";
  cols: number;
  rows: number;
}

interface Modifiers {
  shift: boolean;
  alt: boolean;
  ctrl: boolean;
  meta: boolean;
}

type MouseButton = "left" | "middle" | "right" | "none";
type MouseAction = "press" | "release" | "move" | "wheel_up" | "wheel_down";
```
