```typescript
/** 终端 UI 的盒约束模型，定义子节点可用的宽高范围 */
interface Size {
  readonly width: number;
  readonly height: number;
}

class BoxConstraints {
  readonly minWidth: number;   // 默认 0
  readonly maxWidth: number;   // 默认 Infinity
  readonly minHeight: number;  // 默认 0
  readonly maxHeight: number;  // 默认 Infinity

  constructor(options?: {
    minWidth?: number;
    maxWidth?: number;
    minHeight?: number;
    maxHeight?: number;
  });

  // -- 静态工厂 --
  static tight(width: number, height: number): BoxConstraints;
  static loose(width: number, height: number): BoxConstraints;
  static tightFor(options?: { width?: number; height?: number }): BoxConstraints;

  // -- 计算属性 --
  get hasBoundedWidth(): boolean;   // maxWidth !== Infinity
  get hasBoundedHeight(): boolean;  // maxHeight !== Infinity
  get hasTightWidth(): boolean;     // minWidth >= maxWidth
  get hasTightHeight(): boolean;    // minHeight >= maxHeight
  get isTight(): boolean;           // hasTightWidth && hasTightHeight
  get biggest(): Size;              // { width: maxWidth, height: maxHeight }
  get smallest(): Size;             // { width: minWidth, height: minHeight }

  // -- 方法 --
  constrain(width: number, height: number): Size;
  enforce(other: BoxConstraints): BoxConstraints;
  loosen(): BoxConstraints;
  tighten(options?: { width?: number; height?: number }): BoxConstraints;
  equals(other: BoxConstraints): boolean;
  toString(): string;
}
```

   - 构造函数接受可选 options 对象，缺省字段使用默认值
   - 构造函数断言: `minWidth <= maxWidth`, `minHeight <= maxHeight`, 所有值 `>= 0`

   **静态工厂:**
   - `static tight(width, height)` — min===max===给定值
   - `static loose(width, height)` — min=0, max=给定值
   - `static tightFor({ width?, height? })` — 仅指定维度设为 tight，其余 0..Infinity

   **计算属性:**
   - `hasBoundedWidth` → `maxWidth !== Infinity`
   - `hasBoundedHeight` → `maxHeight !== Infinity`
   - `hasTightWidth` → `minWidth >= maxWidth`
   - `hasTightHeight` → `minHeight >= maxHeight`
   - `isTight` → `hasTightWidth && hasTightHeight`
   - `biggest` → `{ width: maxWidth, height: maxHeight }`
   - `smallest` → `{ width: minWidth, height: minHeight }`

   **方法:**
   - `constrain(width, height)` — 返回 `{ width: clamp(width, min, max), height: clamp(height, min, max) }`
   - `enforce(other)` — 返回新约束: 每个维度取 `clamp(other.min/max, this.min, this.max)`
   - `loosen()` — 返回 `new BoxConstraints({ maxWidth: this.maxWidth, maxHeight: this.maxHeight })` (min 归零)
   - `tighten({ width?, height? })` — 指定维度变 tight (clamp 到当前 min/max 范围)
   - `equals(other)` — 四字段全等
   - `toString()` — 调试用格式化输出
