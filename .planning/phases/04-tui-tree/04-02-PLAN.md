```typescript
// types.ts — 共享类型定义
interface Position {
  readonly x: number;
  readonly y: number;
}

interface ParentData {
  detach(): void;
}

/** 布局/绘制管道调度器接口（Plan 04-07 PipelineOwner 实现） */
interface PipelineOwnerLike {
  requestLayout(node: RenderObject): void;
  requestPaint(node: RenderObject): void;
  removeFromQueues(node: RenderObject): void;
}

/** 构建调度器接口（Plan 04-07 BuildOwner 实现） */
interface BuildOwnerLike {
  scheduleBuildFor(element: Element): void;
}

// render-object.ts
abstract class RenderObject {
  // -- 树结构 --
  parent: RenderObject | undefined;
  readonly children: readonly RenderObject[];
  parentData: ParentData | undefined;

  // -- 脏标记 --
  readonly needsLayout: boolean;
  readonly needsPaint: boolean;
  readonly attached: boolean;
  readonly depth: number;

  // -- 子节点管理 --
  setupParentData(child: RenderObject): void;
  adoptChild(child: RenderObject): void;
  dropChild(child: RenderObject): void;
  removeAllChildren(): void;

  // -- 生命周期 --
  attach(): void;
  detach(): void;

  // -- 脏标记传播 --
  markNeedsLayout(): void;
  markNeedsPaint(): void;

  // -- 布局与绘制 (子类覆写) --
  abstract performLayout(): void;
  paint(screen: Screen, offsetX: number, offsetY: number): void;

  // -- 遍历 --
  visitChildren(visitor: (child: RenderObject) => void): void;

  // -- 清理 --
  dispose(): void;
}
```

创建 `packages/tui/src/tree/render-object.ts`:

1. 导出抽象类 `RenderObject`:
   **字段:**
   - `_parent: RenderObject | undefined`
   - `_children: RenderObject[]`
   - `_parentData: ParentData | undefined`
   - `_needsLayout: boolean = true` (初始需要布局)
   - `_needsPaint: boolean = true` (初始需要绘制)
   - `_attached: boolean = false`
   - `_depth: number = 0`

   **只读属性:**
   - `parent` → `_parent`
   - `children` → `_children` (返回只读数组)
   - `parentData` getter/setter (setter 同时 detach 旧数据)
   - `needsLayout` → `_needsLayout`
   - `needsPaint` → `_needsPaint`
   - `attached` → `_attached`
   - `depth` → `_depth`

   **子节点管理:**
   - `setupParentData(child)` — 默认空操作，子类可覆写初始化 parentData
   - `adoptChild(child)`:
     1. 设置 child._parent = this
     2. child._depth = this._depth + 1
     3. 调用 setupParentData(child)
     4. 添加到 _children
     5. 如果 attached，调用 child.attach()
     6. markNeedsLayout()
   - `dropChild(child)`:
     1. child.parentData?.detach()
     2. child._parentData = undefined
     3. 如果 attached，调用 child.detach()
     4. 从 _children 移除
     5. child._parent = undefined
     6. markNeedsLayout()
   - `removeAllChildren()` — 对每个 child 调用 dropChild

   **生命周期:**
   - `attach()`: _attached = true，递归 children.forEach(c => c.attach())
   - `detach()`: _attached = false，递归 children.forEach(c => c.detach())

   **脏标记:**
   - `markNeedsLayout()`:
     1. _needsLayout = true
     2. 如果有 parent，parent.markNeedsLayout() (向上传播)
     3. 如果没有 parent 且 attached，getPipelineOwner()?.requestLayout(this)
   - `markNeedsPaint()`:
     1. _needsPaint = true
     2. getPipelineOwner()?.requestPaint(this)

   **布局与绘制:**
   - `abstract performLayout(): void`
   - `paint(screen, offsetX, offsetY)` — 默认递归绘制 children (调用 child.paint)

   **遍历:**
   - `visitChildren(visitor)` — _children.forEach(visitor)

   **清理:**
   - `dispose()` — removeAllChildren()
