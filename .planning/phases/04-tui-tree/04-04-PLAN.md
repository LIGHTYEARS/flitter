```typescript
/** Widget 基类前向声明 (Plan 04-05 实现) */
interface Widget {
  key: Key | undefined;
  canUpdate(other: Widget): boolean;
  createElement(): Element;
}

interface Key {
  equals?(other: Key): boolean;
}

abstract class Element {
  widget: Widget;
  parent: Element | undefined;
  readonly children: readonly Element[];
  readonly dirty: boolean;
  readonly mounted: boolean;
  readonly depth: number;
  readonly renderObject: RenderObject | undefined;

  constructor(widget: Widget);

  // -- 树操作 --
  addChild(child: Element): void;
  removeChild(child: Element): void;
  removeAllChildren(): void;

  // -- 生命周期 --
  mount(parent: Element | undefined): void;
  unmount(): void;
  update(newWidget: Widget): void;

  // -- 重建 --
  markNeedsRebuild(): void;
  performRebuild(): void;

  // -- 查找 --
  findAncestorElementOfType(type: Function): Element | null;
  findRenderObject(): RenderObject | undefined;
}
```

1. 先定义 `Widget` 接口 (最小化前向声明，Plan 04-05 会实现完整 Widget 类):
   ```typescript
   export interface Widget {
     key: Key | undefined;
     canUpdate(other: Widget): boolean;
     createElement(): Element;
   }
   export interface Key {}
   ```

2. 导出抽象类 `Element`:

   **字段:**
   - `_widget: Widget`
   - `_parent: Element | undefined`
   - `_children: Element[] = []`
   - `_dirty: boolean = true` (初始需要构建)
   - `_mounted: boolean = false`
   - `_depth: number = 0`

   **只读属性:**
   - `widget` getter/setter
   - `parent` getter
   - `children` → 只读数组
   - `dirty` → `_dirty`
   - `mounted` → `_mounted`
   - `depth` → `_depth`
   - `renderObject` → `undefined` (基类默认，RenderObjectElement 覆写)

   **构造函数:**
   - `constructor(widget: Widget)` — 保存 widget 引用

   **树操作:**
   - `addChild(child)`: child._parent = this, child._depth = this._depth + 1, 添加到 _children
   - `removeChild(child)`: 从 _children 移除, child._parent = undefined
   - `removeAllChildren()`: 对每个 child 调用 removeChild

   **生命周期:**
   - `mount(parent)`:
     1. `_parent = parent`
     2. `_depth = parent ? parent.depth + 1 : 0`
     3. `_mounted = true`
   - `unmount()`:
     1. 递归 unmount children
     2. `_mounted = false`
     3. `_parent = undefined`
   - `update(newWidget)`:
     1. `_widget = newWidget`

   **重建:**
   - `markNeedsRebuild()`:
     1. 如果已经 dirty, return
     2. `_dirty = true`
     3. `getBuildOwner()?.scheduleBuildFor(this)`
   - `performRebuild()`:
     1. `_dirty = false`
     2. 子类覆写具体逻辑

   **查找:**
   - `findAncestorElementOfType(type)`: 向上遍历 parent 链，找到 `instanceof type` 的第一个 Element
   - `findRenderObject()`: 向下搜索第一个有 renderObject 的 Element

3. JSDoc 中文注释
