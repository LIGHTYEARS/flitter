```typescript
/** ComponentElement — 组合型 Element，持有一个子 Element */
abstract class ComponentElement extends Element {
  _child: Element | undefined;

  /** 子类覆写: 返回此位置的子 Widget (调用 widget.build) */
  abstract build(): Widget;

  /** 协调子 Element: 对比新旧 Widget 决定复用或重建 */
  performRebuild(): void;

  /** 核心协调算法 */
  updateChild(child: Element | undefined, newWidget: Widget | undefined): Element | undefined;
}

/** RenderObjectElement — 渲染型 Element，持有一个 RenderObject */
abstract class RenderObjectElement extends Element {
  _renderObject: RenderObject | undefined;
  get renderObject(): RenderObject | undefined;

  mount(parent: Element | undefined): void;
  update(newWidget: Widget): void;
  unmount(): void;

  /** 子类覆写: 创建 RenderObject */
  abstract createRenderObject(): RenderObject;
  /** 子类覆写: 用新 Widget 配置更新 RenderObject */
  abstract updateRenderObject(widget: Widget): void;

  /** 将 renderObject 插入父 RenderObject 的 children */
  insertRenderObjectChild(): void;
  /** 从父 RenderObject 中移除 */
  removeRenderObjectChild(): void;
}

/** 协调辅助: Widget 的 RenderObject 创建接口 */
interface RenderObjectWidget extends Widget {
  createRenderObject(): RenderObject;
  updateRenderObject(renderObject: RenderObject): void;
}
```

创建 `packages/tui/src/tree/component-element.ts`:

1. 导出抽象类 `ComponentElement extends Element`:

   **字段:**
   - `_child: Element | undefined`

   **核心方法:**
   - `abstract build(): Widget` — 子类覆写返回子 Widget

   - `performRebuild()`:
     1. 调用 `build()` 获取新子 Widget
     2. 调用 `updateChild(_child, newWidget)` 协调
     3. `_dirty = false`

   - `updateChild(child, newWidget)`:
     协调算法:
     1. 如果 `newWidget === undefined`:
        - 如果 child 存在: unmount child, removeChild
        - 返回 undefined
     2. 如果 child 存在且 `child.widget.canUpdate(newWidget)`:
        - `child.update(newWidget)` (复用)
        - 返回 child
     3. 否则:
        - 如果 child 存在: unmount child, removeChild (旧的不可复用)
        - 创建新 Element: `newWidget.createElement()`
        - mount 新 Element, addChild
        - 返回新 Element

   - 覆写 `mount(parent)`:
     1. `super.mount(parent)`
     2. `performRebuild()` (首次构建)

   - 覆写 `unmount()`:
     1. 如果 `_child`: unmount child, removeChild
     2. `super.unmount()`

2. JSDoc 中文注释

   **属性:**
   - `renderObject` getter → `_renderObject` (覆写 Element 的默认 undefined)

   **生命周期覆写:**
   - `mount(parent)`:
     1. `super.mount(parent)`
     2. `_renderObject = (this.widget as RenderObjectWidget).createRenderObject()`
     3. 将 renderObject 插入父的渲染树: `insertRenderObjectChild()`
     4. `_dirty = false`

   - `update(newWidget)`:
     1. `super.update(newWidget)`
     2. `(this.widget as RenderObjectWidget).updateRenderObject(_renderObject!)`
     3. `_dirty = false`

   - `unmount()`:
     1. `removeRenderObjectChild()`
     2. `_renderObject = undefined`
     3. `super.unmount()`

   **渲染树桥接:**
   - `insertRenderObjectChild()`:
     1. 向上查找最近的祖先 RenderObjectElement
     2. 将 `_renderObject` adoptChild 到祖先的 renderObject
   - `removeRenderObjectChild()`:
     1. 向上查找最近的祖先 RenderObjectElement
     2. 从祖先的 renderObject 中 dropChild

   - `findAncestorRenderObjectElement()`: 工具方法，向上遍历 parent 找第一个 RenderObjectElement

3. 同时导出 `SingleChildRenderObjectElement extends RenderObjectElement`:
   - 管理单个子 Element (类似 ComponentElement 但带 RenderObject)
   - `_child: Element | undefined`
   - `performRebuild()`: 协调单个子 Element
