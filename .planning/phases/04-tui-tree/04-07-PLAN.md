```typescript
// -- StatelessWidget --
abstract class StatelessWidget extends Widget {
  abstract build(context: BuildContext): Widget;
  createElement(): StatelessElement;
}

class StatelessElement extends ComponentElement {
  build(): Widget; // 委托给 (widget as StatelessWidget).build(this)
}

// -- StatefulWidget --
abstract class StatefulWidget extends Widget {
  abstract createState(): State;
  createElement(): StatefulElement;
}

class StatefulElement extends ComponentElement {
  _state: State;

  mount(parent): void;  // 创建 state, initState, build
  update(newWidget): void;  // state.didUpdateWidget
  unmount(): void;  // state.dispose
  build(): Widget;  // 委托给 state.build
}

// -- State --
abstract class State<T extends StatefulWidget = StatefulWidget> {
  widget: T;
  context: BuildContext;
  mounted: boolean;

  initState(): void;           // 初始化，首次 mount 调用
  didUpdateWidget(oldWidget: T): void;  // Widget 配置变化时调用
  dispose(): void;             // unmount 时清理
  setState(fn?: () => void): void;  // 标记需要重建
  abstract build(context: BuildContext): Widget;  // 返回子 Widget
}

// -- BuildContext (简化版) --
interface BuildContext {
  readonly widget: Widget;
  readonly element: Element;
  findAncestorStateOfType<S extends State>(type: new (...args: any[]) => S): S | null;
  findRenderObject(): RenderObject | undefined;
}
```

创建 `packages/tui/src/tree/stateful-widget.ts`:

1. 导出抽象类 `State<T extends StatefulWidget>`:
   **字段:**
   - `_widget: T`
   - `_element: StatefulElement | undefined`
   - `_mounted: boolean = false`

   **属性:**
   - `widget` getter → `_widget`
   - `context` getter → `_element` 的 BuildContext
   - `mounted` getter → `_mounted`

   **生命周期方法 (子类可覆写):**
   - `initState()` — 空默认实现
   - `didUpdateWidget(oldWidget: T)` — 空默认实现
   - `dispose()` — 空默认实现

   **核心方法:**
   - `setState(fn?)`:
     1. 如果 `!_mounted`，抛出错误 "setState called after dispose"
     2. 如果 fn 存在，同步执行 `fn()`
     3. `_element!.markNeedsRebuild()`
   - `abstract build(context: BuildContext): Widget`

   **内部方法:**
   - `_mount(widget: T, element: StatefulElement)`: 设置 _widget, _element, _mounted=true
   - `_update(newWidget: T)`: 保存旧 widget, _widget = newWidget, 调用 didUpdateWidget(oldWidget)
   - `_unmount()`: _mounted=false, 调用 dispose()

2. 导出抽象类 `StatefulWidget extends Widget`:
   - `abstract createState(): State`
   - `createElement()` → `new StatefulElement(this)`

3. 导出类 `StatefulElement extends ComponentElement`:
   - `_state: State`

   - `mount(parent)`:
     1. `_state = (widget as StatefulWidget).createState()`
     2. `_state._mount(widget, this)`
     3. `_state.initState()`
     4. `super.mount(parent)` (触发 performRebuild → build)

   - `build()` → `_state.build(this._createBuildContext())`

   - `update(newWidget)`:
     1. `_state._update(newWidget)` (内部调用 didUpdateWidget)
     2. `super.update(newWidget)`
     3. `_dirty = true`
     4. `performRebuild()`

   - `unmount()`:
     1. `_state._unmount()` (内部调用 dispose)
     2. `super.unmount()`

   - `get state()` → `_state` (供 GlobalKey.currentState 使用)

4. JSDoc 中文注释
