<!-- Row/Column/Flexible/Expanded Widget + MultiChildRenderObjectElement -->

```typescript
import { Widget } from "../tree/widget.js";
import { Element } from "../tree/element.js";
import { RenderObjectElement, RenderObjectWidget } from "../tree/render-object-element.js";
import { RenderObject } from "../tree/render-object.js";
import {
  RenderFlex, MainAxisAlignment, CrossAxisAlignment, MainAxisSize,
  FlexFit, FlexParentData,
} from "./flex.js";

/**
 * 多子节点渲染对象元素。
 *
 * 管理一组子元素，并将对应的渲染对象同步到父渲染对象的子节点列表中。
 * Row/Column/Stack 等多子节点 Widget 共用此元素类型。
 */
class MultiChildRenderObjectElement extends RenderObjectElement {
  /** 子元素列表 */
  protected _childElements: Element[];

  mount(parent?: Element): void;
  update(newWidget: Widget): void;
  unmount(): void;
}

/**
 * 水平排列 Widget。
 *
 * @example
 * ```ts
 * new Row({
 *   mainAxisAlignment: "spaceBetween",
 *   children: [child1, child2, child3],
 * })
 * ```
 */
class Row extends Widget implements RenderObjectWidget {
  children: Widget[];
  mainAxisAlignment: MainAxisAlignment;
  crossAxisAlignment: CrossAxisAlignment;
  mainAxisSize: MainAxisSize;

  constructor(options: {
    key?: Key;
    children: Widget[];
    mainAxisAlignment?: MainAxisAlignment;
    crossAxisAlignment?: CrossAxisAlignment;
    mainAxisSize?: MainAxisSize;
  });

  createElement(): MultiChildRenderObjectElement;
  createRenderObject(): RenderFlex;
  updateRenderObject(renderObject: RenderObject): void;
}

/**
 * 垂直排列 Widget，与 Row 相同但 direction="vertical"。
 */
class Column extends Widget implements RenderObjectWidget {
  // 同 Row，但 createRenderObject 使用 direction: "vertical"
}

/**
 * 弹性 Widget，为子节点标记 flex 因子。
 *
 * @example
 * ```ts
 * new Flexible({ flex: 2, child: someWidget })
 * ```
 */
class Flexible extends Widget implements RenderObjectWidget {
  child: Widget;
  flex: number;   // 默认 1
  fit: FlexFit;   // 默认 "loose"

  constructor(options: {
    key?: Key;
    child: Widget;
    flex?: number;
    fit?: FlexFit;
  });

  createElement(): Element;
  createRenderObject(): RenderObject;
  updateRenderObject(renderObject: RenderObject): void;
}

/**
 * 扩展 Widget，Flexible 的便捷子类，fit 固定为 "tight"。
 *
 * @example
 * ```ts
 * new Expanded({ child: someWidget })
 * new Expanded({ flex: 2, child: someWidget })
 * ```
 */
class Expanded extends Flexible {
  constructor(options: {
    key?: Key;
    child: Widget;
    flex?: number;
  });
}
```

**1. 创建 `packages/tui/src/widgets/multi-child-render-object-element.ts`:**

导出 `MultiChildRenderObjectElement extends RenderObjectElement`:

- `_childElements: Element[] = []`

- `mount(parent?)`:
  1. 调用 `super.mount(parent)` — 创建渲染对象并插入渲染树
  2. 获取 widget 的 `children` 列表 (类型断言)
  3. 遍历 children: 为每个 childWidget 调用 `childWidget.createElement()`，mount 子元素，addChild，并将子元素的渲染对象 adoptChild 到当前渲染对象

- `update(newWidget)`:
  1. 调用 `super.update(newWidget)`
  2. 获取新旧 children 列表
  3. 简化协调: 卸载所有旧子元素 (unmount + removeChild + dropChild 渲染对象)，然后为新 children 重新创建并挂载
  4. 注意: 简化实现先用全量替换，后续可优化为 key-based diff

- `unmount()`:
  1. 遍历 `_childElements`: unmount 子元素，dropChild 渲染对象，removeChild
  2. 清空 `_childElements`
  3. 调用 `super.unmount()`

**2. 创建 `packages/tui/src/widgets/row.ts`:**

导出 `Row extends Widget` 同时实现 `RenderObjectWidget` 接口:

- 构造函数接受 `{ key?, children, mainAxisAlignment?, crossAxisAlignment?, mainAxisSize? }`
- 存储 children 和布局属性
- `createElement()` → `new MultiChildRenderObjectElement(this)`
- `createRenderObject()` → `new RenderFlex({ direction: "horizontal", mainAxisAlignment, crossAxisAlignment, mainAxisSize })`
- `updateRenderObject(renderObject)` → 更新 RenderFlex 的属性

- 构造函数: `{ key?, child, flex=1, fit="loose" }`

- 实现方式: Flexible 的 createRenderObject 创建一个透传包装器，或直接作为 ParentDataWidget 在挂载时设置父节点的 FlexParentData
