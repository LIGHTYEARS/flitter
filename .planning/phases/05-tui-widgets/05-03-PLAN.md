<!-- Stack/Positioned 层叠布局，基于 Flutter Stack 模式设计 -->

```typescript
import { ParentData } from "../tree/types.js";
import { RenderBox } from "../tree/render-box.js";
import { RenderObject } from "../tree/render-object.js";
import { Widget } from "../tree/widget.js";

/** Stack 对齐方式 */
type StackAlignment =
  | "topLeft" | "topCenter" | "topRight"
  | "centerLeft" | "center" | "centerRight"
  | "bottomLeft" | "bottomCenter" | "bottomRight";

/**
 * Stack 子节点的定位父数据。
 *
 * @example
 * ```ts
 * const pd = new StackParentData();
 * pd.left = 10;
 * pd.top = 5;
 * pd.isPositioned = true;
 * ```
 */
class StackParentData extends ParentData {
  /** 距左边界偏移 */
  left: number | undefined;
  /** 距上边界偏移 */
  top: number | undefined;
  /** 距右边界偏移 */
  right: number | undefined;
  /** 距下边界偏移 */
  bottom: number | undefined;
  /** 是否为定位子节点 */
  isPositioned: boolean;
}

/**
 * 层叠布局渲染对象。
 *
 * 布局算法:
 * 1. 用 loosen'd 约束布局所有非定位子节点
 * 2. 从非定位子节点中取最大宽高作为 Stack 自身尺寸
 * 3. 如果无非定位子节点，使用父约束的最大值
 * 4. 布局定位子节点: 根据 left/right 计算宽度约束，top/bottom 计算高度约束
 * 5. 根据 alignment 设置非定位子节点偏移
 * 6. 根据 left/top 设置定位子节点偏移
 *
 * @example
 * ```ts
 * const stack = new RenderStack({ alignment: "center" });
 * ```
 */
class RenderStack extends RenderBox {
  /** 非定位子节点的默认对齐方式 */
  alignment: StackAlignment;

  constructor(options?: {
    alignment?: StackAlignment; // 默认 "topLeft"
  });

  setupParentData(child: RenderObject): void;
  performLayout(): void;
}

/**
 * 层叠布局 Widget。
 *
 * @example
 * ```ts
 * new Stack({
 *   alignment: "center",
 *   children: [
 *     backgroundWidget,
 *     new Positioned({ top: 0, right: 0, child: closeButton }),
 *   ],
 * })
 * ```
 */
class Stack extends Widget {
  children: Widget[];
  alignment: StackAlignment;

  constructor(options: {
    key?: Key;
    children: Widget[];
    alignment?: StackAlignment;
  });

  createElement(): MultiChildRenderObjectElement;
  createRenderObject(): RenderStack;
  updateRenderObject(renderObject: RenderObject): void;
}

/**
 * 定位子节点 Widget，在 Stack 中精确放置。
 *
 * 至少一个方向的约束 (left/right 或 top/bottom) 应被指定。
 * 同时指定 left 和 right 时，子节点宽度 = Stack 宽度 - left - right。
 * 同时指定 top 和 bottom 时，子节点高度 = Stack 高度 - top - bottom。
 *
 * @example
 * ```ts
 * new Positioned({ left: 10, top: 5, child: someWidget })
 * new Positioned({ left: 0, right: 0, child: fullWidthWidget })
 * ```
 */
class Positioned extends Widget {
  child: Widget;
  left: number | undefined;
  top: number | undefined;
  right: number | undefined;
  bottom: number | undefined;

  constructor(options: {
    key?: Key;
    child: Widget;
    left?: number;
    top?: number;
    right?: number;
    bottom?: number;
  });

  createElement(): Element;
  createRenderObject(): RenderObject;
  updateRenderObject(renderObject: RenderObject): void;
}
```

创建 `packages/tui/src/widgets/stack.ts`:

1. 导出 `StackAlignment` 类型别名 — 9 种对齐位置

2. 导出 `StackParentData extends ParentData`:
   - `left/top/right/bottom: number | undefined = undefined`
   - `isPositioned: boolean = false`

3. 导出 `RenderStack extends RenderBox`:

   - `alignment: StackAlignment` (默认 `"topLeft"`)

   - `setupParentData(child)`:
     如果 `child.parentData` 不是 `StackParentData` 实例，则设置 `child.parentData = new StackParentData()`

   - **performLayout()**:

     **第一阶段: 布局非定位子节点**
     1. 遍历 children，识别定位 (`isPositioned=true`) 和非定位子节点
     2. 用 `constraints.loosen()` 布局每个非定位子节点
     3. 记录最大 width/height 作为 stackWidth/stackHeight

     **确定 Stack 自身尺寸**
     4. 如果有非定位子节点: `size = constraints.constrain(stackWidth, stackHeight)`
     5. 如果全部是定位子节点: `size = constraints.biggest`（如果有界），否则 `size = constraints.smallest`

     **第二阶段: 布局定位子节点**
     6. 对每个定位子节点的 StackParentData:
        - 计算宽度约束:
          - left 和 right 都有值 → `tight(stackWidth - left - right)`
          - 否则 → `loose(stackWidth)`
        - 计算高度约束:
          - top 和 bottom 都有值 → `tight(stackHeight - top - bottom)`
          - 否则 → `loose(stackHeight)`
        - 用计算出的约束布局子节点

     **第三阶段: 定位所有子节点**
     7. 非定位子节点 → 根据 `alignment` 计算偏移:
        - topLeft → (0, 0)
        - topCenter → ((stackW - childW) / 2, 0)
        - topRight → (stackW - childW, 0)
        - centerLeft → (0, (stackH - childH) / 2)
        - center → ((stackW - childW) / 2, (stackH - childH) / 2)
        - centerRight → (stackW - childW, (stackH - childH) / 2)
        - bottomLeft → (0, stackH - childH)
        - bottomCenter → ((stackW - childW) / 2, stackH - childH)
        - bottomRight → (stackW - childW, stackH - childH)

     8. 定位子节点 → 根据 StackParentData 设置偏移:
        - x = left ?? (right !== undefined ? stackW - right - childW : 0)
        - y = top ?? (bottom !== undefined ? stackH - bottom - childH : 0)

4. 导出 `Stack extends Widget` 实现 `RenderObjectWidget`:
   - 构造函数: `{ key?, children, alignment? }`
   - `createElement()` → `new MultiChildRenderObjectElement(this)`
   - `createRenderObject()` → `new RenderStack({ alignment })`
   - `updateRenderObject()` → 更新 alignment

5. 导出 `Positioned extends Widget` 实现 `RenderObjectWidget`:
   - 构造函数: `{ key?, child, left?, top?, right?, bottom? }`
   - 核心行为: 在 mount 时将 left/top/right/bottom 写入父节点的 StackParentData 并设置 `isPositioned = true`
   - 实现方式: createElement 返回的元素在 mount 时设置 parentData

6. JSDoc 中文注释: 每个类/方法含功能说明 + 使用示例
