```typescript
import { RenderBox } from "../tree/render-box.js";
import { RenderObject } from "../tree/render-object.js";
import { Widget } from "../tree/widget.js";
import { StatelessWidget, BuildContext } from "../tree/stateless-widget.js";
import { BoxConstraints } from "../tree/constraints.js";

/**
 * 不可变的四方向边距值对象。
 *
 * @example
 * ```ts
 * EdgeInsets.all(8)             // 四边各 8
 * EdgeInsets.symmetric({ horizontal: 16, vertical: 8 })
 * EdgeInsets.only({ left: 4, top: 2 })
 * ```
 */
class EdgeInsets {
  readonly left: number;
  readonly top: number;
  readonly right: number;
  readonly bottom: number;

  private constructor(left: number, top: number, right: number, bottom: number);

  /** 四边相同间距 */
  static all(value: number): EdgeInsets;
  /** 水平/垂直对称间距 */
  static symmetric(opts?: { horizontal?: number; vertical?: number }): EdgeInsets;
  /** 仅水平间距 (左右相同) */
  static horizontal(value: number): EdgeInsets;
  /** 仅垂直间距 (上下相同) */
  static vertical(value: number): EdgeInsets;
  /** 分别指定各方向间距 */
  static only(opts?: { left?: number; top?: number; right?: number; bottom?: number }): EdgeInsets;
  /** 零间距 */
  static readonly zero: EdgeInsets;

  /** 水平方向总间距 (left + right) */
  get horizontal(): number;
  /** 垂直方向总间距 (top + bottom) */
  get vertical(): number;

  /** 值相等比较 */
  equals(other: EdgeInsets): boolean;
  toString(): string;
}

/**
 * 内边距渲染对象。
 *
 * 通过 deflate 约束缩小子节点可用空间，并将子节点偏移 (padding.left, padding.top)。
 */
class RenderPadding extends RenderBox {
  padding: EdgeInsets;

  constructor(padding: EdgeInsets);

  /**
   * 布局算法:
   * 1. deflate: 将约束的 max 减去 padding，min clamp 到 0
   * 2. 布局唯一子节点
   * 3. 自身 size = 子节点 size + padding
   * 4. 子节点 offset = (padding.left, padding.top)
   */
  performLayout(): void;
}

/**
 * 内边距 Widget。
 */
class Padding extends Widget {
  padding: EdgeInsets;
  child: Widget;

  constructor(options: { key?: Key; padding: EdgeInsets; child: Widget });

  createElement(): RenderObjectElement;
  createRenderObject(): RenderPadding;
  updateRenderObject(renderObject: RenderObject): void;
}

/**
 * 固定尺寸渲染对象。
 *
 * 对指定维度施加紧约束，未指定的维度传递父约束。
 */
class RenderSizedBox extends RenderBox {
  requestedWidth: number | undefined;
  requestedHeight: number | undefined;

  constructor(width?: number, height?: number);

  /**
   * 布局算法:
   * 1. 构造子约束:
   *    - 指定 width → tight(width)，否则用父约束的 min/maxWidth
   *    - 指定 height → tight(height)，否则用父约束的 min/maxHeight
   * 2. 如果有子节点: 用子约束布局，自身 size 取子节点 size
   * 3. 如果无子节点: 自身 size 取约束 constrain(width ?? 0, height ?? 0)
   * 4. 子节点 offset = (0, 0)
   */
  performLayout(): void;
}

/**
 * 固定尺寸 Widget。
 */
class SizedBox extends Widget {
  width: number | undefined;
  height: number | undefined;
  child: Widget | undefined;

  constructor(options: { key?: Key; width?: number; height?: number; child?: Widget });

  createElement(): RenderObjectElement;
  createRenderObject(): RenderSizedBox;
  updateRenderObject(renderObject: RenderObject): void;
}

/**
 * 便捷容器 Widget，组合 Padding + SizedBox。
 *
 * @example
 * ```ts
 * new Container({
 *   padding: EdgeInsets.all(8),
 *   width: 40,
 *   height: 10,
 *   child: someWidget,
 * })
 * ```
 */
class Container extends StatelessWidget {
  padding: EdgeInsets | undefined;
  width: number | undefined;
  height: number | undefined;
  child: Widget | undefined;

  constructor(options?: {
    key?: Key;
    padding?: EdgeInsets;
    width?: number;
    height?: number;
    child?: Widget;
  });

  /**
   * build 策略:
   * 1. 从 child 开始
   * 2. 如果有 width 或 height → 包装 SizedBox
   * 3. 如果有 padding → 包装 Padding
   * 4. 返回最外层 Widget
   */
  build(context: BuildContext): Widget;
}
```

**2. 创建 `packages/tui/src/widgets/padding.ts`:**

导出 `RenderPadding extends RenderBox`:
- 构造函数: `(padding: EdgeInsets)`
- `performLayout()`:
  1. 从 `this._constraints!` 获取约束
  2. deflate: `deflatedConstraints = new BoxConstraints({ minWidth: max(0, c.minWidth - h), maxWidth: max(0, c.maxWidth - h), minHeight: max(0, c.minHeight - v), maxHeight: max(0, c.maxHeight - v) })`，其中 h=padding.horizontal, v=padding.vertical
  3. 如果有子节点 (children[0]): 用 deflated 约束布局，`this.size = { width: child.size.width + h, height: child.size.height + v }` 用 constraints.constrain 夹紧
  4. 子节点 `offset = { x: padding.left, y: padding.top }`
  5. 无子节点: `size = constraints.constrain(h, v)`

导出 `Padding extends Widget` 实现 `RenderObjectWidget`:
- 构造函数: `{ key?, padding, child }`
- `createElement()` → `new SingleChildRenderObjectElement(this)` (创建一个简单的单子节点版本，或复用 RenderObjectElement)
- `createRenderObject()` → `new RenderPadding(padding)`
- `updateRenderObject()` → 更新 padding

**3. 创建 `packages/tui/src/widgets/sized-box.ts`:**

导出 `RenderSizedBox extends RenderBox`:
- 构造函数: `(width?: number, height?: number)`
- `performLayout()`:
  1. 构造子约束:
     - minW = width !== undefined ? width : constraints.minWidth
     - maxW = width !== undefined ? width : constraints.maxWidth
     - minH = height !== undefined ? height : constraints.minHeight
     - maxH = height !== undefined ? height : constraints.maxHeight
     - `childConstraints = new BoxConstraints({ minWidth: minW, maxWidth: maxW, minHeight: minH, maxHeight: maxH })`
  2. 有子节点: `children[0].layout(childConstraints)`，`this.size = children[0].size`
  3. 无子节点: `this.size = constraints.constrain(width ?? 0, height ?? 0)`
  4. 子节点 `offset = { x: 0, y: 0 }`

导出 `SizedBox extends Widget` 实现 `RenderObjectWidget`:
- 构造函数: `{ key?, width?, height?, child? }`
- `createRenderObject()` → `new RenderSizedBox(width, height)`
- `updateRenderObject()` → 更新 requestedWidth/requestedHeight

**4. 创建 `packages/tui/src/widgets/container.ts`:**

导出 `Container extends StatelessWidget`:
- 构造函数: `{ key?, padding?, width?, height?, child? }`
- `build(context)`:
  1. `let current = this.child`
  2. 如果 `width !== undefined || height !== undefined`:
     - `current = new SizedBox({ width, height, child: current })`
  3. 如果 `padding !== undefined`:
     - `current = new Padding({ padding, child: current! })`
  4. 返回 current (如果全部为空则返回 SizedBox({ width: 0, height: 0 }))
