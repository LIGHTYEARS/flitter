```typescript
import { TextStyle } from "../screen/text-style.js";
import { RenderBox } from "../tree/render-box.js";
import { Widget } from "../tree/widget.js";
import { StatelessWidget, BuildContext } from "../tree/stateless-widget.js";
import { RenderObjectWidget } from "../tree/render-object-element.js";
import type { Screen } from "../screen/screen.js";
import type { BoxConstraints, Size } from "../tree/constraints.js";
import type { Element } from "../tree/element.js";

// ── text-span.ts ──

/** 样式化文本树节点，支持嵌套子节点构建富文本 */
class TextSpan {
  readonly text?: string;
  readonly style?: TextStyle;
  readonly children?: TextSpan[];

  constructor(options: { text?: string; style?: TextStyle; children?: TextSpan[] });

  /** 递归拼接所有文本节点为纯文本字符串 */
  toPlainText(): string;

  /**
   * 深度优先遍历所有 TextSpan 节点。
   * visitor 返回 false 时停止遍历。
   */
  visitTextSpan(visitor: (span: TextSpan) => boolean): boolean;

  /** 递归值相等比较 */
  equals(other: TextSpan): boolean;
}

// ── rich-text.ts ──

/** 文本段落渲染对象，负责文本测量和绘制 */
class RenderParagraph extends RenderBox {
  textSpan: TextSpan;

  constructor(textSpan: TextSpan);

  /**
   * 布局计算：
   * 1. 通过 visitTextSpan 收集所有文本片段
   * 2. 使用 charWidth() 逐字符计算宽度
   * 3. 在约束宽度处换行
   * 4. 设置 size = { width: 最大行宽, height: 行数 }
   */
  performLayout(): void;

  /**
   * 绘制：
   * 1. 遍历已排版的行
   * 2. 对每个字符调用 screen.writeChar()
   * 3. 正确处理 CJK 双宽字符的占位
   */
  performPaint(screen: Screen, offsetX: number, offsetY: number): void;
}

/** 富文本 Widget，创建 RenderParagraph */
class RichText extends Widget implements RenderObjectWidget {
  readonly text: TextSpan;

  constructor(options: { text: TextSpan; key?: Key });

  createElement(): Element;
  createRenderObject(): RenderParagraph;
  updateRenderObject(renderObject: RenderParagraph): void;
}

// ── text.ts ──

/** 便捷文本 Widget，单样式纯文本 */
class Text extends StatelessWidget {
  readonly data: string;
  readonly style?: TextStyle;

  constructor(options: { data: string; style?: TextStyle; key?: Key });

  /** 构建 RichText({ text: new TextSpan({ text: data, style }) }) */
  build(context: BuildContext): Widget;
}
```

   - 导出 `RenderParagraph extends RenderBox`:
     - `textSpan` 属性
     - 内部维护排版结果: `_lines: Array<Array<{ char: string; style: TextStyle; width: number }>>`
     - `performLayout()`:
       a. 获取约束最大宽度 `maxWidth = constraints.maxWidth`
       b. visitTextSpan 收集所有 { char, style, width } 片段
       c. 使用 `charWidth()` 计算每个 grapheme 的宽度
       d. 按 maxWidth 换行（如果字符宽度会超出当前行，开始新行）
       e. 设置 `size = { width: 实际最大行宽, height: 行数 }`
     - `performPaint(screen, offsetX, offsetY)`:
       a. 遍历 _lines
       b. 对每行中的每个字符调用 `screen.writeChar(x, y, char, style, width)`
       c. x 递增 width（CJK 字符递增 2）

   - 导出 `RichText extends Widget` 实现 `RenderObjectWidget`:
     - `text: TextSpan` 属性
     - `createElement()` → 创建 `SingleChildRenderObjectElement`（使用简化版 LeafRenderObjectElement）
     - `createRenderObject()` → `new RenderParagraph(this.text)`
     - `updateRenderObject(ro)` → 更新 `ro.textSpan = this.text`，标记需要重新布局

3. 创建 `packages/tui/src/widgets/text.ts`:
   - 导出 `Text extends StatelessWidget`:
     - `data: string`, `style?: TextStyle`
     - `build(context)` → `new RichText({ text: new TextSpan({ text: data, style }) })`

4. 使用 graphemeSegments + charWidth 处理文本宽度（从 text/char-width.ts 导入）
