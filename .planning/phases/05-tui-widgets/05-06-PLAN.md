```typescript
/**
 * 检测码点是否为 CJK（中日韩）双宽字符。
 *
 * 包含范围:
 * - CJK 统一表意文字: U+4E00-9FFF, U+3400-4DBF, U+20000-2A6DF, U+2A700-2B73F,
 *   U+2B740-2B81F, U+2B820-2CEAF, U+2CEB0-2EBEF, U+30000-3134F
 * - 谚文音节: U+AC00-D7AF
 * - 平假名: U+3040-309F
 * - 片假名: U+30A0-30FF, U+31F0-31FF
 * - 全角字符: U+FF01-FF60, U+FFE0-FFE6
 * - CJK 标点: U+3000-303F
 * - CJK 兼容: U+FE30-FE4F, U+FE50-FE6F
 * - 区域指示符: U+1F1E6-1F1FF
 * - 方括号: U+2329-232A
 */
function isCjk(codePoint: number): boolean;

/**
 * 检测码点是否为零宽字符。
 *
 * 包含: ZWJ (U+200D), ZWNJ (U+200C), BOM (U+FEFF),
 * 软连字符 (U+00AD), 零宽空格 (U+200B),
 * 左/右标记 (U+200E-200F), 格式控制字符 (U+2060-2069),
 * 变体选择符 (U+FE00-FE0F), 组合标记 (U+0300-036F)
 */
function isZeroWidth(codePoint: number): boolean;

/**
 * 单个码点的显示宽度。
 *
 * @returns 0 (零宽), 1 (普通), 或 2 (CJK/全角)
 */
function codePointWidth(codePoint: number): number;

/**
 * 将文本分割为字素簇数组。
 *
 * 使用 `Intl.Segmenter("en", { granularity: "grapheme" })` 进行分割。
 */
function graphemeSegments(text: string): string[];

/**
 * 单个字素簇的显示宽度（带缓存）。
 *
 * 对于单码点字素，直接使用 codePointWidth。
 * 对于多码点字素（如 ZWJ 序列），按组成码点的最大宽度计算。
 * 结果使用 Map 缓存以提升性能。
 */
function charWidth(grapheme: string): number;

/**
 * 整个字符串的总显示宽度。
 *
 * 将文本分割为字素簇，对每个字素调用 charWidth 后求和。
 */
function textWidth(text: string): number;
```
