```typescript
// ── emoji.ts ──

/**
 * 检测码点是否为 Emoji。
 *
 * 包含范围:
 * - 表情符号: U+1F600-1F64F
 * - 杂项符号与象形: U+1F300-1F5FF
 * - 交通与地图符号: U+1F680-1F6FF
 * - 补充符号与象形: U+1F900-1F9FF
 * - 符号与象形扩展 A: U+1FA00-1FA6F
 * - 符号与象形扩展 B: U+1FA70-1FAFF
 * - 丁巴特: U+2702-27B0
 * - 杂项符号: U+2600-26FF
 * - 杂项技术符号 (部分): U+2300-23FF (部分子范围)
 * - 封闭字母数字补充: U+1F100-1F1FF
 * - Keycap 数字: U+0023, U+002A, U+0030-0039 (需配合 U+20E3)
 */
function isEmoji(codePoint: number): boolean;

/**
 * 检测码点是否有默认 Emoji 呈现方式。
 *
 * 具有默认 Emoji 呈现的码点在没有变体选择符时以 Emoji 形式显示。
 * 大部分在 U+1F000+ 范围内的 Emoji 都有默认 Emoji 呈现。
 */
function isEmojiPresentation(codePoint: number): boolean;

// ── char-width.ts 更新 ──

/**
 * 更新 codePointWidth 以处理 Emoji:
 * - 默认 Emoji 呈现 → 2
 * - 非默认 Emoji 呈现的 Emoji 码点 → 1 (需 VS16 才为 2)
 *
 * 更新 charWidth 以处理 Emoji 字素簇:
 * - 包含 VS15 (U+FE0E) → 1 (文本呈现)
 * - 包含 VS16 (U+FE0F) → 2 (Emoji 呈现)
 * - ZWJ 序列 (包含 U+200D) → 2
 * - 肤色修饰符 (U+1F3FB-1F3FF) → 组合后字素宽度 2
 * - 旗帜序列 (区域指示符对) → 2
 */
```
