```typescript
import { Color } from "../screen/color.js";
import { Widget } from "../tree/widget.js";
import { StatelessWidget, BuildContext } from "../tree/stateless-widget.js";

// ── color-scheme.ts ──

/** AppColorScheme 构造选项 */
interface AppColorSchemeOptions {
  foreground: Color;
  mutedForeground: Color;
  background: Color;
  cursor: Color;
  primary: Color;
  secondary: Color;
  accent: Color;
  border: Color;
  success: Color;
  warning: Color;
  info: Color;
  destructive: Color;
  selection: Color;
  copyHighlight: Color;
  tableBorder: Color;
}

/** 语义化颜色方案，持有 15 个颜色角色 */
class AppColorScheme {
  readonly foreground: Color;       // 主要文本色
  readonly mutedForeground: Color;  // 次要/暗淡文本色
  readonly background: Color;       // 背景色
  readonly cursor: Color;           // 光标色
  readonly primary: Color;          // 主要强调色
  readonly secondary: Color;        // 次要强调色
  readonly accent: Color;           // 装饰/高亮色
  readonly border: Color;           // 边框色
  readonly success: Color;          // 成功状态色
  readonly warning: Color;          // 警告状态色
  readonly info: Color;             // 信息状态色
  readonly destructive: Color;      // 危险/删除色
  readonly selection: Color;        // 选区背景色
  readonly copyHighlight: Color;    // 复制高亮色
  readonly tableBorder: Color;      // 表格边框色

  constructor(options: Partial<AppColorSchemeOptions>);

  /** 暗色主题默认配色 */
  static default(): AppColorScheme;

  /**
   * 从 RGB 配置创建自定义配色方案。
   * config 的 key 为颜色角色名，value 为 { r, g, b } 对象。
   */
  static fromRgb(config: Record<string, { r: number; g: number; b: number }>): AppColorScheme;

  /** 创建部分覆盖的新实例 */
  copyWith(overrides: Partial<AppColorSchemeOptions>): AppColorScheme;

  /** 值相等比较 */
  equals(other: AppColorScheme): boolean;
}

// ── theme.ts ──

/** 主题数据，持有配色方案 */
interface ThemeData {
  readonly colorScheme: AppColorScheme;
}

/**
 * 主题 Widget，包裹子 Widget 并提供 ThemeData。
 *
 * Phase 5 简化实现：Theme.of(context) 使用全局默认值。
 * InheritedWidget 正式实现延后到 Phase 6。
 */
class Theme extends StatelessWidget {
  readonly data: ThemeData;
  readonly child: Widget;

  constructor(options: { data: ThemeData; child: Widget; key?: Key });

  /**
   * 获取当前主题数据。
   *
   * Phase 5: 如果找不到祖先 Theme，返回全局默认暗色主题。
   * Phase 6: 将通过 InheritedWidget 向上查找最近的 Theme。
   */
  static of(context: BuildContext): ThemeData;

  /** 创建暗色主题 */
  static dark(): ThemeData;

  /** 创建亮色主题 */
  static light(): ThemeData;

  /** 创建默认主题 Widget */
  static withDefault(options: { child: Widget }): Theme;

  build(context: BuildContext): Widget;
}
```
