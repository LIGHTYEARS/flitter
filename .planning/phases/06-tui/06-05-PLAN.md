```typescript
export type MarkdownNodeType =
  | 'document' | 'paragraph' | 'heading'
  | 'strong' | 'emphasis' | 'delete'
  | 'code' | 'codeSpan' | 'link'
  | 'list' | 'listItem' | 'table' | 'tableRow' | 'tableCell'
  | 'blockquote' | 'thematicBreak' | 'lineBreak'
  | 'text' | 'html';

export interface MarkdownNode {
  type: MarkdownNodeType;
  value?: string;          // text/code 节点的文本值
  children?: MarkdownNode[];
  // 特殊属性
  lang?: string;           // code 节点的语言标记
  level?: number;          // heading 级别 (1-6)
  ordered?: boolean;       // list 是否有序
  checked?: boolean | null;// listItem 任务列表勾选状态
  url?: string;            // link URL
  title?: string;          // link title
  align?: ('left' | 'center' | 'right' | null)[]; // table 对齐方式
}
```

导出 `MarkdownParser` 类:
```typescript
export class MarkdownParser {
  parse(markdown: string): MarkdownNode[];
  appendText(text: string): MarkdownNode[];  // 增量解析

  // 内部: 使用 micromark(text, { extensions: [gfm()], htmlExtensions: [gfmHtml()] })
  // 但不走 HTML 路径，而是使用 micromark 的 events/tokens 直接构建 AST
  // 关键: 使用 micromark 的 compile-to-events API 而非默认 HTML 编译
  private _tokensToAst(tokens: Event[]): MarkdownNode[];
}
```

关键实现:
- 使用 `import { micromark } from 'micromark'` + GFM 扩展
- 核心策略: 调用 `micromark(text, { extensions: [gfm()] })` 获取 HTML, 然后用简单的正则/状态机将 HTML 转为 AST
- 或者更好: 使用 `import { parse } from 'micromark/lib/parse'` + `import { postprocess } from 'micromark/lib/postprocess'` + `import { preprocess } from 'micromark/lib/preprocess'` 获取 events, 然后 _tokensToAst 将 events 转为自定义 AST
- 增量追加: appendText 重新解析全文 (micromark 不支持真正的增量解析), 但缓存之前的结果以优化

- 有序/无序列表, 嵌套列表
- GFM: 表格, 删除线, 任务列表

```typescript
export interface SyntaxTheme {
  keyword: TextStyle;    // const, let, var, if, else, for, while, return, function, class...
  string: TextStyle;     // "...", '...', `...`
  number: TextStyle;     // 数字字面量
  comment: TextStyle;    // // ..., /* ... */
  function: TextStyle;   // 函数名
  variable: TextStyle;   // 变量名
  type: TextStyle;       // 类型名 (大写开头)
  operator: TextStyle;   // =, +, -, *, /, ==, ===
  punctuation: TextStyle;// {, }, (, ), [, ], ;, ,
  plain: TextStyle;      // 默认
}

export class SyntaxHighlighter {
  constructor(theme: SyntaxTheme);

  highlight(code: string, lang: string): TextSpan[];
  // 简单基于正则的 token 高亮:
  // 1. 匹配关键字: /\b(const|let|var|if|else|...)\b/
  // 2. 匹配字符串: /"[^"]*"|'[^']*'|`[^`]*`/
  // 3. 匹配数字: /\b\d+(\.\d+)?\b/
  // 4. 匹配注释: /\/\/.*$|\/\*[\s\S]*?\*\//
  // 5. 其他 → plain
  // 注意: 不做完整的语法分析, 仅简单 token 级高亮 (与原版一致)
```

```typescript
export class MarkdownRenderer {
  constructor(options?: { syntaxTheme?: SyntaxTheme });

  render(nodes: MarkdownNode[]): TextSpan[];
  // 每个 MarkdownNode 递归转换:
  // - paragraph → TextSpan(children: ...递归子节点) + "\n\n"
  // - text → TextSpan(text: node.value)
  // - strong → TextSpan(style: bold, children: ...递归)
  // - emphasis → TextSpan(style: italic, children: ...递归)
  // - delete → TextSpan(style: strikethrough, children: ...递归)
  // - codeSpan → TextSpan(text: node.value, style: { bg: codeBackground })
  // - code → 调用 SyntaxHighlighter.highlight + 背景色区块
  // - heading → TextSpan(style: bold, children: ["# " + ...递归])
  // - list → 每个 listItem 前加 "  - " 或 "  N. ", 任务列表加 "[x] " / "[ ] "
  // - blockquote → 每行前加 "│ "
  // - link → TextSpan(text: "[title](url)", style: underline + link color)
  // - table → 简单文本格式化 (│ 分隔符 + ─ 横线)
  // - thematicBreak → "───" 水平线
  // - lineBreak → "\n"

  renderStreaming(nodes: MarkdownNode[]): TextSpan[];
  // 与 render 相同，但跳过尾部的空段落 (流式生成时最后一个段落可能不完整)
}
```
