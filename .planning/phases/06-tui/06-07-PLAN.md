export class Clipboard {
  private _strategy: 'pbcopy' | 'wl-copy' | 'xclip' | 'osc52' | null = null;

  constructor();

  /** 自动检测平台, 初始化策略 */
  private _detectPlatform(): void;
  // process.platform === 'darwin' → 'pbcopy'
  // process.env.WAYLAND_DISPLAY → 'wl-copy'
  // process.env.DISPLAY → 'xclip'
  // fallback → 'osc52'

  /** 写入剪贴板 */
  async writeText(text: string): Promise<boolean>;
  // pbcopy: spawn('pbcopy'), stdin.write(text), stdin.end()
  // wl-copy: spawn('wl-copy'), stdin.write(text)
  // xclip: spawn('xclip', ['-selection', 'clipboard']), stdin.write(text)
  // osc52: process.stdout.write(`\x1b]52;c;${btoa(text)}\x07`)

  /** 读取剪贴板 */
  async readText(): Promise<string>;
  // pbpaste: spawn('pbpaste'), 收集 stdout
  // wl-paste: spawn('wl-paste')
  // xclip: spawn('xclip', ['-selection', 'clipboard', '-o'])
  // osc52: 不支持读取, 返回 ""
}
```

export interface SelectionPosition {
  selectableId: string;
  offset: number;             // grapheme offset within selectable
}

export interface Selectable {
  id: string;
  getText(): string;
  getGlobalBounds(): { top: number; left: number; width: number; height: number };
  setHighlightRange(start: number, end: number): void;  // 高亮选中范围
  clearHighlight(): void;
}

export class SelectionArea {
  private _selectables: Map<string, Selectable> = new Map();
  private _orderedCache: Selectable[] = [];             // 按文档序排列
  private _selection: { anchor: SelectionPosition; extent: SelectionPosition } | null = null;
  private _isDragging: boolean = false;
  private _dragAnchor: SelectionPosition | null = null;
  private _clipboard: Clipboard;
  private _copyHighlightTimer: ReturnType<typeof setTimeout> | null = null;

  constructor(clipboard?: Clipboard);

  /** 注册/取消可选组件 */
  register(selectable: Selectable): void;
  unregister(selectableId: string): void;

  /** 选区操作 */
  setSelection(anchor: SelectionPosition, extent: SelectionPosition): void;
  getSelection(): { anchor: SelectionPosition; extent: SelectionPosition } | null;
  clear(): void;
  selectAll(): void;           // 选中所有 selectable 的全部内容

  /** 拖选 */
  beginDrag(position: SelectionPosition): void;
  updateDrag(position: SelectionPosition): void;
  async endDrag(): Promise<void>;   // 结束拖选, 自动复制到剪贴板

  /** 复制 */
  copySelection(): string;     // 收集选区文本
  async copyToClipboard(): Promise<boolean>;

  /** 复制高亮 (300ms 反馈) */
  startCopyHighlight(): void;

  /** 内部: 按文档序排列 selectables */
  private _refreshOrderedCache(): void;
  /** 内部: 将全局选区分发到各 selectable */
  private _propagateSelection(): void;
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-14 | I (Info Disclosure) | Clipboard.writeText | mitigate | 只写入用户主动选择并复制的文本，不自动复制敏感内容；spawn 的命令是硬编码的 (pbcopy/wl-copy/xclip)，不接受用户输入作为命令 |
| T-06-15 | E (Elevation) | Clipboard spawn | mitigate | 不使用 shell=true，直接 spawn 固定命令名，文本通过 stdin 传递不进入命令参数 |
| T-06-16 | D (DoS) | SelectionArea copySelection | accept | 选区文本量受终端显示内容限制，通常不超过几 MB，复制操作为用户主动触发 |
