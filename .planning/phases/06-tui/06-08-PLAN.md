```typescript
/**
 * 计算排序后的第 p 百分位值。
 * @param samples 样本数组
 * @param p 百分位 (0-1, 如 0.95=P95, 0.99=P99)
 * @returns 百分位值, 空数组返回 0
 */
export function percentile(samples: number[], p: number): number {
  if (samples.length === 0) return 0;
  const sorted = [...samples].sort((a, b) => a - b);
  const k = Math.max(0, Math.min(p, 1));
  const index = Math.ceil(sorted.length * k) - 1;
  return sorted[Math.max(0, index)] ?? 0;
}
```

```typescript
export class FrameStatsOverlay {
  private _visible: boolean = false;
  private _frameBudgetMs: number = 16.67;
  readonly WIDTH = 34;
  readonly HEIGHT = 14;

  constructor(options?: { frameBudgetMs?: number });

  get visible(): boolean;
  toggle(): void;
  show(): void;
  hide(): void;

  /**
   * 在 Screen 上直接绘制性能面板。不走 Widget 树。
   * 由 WidgetsBinding.drawFrame() 最后一步调用。
   */
  draw(screen: Screen, tracker: PerformanceTracker): void;
  // 1. 如果 !_visible → 直接返回
  // 2. 计算位置: x = screen.width - WIDTH - 2, y = 1
  // 3. 绘制边框: ╭╮╰╯│─ 字符
  // 4. 绘制标题: "Gotta Go Fast"
  // 5. 绘制每行数据:
  //    格式: "│  Key  1.23   2.34   3.45│"
  //    列标题: "      Last    P95    P99"
  //    行: Key, Mouse, (空行), Build, Layout, Paint, Render, (空行), Frame, Repnt, Bytes
  // 6. 颜色编码:
  //    - _getColor(valueMs): value >= frameBudget → destructive, >= 0.7*frameBudget → warning, else → normal
  //    - Repaint%: >= 50 → destructive, >= 20 → warning, else → normal

  /** 内部: 格式化数值为固定宽度字符串 */
  private _formatValue(value: number, type: 'ms' | 'percent' | 'bytes'): string;
  // ms: 保留 2 位小数, 如 "1.23"
  // percent: 保留 1 位小数 + %, 如 "5.2%"
  // bytes: k 后缀, 如 "1.2k"

  /** 内部: 根据值返回颜色 */
  private _getColor(value: number, type: 'ms' | 'percent'): Color;

  /** 内部: 在 screen 上写入一个字符串 */
  private _drawText(screen: Screen, row: number, col: number, text: string, style: TextStyle): void;
}
```
