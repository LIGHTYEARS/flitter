## SSEParser

```typescript
/**
 * Server-Sent Events 流解析器
 *
 * 将原始字节流按 SSE 协议规范解析为结构化事件。
 * 处理 UTF-8 多字节字符切断（PIT-D1）。
 *
 * @example
 * const parser = new SSEParser();
 * for await (const event of parser.parse(readableStream)) {
 *   console.log(event.event, event.data);
 * }
 */
class SSEParser {
  /**
   * 解析 ReadableStream<Uint8Array> 为 SSE 事件流
   */
  async *parse(stream: ReadableStream<Uint8Array>): AsyncGenerator<SSEEvent>;
}

interface SSEEvent {
  /** 事件类型 (默认 "message") */
  event: string;
  /** 事件数据 (可能跨多行, 用 \n 连接) */
  data: string;
  /** 事件 ID */
  id?: string;
  /** 重连间隔 (ms) */
  retry?: number;
}
```

## fetchSSE

```typescript
/**
 * 发起 HTTP 请求并返回 SSE 事件流
 *
 * 封装 fetch + SSEParser + RetryPolicy，提供统一的流式请求接口。
 *
 * @example
 * for await (const event of fetchSSE(url, options)) {
 *   if (event.data === "[DONE]") break;
 *   const delta = JSON.parse(event.data);
 * }
 */
async function *fetchSSE(
  url: string,
  options: FetchSSEOptions,
): AsyncGenerator<SSEEvent>;

interface FetchSSEOptions {
  method?: "GET" | "POST";
  headers: Record<string, string>;
  body?: string;
  signal?: AbortSignal;
  retry?: RetryOptions;
  /** 连接超时 (ms), 默认 30_000 */
  timeout?: number;
}
```

## RetryPolicy

```typescript
/**
 * 指数退避重试策略
 *
 * 公式: min(base * 2^attempt, maxDelay) * (1 - random * jitter)
 * 默认: base=500ms, maxDelay=8000ms, jitter=0.25, maxRetries=3
 */
class RetryPolicy {
  constructor(options?: RetryOptions);

  /** 判断是否应该重试 */
  shouldRetry(status: number, attempt: number, headers?: Headers): boolean;

  /** 计算下一次重试的等待时间 (ms) */
  getRetryDelay(attempt: number, headers?: Headers): number;

  /** 执行带重试的异步操作 */
  async execute<T>(fn: () => Promise<T>, signal?: AbortSignal): Promise<T>;
}

interface RetryOptions {
  maxRetries?: number;     // 默认 3
  baseDelayMs?: number;    // 默认 500
  maxDelayMs?: number;     // 默认 8000
  jitter?: number;         // 默认 0.25
  retryableStatuses?: number[];  // 默认 [408, 409, 429, 500, 502, 503, 504]
}
```

   - `constructor(options?)` — 合并默认值
   - `shouldRetry(status, attempt, headers?)`:
     - `attempt >= maxRetries` → false
     - 检查 `x-should-retry` header (显式 true/false 覆盖)
     - 401/403 → false (认证错误不重试)

     - 其他 → false
   - `getRetryDelay(attempt, headers?)`:
     - 检查 `Retry-After` header:
       - 数字值 → 转为 ms
       - 日期值 → 计算差值
     - 无 header → 指数退避: `min(baseDelayMs * 2^attempt, maxDelayMs) * (1 - random * jitter)`
   - `async execute<T>(fn, signal?)`:
     - 循环执行 fn
     - 捕获异常，检查 shouldRetry
     - 等待 getRetryDelay 时间
     - AbortSignal 检查
     - 超过重试次数后抛出最后一个错误
