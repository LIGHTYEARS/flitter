## Anthropic Messages API (自行封装, 不使用官方 SDK)

```typescript
/** POST https://api.anthropic.com/v1/messages */
interface AnthropicRequest {
  model: string;
  max_tokens: number;
  messages: AnthropicMessage[];
  system?: AnthropicSystemBlock[];
  tools?: AnthropicTool[];
  stream: true;
  thinking?: {
    type: "enabled" | "adaptive";
    budget_tokens?: number;
  };
  output_config?: {
    effort: "low" | "medium" | "high" | "max";
  };
  speed?: "fast";
  temperature?: number;
}

/** Anthropic 消息格式 */
interface AnthropicMessage {
  role: "user" | "assistant";
  content: AnthropicContentBlock[];
}

/** Anthropic system block (支持 cache_control) */
interface AnthropicSystemBlock {
  type: "text";
  text: string;
  cache_control?: { type: "ephemeral" };
}

type AnthropicContentBlock =
  | { type: "text"; text: string; cache_control?: { type: "ephemeral" } }
  | { type: "image"; source: { type: "base64"; media_type: string; data: string } }
  | { type: "tool_use"; id: string; name: string; input: Record<string, unknown> }
  | { type: "tool_result"; tool_use_id: string; content: string | AnthropicContentBlock[] }
  | { type: "thinking"; thinking: string; signature: string }
  | { type: "redacted_thinking"; data: string };

/** Anthropic 工具定义 */
interface AnthropicTool {
  name: string;
  description: string;
  input_schema: Record<string, unknown>;
}

/** SSE 事件类型 */
type AnthropicSSEEvent =
  | { type: "message_start"; message: { id: string; model: string; usage: AnthropicUsage } }
  | { type: "content_block_start"; index: number; content_block: AnthropicContentBlock }
  | { type: "content_block_delta"; index: number; delta: AnthropicDelta }
  | { type: "content_block_stop"; index: number }
  | { type: "message_delta"; delta: { stop_reason: string }; usage: { output_tokens: number } }
  | { type: "message_stop" };

type AnthropicDelta =
  | { type: "text_delta"; text: string }
  | { type: "input_json_delta"; partial_json: string }
  | { type: "thinking_delta"; thinking: string }
  | { type: "signature_delta"; signature: string };

interface AnthropicUsage {
  input_tokens: number;
  output_tokens: number;
  cache_creation_input_tokens?: number;
  cache_read_input_tokens?: number;
}
```

## AnthropicTransformer

```typescript
class AnthropicTransformer extends BaseMessageTransformer<AnthropicMessage, AnthropicSSEEvent> {
  toProviderMessages(messages: Message[], systemPrompt: TextBlock[]): {
    system: AnthropicSystemBlock[];
    messages: AnthropicMessage[];
  };
  fromProviderDelta(event: AnthropicSSEEvent, state: TransformState): StreamDelta;
}
```

2. 实现 `AnthropicTransformer` 类继承 `BaseMessageTransformer`:

**toProviderMessages(messages, systemPrompt)**:
  - system: 将 systemPrompt TextBlock[] 转为 AnthropicSystemBlock[] (带 cache_control)
  - messages: 遍历 Message[]:
    - UserMessage → role:"user", content 转换:
      - TextBlock → { type:"text", text }
      - ImageBlock → { type:"image", source:{ type:"base64", ... } }
      - ToolResultBlock → { type:"tool_result", tool_use_id, content }
    - AssistantMessage → role:"assistant", content 转换:
      - TextContentBlock → { type:"text", text }
      - ToolUseBlock → { type:"tool_use", id, name, input }
      - ThinkingBlock → { type:"thinking", thinking, signature } (仅保留 provider="anthropic")
      - RedactedThinkingBlock → { type:"redacted_thinking", data }
    - InfoMessage → 跳过 (不发送给 API)
  - 合并相邻同角色消息 (Anthropic 要求交替 user/assistant)

**fromProviderDelta(event, state)**:
  - message_start → 初始化 usage, 返回 streaming delta
  - content_block_start → state.addBlock(index, block)
  - content_block_delta → state.updateBlock(index, delta):
    - text_delta → 追加 text
    - input_json_delta → 追加 partial JSON 字符串
    - thinking_delta → 追加 thinking 文本
    - signature_delta → 设置 signature
  - content_block_stop → state.completeBlock(index), 解析 tool_use 的完整 JSON
  - message_delta → 提取 stop_reason + output_tokens
  - message_stop → createCompleteDelta()
  </action>
