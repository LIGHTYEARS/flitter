## OpenAI Responses API (自行封装)

```typescript
/** POST https://api.openai.com/v1/responses */
interface OpenAIRequest {
  model: string;
  input: OpenAIInputItem[];
  store?: boolean;
  include?: string[];  // ["reasoning.encrypted_content"]
  tools?: OpenAITool[];
  stream: true;
  max_output_tokens?: number;
  prompt_cache_key?: string;
  parallel_tool_calls?: boolean;
  stream_options?: { include_obfuscation: boolean };
  reasoning?: {
    effort: "none" | "minimal" | "low" | "medium" | "high" | "xhigh";
    summary: "auto";
  };
  service_tier?: "speed" | "balanced" | "quality";
  temperature?: number;
}

type OpenAIInputItem =
  | { role: "system"; content: string }
  | { role: "user"; content: string | OpenAIContentPart[] }
  | { role: "assistant"; content: string | OpenAIContentPart[] }
  | { role: "tool"; tool_call_id: string; content: string };

type OpenAIContentPart =
  | { type: "input_text"; text: string }
  | { type: "input_image"; image_url: { url: string } };

interface OpenAITool {
  type: "function";
  name: string;
  description: string;
  parameters: Record<string, unknown>;
}

/** SSE 事件类型 (Responses API) */
type OpenAISSEEvent =
  | { type: "response.created"; response: OpenAIResponse }
  | { type: "response.in_progress" }
  | { type: "response.output_item.added"; item: OpenAIOutputItem }
  | { type: "response.content_part.added"; part: OpenAIContentPart }
  | { type: "response.output_text.delta"; delta: string }
  | { type: "response.reasoning.delta"; delta: string }
  | { type: "response.function_call_arguments.delta"; delta: string }
  | { type: "response.output_item.done"; item: OpenAIOutputItem }
  | { type: "response.content_part.done"; part: OpenAIContentPart }
  | { type: "response.completed"; response: OpenAIResponse }
  | { type: "response.failed"; response: OpenAIResponse; error?: { message: string } };

interface OpenAIResponse {
  id: string;
  model: string;
  status: "in_progress" | "completed" | "failed" | "cancelled";
  output: OpenAIOutputItem[];
  usage?: {
    input_tokens: number;
    output_tokens: number;
    input_tokens_details?: { cached_tokens: number };
  };
}

type OpenAIOutputItem =
  | { type: "message"; content: OpenAIMessageContent[] }
  | { type: "reasoning"; content: OpenAIReasoningContent[] }
  | { type: "function_call"; id: string; name: string; arguments: string; call_id: string };

interface OpenAIMessageContent {
  type: "output_text";
  text: string;
}

interface OpenAIReasoningContent {
  type: "reasoning";
  text: string;
  encrypted_content?: string;
}
```

1. 定义 OpenAI 原生类型
2. 实现 `OpenAITransformer` 类继承 `BaseMessageTransformer`:

**toProviderMessages(messages, systemPrompt)**:
  - system: systemPrompt → { role:"system", content: joined text }
  - messages: 遍历 Message[]:
    - UserMessage → role:"user", content 转换:
      - TextBlock → { type:"input_text", text }
      - ImageBlock → { type:"input_image", image_url:{ url } }
      - ToolResultBlock → { role:"tool", tool_call_id, content }
    - AssistantMessage → role:"assistant", 提取 text content
      - ToolUseBlock → 对应为 function_call output item
      - ThinkingBlock → 仅保留 provider="openai" 的 reasoning
    - InfoMessage → 跳过

**fromProviderDelta(event, state)**:
  - response.created → 初始化, streaming delta
  - response.output_item.added → 判断类型:
    - message → addBlock (text)
    - reasoning → addBlock (thinking)
    - function_call → addBlock (tool_use)
  - response.output_text.delta → updateBlock 追加文本
  - response.reasoning.delta → updateBlock 追加 thinking
  - response.function_call_arguments.delta → updateBlock 追加 JSON
  - response.output_item.done → completeBlock
  - response.completed → createCompleteDelta + usage 解析
    - usage.input_tokens_details.cached_tokens → cacheReadInputTokens
    - cacheCreationInputTokens = input_tokens - cached_tokens
  - response.failed → createErrorDelta
  </action>
