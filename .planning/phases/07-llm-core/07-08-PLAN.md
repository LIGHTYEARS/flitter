<interfaces>
## MockSSEServer

```typescript
/**
 * 轻量级 Mock SSE 服务器
 *
 * 使用 node:http 创建本地 HTTP 服务器，返回预录制的 SSE 事件序列。
 * 每个测试用例配置不同的响应 fixture。
 */
class MockSSEServer {
  /** 启动服务器，返回 base URL */
  start(): Promise<string>;

  /** 配置下一次请求的响应 */
  setResponse(options: {
    /** SSE 事件序列 (按顺序发送) */
    events: SSEEvent[];
    /** 响应状态码 (默认 200) */
    statusCode?: number;
    /** 响应头 (默认 Content-Type: text/event-stream) */
    headers?: Record<string, string>;
    /** 事件间延迟 (ms, 默认 0) */
    delayBetweenEvents?: number;
    /** 发送部分事件后断开连接 (模拟断流) */
    disconnectAfter?: number;
  }): void;

  /** 获取最后一次收到的请求 */
  getLastRequest(): { method: string; url: string; headers: Record<string, string>; body: unknown };

  /** 关闭服务器 */
  close(): Promise<void>;
}

interface SSEEvent {
  event?: string;  // SSE event type (如 "message_start")
  data: string;    // SSE data payload
  id?: string;     // SSE event id
}
```
