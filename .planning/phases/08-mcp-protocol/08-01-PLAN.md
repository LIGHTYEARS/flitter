## 实现细节

### 1. types.ts — MCP 类型定义

```ts
// 复用 @flitter/schemas 的 JSON-RPC types
import type { JSONRPCRequest, JSONRPCResponse, JSONRPCNotification } from "@flitter/schemas";

// JSON-RPC 消息联合类型
export type JSONRPCMessage = JSONRPCRequest | JSONRPCResponse | JSONRPCNotification;

// MCP 错误码 (数值常量，与 JSON-RPC 规范一致)
export const ErrorCode = {
  ConnectionClosed: -32000,
  RequestTimeout: -32001,
  ParseError: -32700,
  InvalidRequest: -32600,
  MethodNotFound: -32601,
  InvalidParams: -32602,
  InternalError: -32603,
} as const;

// MCP 协议方法常量
export const Method = {
  Initialize: "initialize",
  Initialized: "notifications/initialized",
  Ping: "ping",
  // Tools
  ToolsList: "tools/list",
  ToolsCall: "tools/call",
  ToolsListChanged: "notifications/tools/list_changed",
  // Resources
  ResourcesList: "resources/list",
  ResourcesRead: "resources/read",
  ResourcesListChanged: "notifications/resources/list_changed",
  // Prompts
  PromptsList: "prompts/list",
  PromptsGet: "prompts/get",
  PromptsListChanged: "notifications/prompts/list_changed",
  // Logging
  LoggingSetLevel: "logging/setLevel",
  LogMessage: "notifications/message",
  // Progress & Cancel
  Progress: "notifications/progress",
  Cancelled: "notifications/cancelled",
} as const;

// MCP 协议版本
export const LATEST_PROTOCOL_VERSION = "2025-03-26";
export const SUPPORTED_VERSIONS = ["2025-03-26", "2024-11-05", "2024-10-07"];

// 客户端能力
export interface ClientCapabilities {
  roots?: { listChanged?: boolean };
  sampling?: Record<string, unknown>;
  elicitation?: Record<string, unknown>;
}

// 服务端能力
export interface ServerCapabilities {
  tools?: { listChanged?: boolean };
  resources?: { subscribe?: boolean; listChanged?: boolean };
  prompts?: { listChanged?: boolean };
  logging?: Record<string, unknown>;
  completions?: Record<string, unknown>;
}

// Implementation info
export interface ImplementationInfo {
  name: string;
  version: string;
}

// Initialize result
export interface InitializeResult {
  protocolVersion: string;
  capabilities: ServerCapabilities;
  serverInfo: ImplementationInfo;
  instructions?: string;
}

// MCP Transport 接口
export interface MCPTransport {
  start(): Promise<void>;
  send(message: JSONRPCMessage): Promise<void>;
  close(): Promise<void>;
  onmessage?: (message: JSONRPCMessage) => void;
  onclose?: () => void;
  onerror?: (error: Error) => void;
}

// MCP Tool (from server)
export interface MCPTool {
  name: string;
  description?: string;
  inputSchema: Record<string, unknown>;
}

// MCP Tool Call Result
export interface MCPToolResult {
  content: Array<{ type: "text"; text: string } | { type: "image"; data: string; mimeType: string }>;
  isError?: boolean;
}

// MCP Resource
export interface MCPResource {
  uri: string;
  name: string;
  description?: string;
  mimeType?: string;
}

// MCP Prompt
export interface MCPPrompt {
  name: string;
  description?: string;
  arguments?: Array<{ name: string; description?: string; required?: boolean }>;
}
```

### 2. protocol.ts — 编解码 + 请求管理

```ts
// createRequest(method, params?) → JSONRPCRequest (auto-increment id)
// createNotification(method, params?) → JSONRPCNotification
// createSuccessResponse(id, result) → JSONRPCSuccessResponse
// createErrorResponse(id, code, message, data?) → JSONRPCErrorResponse
// parseMessage(input: string) → JSONRPCMessage (使用 Zod union parse)

// McpError extends Error { code, data, static fromError(err) }

// RequestManager:
//   - request(method, params, timeout?) → Promise<unknown>  (发送请求，注册 pending，等待响应)
//   - handleResponse(response) → void  (匹配 pending request 并 resolve/reject)
//   - cancelAll(reason?) → void  (取消所有 pending)
//   - pendingCount → number
```
