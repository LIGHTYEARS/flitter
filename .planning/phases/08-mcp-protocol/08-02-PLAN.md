## 实现细节

### 1. read-buffer.ts — 行分隔 JSON 缓冲区

```ts
class ReadBuffer {
  private _buffer = Buffer.alloc(0);

  append(chunk: Buffer): void {
    this._buffer = Buffer.concat([this._buffer, chunk]);
  }

  readMessage(): JSONRPCMessage | null {
    const idx = this._buffer.indexOf(0x0a); // '\n'
    if (idx === -1) return null;
    const line = this._buffer.subarray(0, idx).toString("utf-8");
    this._buffer = this._buffer.subarray(idx + 1);
    return parseMessage(line); // 使用 Plan 08-01 的解析器
  }

  clear(): void {
    this._buffer = Buffer.alloc(0);
  }
}
```

### 2. stdio.ts — Stdio 传输

关键实现:
- `spawn(command, args, { stdio: ["pipe","pipe","pipe"], shell: false, cwd, env })`
