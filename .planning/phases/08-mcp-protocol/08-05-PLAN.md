### 1. types.ts — MCP OAuth 类型 + Zod Schemas

```ts
// Zod schemas (从 mcp-oauth-schemas.js 直译)

export const ProtectedResourceMetadataSchema = z.object({
  resource: z.string().url(),
  authorization_servers: z.array(z.string().url()).optional(),
  scopes_supported: z.array(z.string()).optional(),
  // ... other fields
});

export const AuthorizationServerMetadataSchema = z.object({
  issuer: z.string(),
  authorization_endpoint: z.string().url(),
  token_endpoint: z.string().url(),
  registration_endpoint: z.string().url().optional(),
  response_types_supported: z.array(z.string()),
  code_challenge_methods_supported: z.array(z.string()).optional(),
  scopes_supported: z.array(z.string()).optional(),
  client_id_metadata_document_supported: z.boolean().optional(),
});

export const OAuthTokensSchema = z.object({
  access_token: z.string(),
  token_type: z.string(),
  expires_in: z.number().optional(),
  scope: z.string().optional(),
  refresh_token: z.string().optional(),
}).passthrough();

export const OAuthClientInfoSchema = z.object({
  client_id: z.string(),
  client_secret: z.string().optional(),
  client_id_issued_at: z.number().optional(),
  client_secret_expires_at: z.number().optional(),
}).passthrough();

// MCPAuthProvider interface (对接 transport)
export interface MCPAuthProvider {
  // Client metadata for dynamic registration
  readonly redirectUrl: URL;
  readonly clientMetadata: OAuthClientRegistration;
  readonly clientMetadataUrl?: string;

  // State storage callbacks (consumer implements persistence)
  tokens(): OAuthTokens | undefined;
  saveTokens(tokens: OAuthTokens): void;
  redirectToAuthorization(url: URL): void;
  saveCodeVerifier(verifier: string): void;
  codeVerifier(): string;
  clientInformation(): OAuthClientInfo | undefined;
  saveClientInformation(info: OAuthClientInfo): void;
  invalidateCredentials?(scope: "all" | "tokens"): void;
  validateResourceURL?(requested: URL, configured?: string): URL | undefined;

  // Auth state notifications
  onAuthStateChange?: (state: string) => void;
}
```

```ts
    resource?: URL;
    fetchFn?: typeof fetch;
  }

export async function refreshAccessToken(
  issuerUrl: URL,
  opts: {
    metadata?: AuthorizationServerMetadata;
    clientInformation: OAuthClientInfo;
    refreshToken: string;
    resource?: URL;
    fetchFn?: typeof fetch;
  }
): Promise<OAuthTokens>

// ─── 完整 OAuth 流程编排 ────────────────────────────
// 对应逆向 Q_ + Dq
export async function auth(
  authProvider: MCPAuthProvider,
  opts: {
    serverUrl: URL;
    authorizationCode?: string;
    scope?: string;
    resourceMetadataUrl?: URL;
    fetchFn?: typeof fetch;
  }
): Promise<"AUTHORIZED" | "REDIRECT">
// 流程:
// 1. discoverProtectedResource(serverUrl) → resource metadata
// 2. 从 resource metadata 取 authorization_servers[0]，没有则 fallback 到 serverUrl origin
// 3. validateResourceURL (如果 provider 有此方法)
// 4. discoverAuthorizationServer(issuerUrl) → auth server metadata
// 5. clientInformation() — 已有则复用
//    - 如果没有且 client_id_metadata_document_supported + clientMetadataUrl → 用 URL 作为 client_id
//    - 否则 registerClient() 动态注册
// 6. 如果有 authorizationCode → exchangeAuthorizationCode() → saveTokens → "AUTHORIZED"
// 7. 如果有 refresh_token → refreshAccessToken() → saveTokens → "AUTHORIZED"
// 8. 否则 buildAuthorizationUrl() → saveCodeVerifier → redirectToAuthorization → "REDIRECT"
//
// retry 逻辑: invalidation errors → invalidateCredentials → retry once
```

### 3. 测试策略

使用 `http.createServer` 模拟 OAuth 端点:

- **parseWWWAuthenticate**: 各种 WWW-Authenticate header 格式解析
- **well-known 发现**: path-suffix 模式 + origin-root 模式 + 404 处理
- **客户端注册**: 动态注册 + client_id_metadata_document 模式
- **PKCE 授权 URL**: 参数正确性 (code_challenge, redirect_uri, scope, state, resource)
- **Token 交换**: 授权码交换 + refresh token 更新
- **auth() 完整流程**: 首次授权 → REDIRECT → 带 code 回来 → AUTHORIZED
- **auth() refresh 路径**: 已有 refresh_token → AUTHORIZED
- **auth() retry**: invalidation 后重试
