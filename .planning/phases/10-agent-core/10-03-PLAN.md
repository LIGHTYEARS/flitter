### 1. read.ts -- Read 文件读取工具

```ts
import * as fs from "node:fs";
import * as path from "node:path";
import type { ToolSpec, ToolResult, ToolContext } from "../types";

const DEFAULT_LIMIT = 2000;
const MAX_LINE_LENGTH = 2000;

/**
 * 检测文件是否为二进制
 * 策略: 读取前 8192 字节, 检查是否包含 NULL 字节 (0x00)
 */
function isBinaryFile(filePath: string): Promise<boolean>;

/**
 * 格式化行号输出 (cat -n 格式)
 * 格式: "     1\tline content"
 * - 行号右对齐, 最小 6 字符宽度
 * - 行号与内容之间用 tab 分隔
 * - 超过 MAX_LINE_LENGTH 的行截断并附加 " [truncated]"
 */
function formatWithLineNumbers(
  lines: string[],
  startLine: number
): string;

export const ReadTool: ToolSpec = {
  name: "Read",
  description: "Reads a file from the local filesystem. Returns file content with line numbers.",
  inputSchema: {
    type: "object",
    properties: {
      file_path: {
        type: "string",
        description: "The absolute path to the file to read",
      },
      offset: {
        type: "number",
        description: "The line number to start reading from (1-based). Only provide if the file is too large to read at once",
      },
      limit: {
        type: "number",
        description: "The number of lines to read. Only provide if the file is too large to read at once.",
      },
    },
    required: ["file_path"],
  },
  source: "builtin",
  isReadOnly: true,

  execute(args, context): Promise<ToolResult> {
    // 1. 验证 file_path 参数
    // 2. 解析 offset (默认 0) 和 limit (默认 2000)
    // 3. 检查文件是否存在 → 不存在返回 { status: "error", error: "File not found: ..." }
    // 4. 二进制检测 → 是二进制返回 { status: "error", error: "Cannot read binary file: ..." }
    // 5. 读取文件内容, 按行分割
    // 6. 应用 offset/limit 切片
    // 7. 格式化行号输出 (cat -n 格式)
    // 8. 返回 { status: "done", content: formattedOutput }
  },

  get executionProfile() {
    // 动态生成: 在 execute 调用前由 orchestrator 通过 args 获取
    // 实际实现: execute 内部返回 profile, 或者工具定义为函数
    return undefined; // 由 orchestrator 根据 args.file_path 动态构建
  },
};

// 辅助函数: 根据参数生成 executionProfile
export function readExecutionProfile(args: Record<string, unknown>) {
  const filePath = args.file_path as string;
  return {
    resourceKeys: [{ key: filePath, mode: "read" as const }],
  };
}
```

**executionProfile 动态生成说明**: Read/Write/Edit 工具的资源键依赖于运行时参数 (`file_path`)。在 ToolSpec 上可定义 `executionProfile` 为函数 `(args) => ExecutionProfile`, 或者由 orchestrator 调用工具的 `getExecutionProfile?.(args)` 方法。实现时优先使用函数形式:

```ts
// ToolSpec 扩展: executionProfile 可以是静态对象或函数
executionProfile:
  | ExecutionProfile
  | ((args: Record<string, unknown>) => ExecutionProfile);
```

如果 Plan 10-01 的 ToolSpec 未定义函数形式, 则在工具模块中导出 `xxxExecutionProfile(args)` 辅助函数, 由 orchestrator 在分批前调用。

### 2. write.ts -- Write 文件写入工具

```ts
import * as fs from "node:fs";
import * as path from "node:path";
import type { ToolSpec, ToolResult, ToolContext } from "../types";

export const WriteTool: ToolSpec = {
  name: "Write",
  description: "Writes content to a file on the local filesystem. Creates parent directories if they don't exist.",
  inputSchema: {
    type: "object",
    properties: {
      file_path: {
        type: "string",
        description: "The absolute path to the file to write",
      },
      content: {
        type: "string",
        description: "The content to write to the file",
      },
    },
    required: ["file_path", "content"],
  },
  source: "builtin",
  isReadOnly: false,

  async execute(args, context): Promise<ToolResult> {
    // 1. 验证参数: file_path 和 content
    // 2. 解析目录路径
    // 3. 创建中间目录 (fs.promises.mkdir with recursive: true)
    // 4. 写入文件 (fs.promises.writeFile)
    // 5. 计算行数 (content.split('\n').length)
    // 6. 返回 { status: "done", content: `Successfully wrote to ${filePath} (${lineCount} lines)`, outputFiles: [filePath] }
    //
    // 错误处理:
    // - 权限错误 → { status: "error", error: "Permission denied: ..." }
    // - 其他 IO 错误 → { status: "error", error: message }
  },
};

export function writeExecutionProfile(args: Record<string, unknown>) {
  const filePath = args.file_path as string;
  return {
    resourceKeys: [{ key: filePath, mode: "write" as const }],
  };
}
```

### 3. edit.ts -- Edit 精确字符串替换工具

```ts
import * as fs from "node:fs";
import type { ToolSpec, ToolResult, ToolContext } from "../types";

export const EditTool: ToolSpec = {
  name: "Edit",
  description: "Performs exact string replacements in files. Fails if old_string is not found or not unique (unless replace_all is true).",
  inputSchema: {
    type: "object",
    properties: {
      file_path: {
        type: "string",
        description: "The absolute path to the file to modify",
      },
      old_string: {
        type: "string",
        description: "The exact text to find and replace",
      },
      new_string: {
        type: "string",
        description: "The text to replace it with (must be different from old_string)",
      },
      replace_all: {
        type: "boolean",
        description: "Replace all occurrences of old_string (default false)",
        default: false,
      },
    },
    required: ["file_path", "old_string", "new_string"],
  },
  source: "builtin",
  isReadOnly: false,

  async execute(args, context): Promise<ToolResult> {
    // 1. 验证参数: file_path, old_string, new_string
    // 2. 验证 old_string !== new_string → error if equal
    // 3. 读取文件内容 → 不存在返回 error
    // 4. 计算 old_string 在文件中的出现次数
    // 5. 如果出现次数 === 0 → { status: "error", error: "old_string not found in file" }
    // 6. 如果出现次数 > 1 且 replace_all !== true →
    //    { status: "error", error: "old_string found N times. Use replace_all to replace all occurrences, or provide more context to make it unique." }
    // 7. 执行替换:
    //    - replace_all=true: 全局替换
    //    - 否则: 替换第一次出现 (此时已验证仅出现一次)
    // 8. 写回文件
    // 9. 返回 { status: "done", content: `Replaced ${count} occurrence(s) in ${filePath}`, outputFiles: [filePath] }
  },
};

export function editExecutionProfile(args: Record<string, unknown>) {
  const filePath = args.file_path as string;
  return {
    resourceKeys: [{ key: filePath, mode: "write" as const }],
  };
}
```
