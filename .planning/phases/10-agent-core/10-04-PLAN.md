import { spawn as cpSpawn } from "node:child_process";
import type { ToolSpec, ToolResult, ToolContext } from "../types";

const DEFAULT_TIMEOUT = 120_000;     // 2 minutes
const MAX_TIMEOUT = 600_000;         // 10 minutes
const MAX_OUTPUT_LENGTH = 30_000;    // chars
const SIGTERM_GRACE_PERIOD = 5_000;  // 5 seconds before SIGKILL

// ─── 输出截断 ────────────────────────────────────────────

/**
 * 截断输出: 保留首尾各约一半, 中间插入省略标记
 *
 * 逆向: process-runner.js 的输出截断逻辑
 *
 * 截断策略:
 * - 如果 output.length <= MAX_OUTPUT_LENGTH → 原样返回
 * - 否则: 取前 halfLength 字符 + 省略标记 + 后 halfLength 字符
 * - 省略标记: "\n\n[output truncated -- {omittedCount} chars omitted]\n\n"
 */
function truncateOutput(output: string): string;

// ─── 进程执行 ────────────────────────────────────────────

/**
 * 执行 shell 命令并收集输出
 *
 * 使用 child_process.spawn (非 execFile):
 * - shell: true → 支持管道、重定向、环境变量展开
 * - 合并 stdout/stderr 为单一输出 (实时收集)
 *
 * 超时处理:
 * 1. setTimeout(timeout) 触发 SIGTERM
 * 2. 再等 SIGTERM_GRACE_PERIOD
 * 3. 如果进程还活着 → SIGKILL
 *
 * AbortSignal 处理:
 * - signal.addEventListener('abort', ...) → 发送 SIGTERM + 同上 grace period
 */
interface ShellResult {
  stdout: string;
  stderr: string;
  exitCode: number;
  timedOut: boolean;
}

async function executeShell(
  command: string,
  options: {
    cwd: string;
    timeout: number;
    signal: AbortSignal;
    env?: NodeJS.ProcessEnv;
  }
): Promise<ShellResult>;

// ─── ToolSpec 定义 ──────────────────────────────────────

  },

  // BashTool 始终串行执行 -- shell 命令有副作用
  executionProfile: { serial: true },

  async execute(args, context): Promise<ToolResult> {
    // 1. 验证 command 参数 (必须非空字符串)
    // 2. 解析 timeout: clamp(args.timeout ?? DEFAULT_TIMEOUT, 0, MAX_TIMEOUT)
    // 3. 调用 executeShell(command, { cwd: context.workingDirectory, timeout, signal: context.signal })

    //    - data: { stdout, stderr, exitCode } (结构化数据)

    // 6. 返回 ToolResult

    // 异常处理 (spawn 本身失败, 如 command not found):
    // - { status: "error", error: errorMessage }
  },
};

### 2. executeShell 详细流程

```ts
async function executeShell(command, options): Promise<ShellResult> {
  return new Promise((resolve, reject) => {
    const stdoutChunks: Buffer[] = [];
    const stderrChunks: Buffer[] = [];
    let timedOut = false;
    let killed = false;

    // 1. spawn with shell mode
    const child = cpSpawn(command, [], {
      cwd: options.cwd,
      env: options.env ?? process.env,
      shell: true,
      stdio: ["ignore", "pipe", "pipe"],  // stdin ignored, stdout/stderr piped
    });

    // 2. 收集输出
    child.stdout?.on("data", (chunk) => stdoutChunks.push(chunk));
    child.stderr?.on("data", (chunk) => stderrChunks.push(chunk));

    // 3. 超时处理
    const timeoutId = setTimeout(() => {
      timedOut = true;
      killGracefully(child);
    }, options.timeout);

    // 4. AbortSignal 处理
    const abortHandler = () => {
      killed = true;
      killGracefully(child);
    };
    if (!options.signal.aborted) {
      options.signal.addEventListener("abort", abortHandler, { once: true });
    } else {
      // 已经 aborted
      killed = true;
      killGracefully(child);
    }

    // 5. 进程退出
    child.on("close", (code, signal) => {
      clearTimeout(timeoutId);
      options.signal.removeEventListener("abort", abortHandler);

      const stdout = Buffer.concat(stdoutChunks).toString("utf-8");
      const stderr = Buffer.concat(stderrChunks).toString("utf-8");

      if (killed && !timedOut) {
        // AbortSignal triggered
        reject(new DOMException("The operation was aborted", "AbortError"));
        return;
      }

      resolve({
        stdout,
        stderr,
        exitCode: code ?? 1,
        timedOut,
      });
    });

    child.on("error", (err) => {
      clearTimeout(timeoutId);
      options.signal.removeEventListener("abort", abortHandler);
      reject(err);
    });
  });
}

/**
 * 优雅终止: SIGTERM → grace period → SIGKILL
 */
function killGracefully(child: ChildProcess): void {
  try {
    child.kill("SIGTERM");
  } catch { /* ignore if already dead */ }

  setTimeout(() => {
    try {
      if (!child.killed) {
        child.kill("SIGKILL");
      }
    } catch { /* ignore */ }
  }, SIGTERM_GRACE_PERIOD);
}
```

### 3. 输出截断详细算法

```ts
function truncateOutput(output: string): string {
  if (output.length <= MAX_OUTPUT_LENGTH) return output;

  const marker = "\n\n[output truncated]";
  const halfLength = Math.floor((MAX_OUTPUT_LENGTH - marker.length) / 2);
  const omitted = output.length - halfLength * 2;

  const head = output.slice(0, halfLength);
  const tail = output.slice(output.length - halfLength);

  return `${head}\n\n[output truncated -- ${omitted} chars omitted]\n\n${tail}`;
}
```
