### 1. grep.ts -- Grep 正则内容搜索工具

```ts
import * as fs from "node:fs";
import * as path from "node:path";
import { spawn } from "@flitter/util";
import type { ToolSpec, ToolResult, ToolContext } from "../types";

// ─── 类型 ────────────────────────────────────────────────

type OutputMode = "content" | "files_with_matches" | "count";

// ─── rg 可用性检测 (缓存) ───────────────────────────────

let rgAvailable: boolean | null = null;

async function isRgAvailable(): Promise<boolean> {
  if (rgAvailable !== null) return rgAvailable;
  try {
    const result = await spawn("rg", ["--version"]);
    rgAvailable = result.exitCode === 0;
  } catch {
    rgAvailable = false;
  }
  return rgAvailable;
}

// ─── rg 命令构建 ────────────────────────────────────────

/**
 * 构建 rg 命令参数
 *
 * 映射关系:
 * - pattern → rg 的搜索模式
 * - output_mode: "content" → 默认 (输出匹配行)
 *                "files_with_matches" → --files-with-matches (-l)
 *                "count" → --count (-c)
 * - context/-C → --context N
 * - -A → --after-context N
 * - -B → --before-context N
 * - -i → --ignore-case
 * - -n → --line-number (content 模式默认开启)
 * - type → --type TYPE
 * - glob → --glob PATTERN
 * - multiline → --multiline --multiline-dotall
 * - head_limit → 通过 head 或截断实现
 * - offset → 跳过前 N 行/条目
 * - path → 搜索路径 (默认 cwd)
 */
function buildRgArgs(args: Record<string, unknown>, cwd: string): string[];

// ─── Node.js 回退实现 ──────────────────────────────────

/**
 * 当 rg 不可用时的 Node.js 内置正则搜索
 *
 * 策略:
 * 1. 递归遍历目录 (跳过 node_modules, .git, 二进制文件)
 * 2. 对每个文本文件按行匹配正则
 * 3. 根据 output_mode 格式化输出
 *
 * 限制:
 * - 不支持 type 过滤 (仅使用 glob)
 * - 性能弱于 rg, 适用于小目录或 rg 不可用场景
 */
async function grepFallback(
  pattern: string,
  searchPath: string,
  options: {
    outputMode: OutputMode;
    context: number;
    ignoreCase: boolean;
    lineNumbers: boolean;
    globFilter?: string;
    headLimit: number;
    offset: number;
    multiline: boolean;
  }
): Promise<string>;

// ─── ToolSpec 定义 ──────────────────────────────────────

export const GrepTool: ToolSpec = {
  name: "Grep",
  description:
    "A powerful search tool built on ripgrep. " +
    "Supports full regex syntax, file type filtering, and multiple output modes.",
  inputSchema: {
    type: "object",
    properties: {
      pattern: {
        type: "string",
        description: "The regular expression pattern to search for",
      },
      path: {
        type: "string",
        description:
          "File or directory to search in. Defaults to current working directory.",
      },
      type: {
        type: "string",
        description:
          "File type to search (rg --type). Common types: js, py, rust, go, java, etc.",
      },
      glob: {
        type: "string",
        description:
          'Glob pattern to filter files (e.g. "*.js", "*.{ts,tsx}")',
      },
      output_mode: {
        type: "string",
        enum: ["content", "files_with_matches", "count"],
        description:
          'Output mode. "content" shows matching lines, "files_with_matches" shows file paths (default), "count" shows match counts.',
      },
      context: {
        type: "number",
        description: "Number of context lines before and after each match",
      },
      "-A": {
        type: "number",
        description: "Number of lines to show after each match",
      },
      "-B": {
        type: "number",
        description: "Number of lines to show before each match",
      },
      "-C": {
        type: "number",
        description: "Alias for context",
      },
      "-i": {
        type: "boolean",
        description: "Case insensitive search",
      },
      "-n": {
        type: "boolean",
        description: "Show line numbers in output. Defaults to true for content mode.",
      },
      multiline: {
        type: "boolean",
        description: "Enable multiline mode where . matches newlines",
      },
      head_limit: {
        type: "number",
        description: "Limit output to first N lines/entries",
      },
      offset: {
        type: "number",
        description: "Skip first N lines/entries before applying head_limit",
      },
    },
    required: ["pattern"],
  },
  source: "builtin",
  isReadOnly: true,

  async execute(args, context): Promise<ToolResult> {
    // 1. 解析参数: pattern, path (默认 cwd), outputMode (默认 "files_with_matches"), 等
    const searchPath = (args.path as string) || context.workingDirectory;
    const outputMode = (args.output_mode as OutputMode) || "files_with_matches";

    // 2. 检查 rg 是否可用
    const useRg = await isRgAvailable();

    let output: string;

    if (useRg) {
      // 3a. 构建 rg 参数 → spawn rg → 收集输出
      const rgArgs = buildRgArgs(args, searchPath);
      try {
        const result = await spawn("rg", rgArgs, {
          cwd: searchPath,
          timeout: 30_000,
          signal: context.signal,
        });
        output = result.stdout;
        // rg exitCode 1 = no match (不是错误)
        if (result.exitCode > 1) {
          return { status: "error", error: `rg failed with exit code ${result.exitCode}: ${result.stderr}` };
        }
      } catch (err) {
        return { status: "error", error: `rg execution failed: ${(err as Error).message}` };
      }
    } else {
      // 3b. 回退到 Node.js 内置搜索
      output = await grepFallback(args.pattern as string, searchPath, {
        outputMode,
        context: (args.context ?? args["-C"] ?? 0) as number,
        ignoreCase: (args["-i"] ?? false) as boolean,
        lineNumbers: (args["-n"] ?? true) as boolean,
        globFilter: args.glob as string | undefined,
        headLimit: (args.head_limit ?? 0) as number,
        offset: (args.offset ?? 0) as number,
        multiline: (args.multiline ?? false) as boolean,
      });
    }

    // 4. 应用 head_limit 和 offset (如果 rg 模式下尚未应用)
    // 5. 返回结果
    if (!output.trim()) {
      return { status: "done", content: "No matches found." };
    }
    return { status: "done", content: output };
  },
};

// 辅助函数: 根据参数生成 executionProfile
export function grepExecutionProfile(args: Record<string, unknown>, cwd: string) {
  const searchPath = (args.path as string) || cwd;
  return {
    resourceKeys: [{ key: searchPath, mode: "read" as const }],
  };
}
```

### 2. glob.ts -- Glob 文件模式匹配工具

```ts
import * as fs from "node:fs";
import * as path from "node:path";
import { FileScanner } from "@flitter/util";
import type { ToolSpec, ToolResult, ToolContext } from "../types";

/**
 * 将 glob 模式 (如 "**\/*.ts") 转换为正则表达式
 * 用于从 FileScanner 结果中过滤匹配文件
 */
function globPatternToRegex(pattern: string): RegExp;

/**
 * 按修改时间排序文件路径 (最新的在前)
 */
async function sortByMtime(filePaths: string[]): Promise<string[]>;

export const GlobTool: ToolSpec = {
  name: "Glob",
  description:
    "Fast file pattern matching tool that works with any codebase size. " +
    "Supports glob patterns like \"**/*.js\" or \"src/**/*.ts\". " +
    "Returns matching file paths sorted by modification time.",
  inputSchema: {
    type: "object",
    properties: {
      pattern: {
        type: "string",
        description: "The glob pattern to match files against",
      },
      path: {
        type: "string",
        description:
          "The directory to search in. If not specified, the current working directory will be used.",
      },
    },
    required: ["pattern"],
  },
  source: "builtin",
  isReadOnly: true,

  async execute(args, context): Promise<ToolResult> {
    const pattern = args.pattern as string;
    const searchPath = (args.path as string) || context.workingDirectory;

    // 1. 验证 searchPath 存在
    // 2. 使用 FileScanner 扫描目录
    const scanner = new FileScanner([searchPath], {
      ignorePatterns: ["node_modules", ".git"],
      abortSignal: context.signal,
    });
    const scanResult = await scanner.scan();

    // 3. 将 glob 模式转为正则, 过滤扫描结果
    const regex = globPatternToRegex(pattern);
    const matched = scanResult.entries
      .filter((e) => !e.isDirectory)
      .filter((e) => {
        const relativePath = path.relative(searchPath, e.path);
        return regex.test(relativePath) || regex.test(e.name);
      })
      .map((e) => e.path);

    // 4. 按修改时间排序
    const sorted = await sortByMtime(matched);

    // 5. 返回结果
    if (sorted.length === 0) {
      return { status: "done", content: "No files matched the pattern." };
    }
    return { status: "done", content: sorted.join("\n") };
  },
};

export function globExecutionProfile(args: Record<string, unknown>, cwd: string) {
  const searchPath = (args.path as string) || cwd;
  return {
    resourceKeys: [{ key: searchPath, mode: "read" as const }],
  };
}
```

### 3. fuzzy-find.ts -- FuzzyFind 模糊文件查找工具

```ts
import { FileScanner, FuzzyMatcher } from "@flitter/util";
import type { ToolSpec, ToolResult, ToolContext } from "../types";

const DEFAULT_LIMIT = 20;

export const FuzzyFindTool: ToolSpec = {
  name: "FuzzyFind",
  description:
    "Fuzzy file search tool. Matches file paths using multi-tier scoring: " +
    "exact > prefix > suffix > substring > fuzzy.",
  inputSchema: {
    type: "object",
    properties: {
      query: {
        type: "string",
        description: "The search query to match against file paths",
      },
      path: {
        type: "string",
        description:
          "The directory to search in. If not specified, the current working directory will be used.",
      },
      limit: {
        type: "number",
        description: "Maximum number of results to return. Default 20.",
      },
    },
    required: ["query"],
  },
  source: "builtin",
  isReadOnly: true,

  async execute(args, context): Promise<ToolResult> {
    const query = args.query as string;
    const searchPath = (args.path as string) || context.workingDirectory;
    const limit = (args.limit as number) || DEFAULT_LIMIT;

    // 1. 使用 FileScanner 扫描目录
    const scanner = new FileScanner([searchPath], {
      ignorePatterns: ["node_modules", ".git"],
      abortSignal: context.signal,
    });
    const scanResult = await scanner.scan();

    // 2. 使用 FuzzyMatcher 进行模糊匹配
    const matcher = new FuzzyMatcher(query, {
      maxResults: limit,
    });
    const results = matcher.match(scanResult.entries);

    // 3. 格式化输出: "path (score: N)"
    if (results.length === 0) {
      return { status: "done", content: "No matching files found." };
    }

    const output = results
      .map((r) => `${r.entry.path} (score: ${r.score})`)
      .join("\n");

    return { status: "done", content: output };
  },
};

export function fuzzyFindExecutionProfile(args: Record<string, unknown>, cwd: string) {
  const searchPath = (args.path as string) || cwd;
  return {
    resourceKeys: [{ key: searchPath, mode: "read" as const }],
  };
}
```
