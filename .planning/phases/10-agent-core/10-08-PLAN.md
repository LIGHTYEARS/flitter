### 1. context-blocks.ts -- 上下文块收集

```ts
import type { SystemPromptBlock } from "@flitter/llm";
import type { ConfigService } from "@flitter/data";
import type { SkillService } from "@flitter/data";
import type { GitService } from "@flitter/util";
import { discoverGuidanceFiles } from "@flitter/data";
import * as os from "node:os";
import * as path from "node:path";
import * as fs from "node:fs/promises";

// ─── 选项类型 ────────────────────────────────────────────

export interface ContextBlocksOptions {
  /** 配置服务 (用于读取 settings.systemPrompt) */
  configService: ConfigService;
  /** 技能服务 (用于获取可用技能列表) */
  skillService: SkillService;
  /** 工作区根目录 */
  workspaceRoot: string;
  /** 当前工作目录 */
  workingDirectory: string;
  /** Git 服务 (可选, 用于仓库信息) */
  gitService?: GitService;
  /** 取消信号 */
  signal?: AbortSignal;
}

// ─── 主函数 ──────────────────────────────────────────────

/**
 * 收集多个上下文段落, 每段为一个 SystemPromptBlock
 * 逆向: fwR (collectContextBlocks, tool-execution-engine.js 434-547)
 *
 * 收集顺序:
 * 1. 环境块: 今日日期, 工作目录, OS, 平台, git 状态
 * 2. Guidance 块: discoverGuidanceFiles → 每个文件包裹在 <instructions> 标签中
 * 3. Skills 块: SkillService.getSkills() 格式化为 bullet 列表
 * 4. 自定义系统提示词: config.settings.systemPrompt (如果存在)
 * 5. 仓库信息: git remote URL, 分支信息
 *
 * @param opts - 上下文收集选项
 * @returns 系统提示词块数组
 */
export async function collectContextBlocks(
  opts: ContextBlocksOptions,
): Promise<SystemPromptBlock[]> {
  const blocks: SystemPromptBlock[] = [];

  // ─── 1. 环境块 ──────────────────────────────────────
  blocks.push(buildEnvironmentBlock(opts));

  // ─── 2. Guidance 块 ─────────────────────────────────
  const guidanceBlocks = await buildGuidanceBlocks(opts);
  blocks.push(...guidanceBlocks);

  // ─── 3. Skills 块 ───────────────────────────────────
  const skillsBlock = await buildSkillsBlock(opts);
  if (skillsBlock) blocks.push(skillsBlock);

  // ─── 4. 自定义系统提示词 ────────────────────────────
  const customBlock = buildCustomSystemPromptBlock(opts);
  if (customBlock) blocks.push(customBlock);

  // ─── 5. 仓库信息 ────────────────────────────────────
  const repoBlock = await buildRepositoryBlock(opts);
  if (repoBlock) blocks.push(repoBlock);

  return blocks;
}

// ─── 内部构建函数 ────────────────────────────────────────

/**
 * 构建环境信息块
 * 包含: 今日日期, 工作目录, OS, 平台, git 仓库状态
 */
function buildEnvironmentBlock(opts: ContextBlocksOptions): SystemPromptBlock {
  const today = new Date().toISOString().split("T")[0]; // YYYY-MM-DD
  const platform = os.platform();
  const osVersion = os.release();

  const lines = [
    `Today's date: ${today}`,
    `Working directory: ${opts.workingDirectory}`,
    `Platform: ${platform}`,
    `OS Version: ${osVersion}`,
  ];

  // git 仓库状态: 是否在 git 仓库中
  const isGitRepo = opts.gitService !== undefined;
  lines.push(`Is directory a git repo: ${isGitRepo ? "Yes" : "No"}`);

  return {
    type: "text",
    text: lines.join("\n"),
  };
}

/**
 * 构建 guidance 块列表
 * 逆向: fwR ~475-510
 *
 * 通过 discoverGuidanceFiles 发现 guidance 文件,
 * 读取每个文件内容, 包裹在 <instructions> 标签中。
 *
 * 每个 guidance 文件生成一个独立的 SystemPromptBlock
 * (用于对重要文件设置 cache_control)
 */
async function buildGuidanceBlocks(
  opts: ContextBlocksOptions,
): Promise<SystemPromptBlock[]> {
  const blocks: SystemPromptBlock[] = [];

  try {
    const guidanceFiles = await discoverGuidanceFiles(
      opts.workspaceRoot,
      opts.workingDirectory,
    );

    for (const filePath of guidanceFiles) {
      // 尊重 AbortSignal
      if (opts.signal?.aborted) break;

      try {
        const content = await fs.readFile(filePath, "utf-8");
        if (content.trim().length === 0) continue;

        const relativePath = path.relative(opts.workspaceRoot, filePath);
        const wrappedText = [
          `<instructions source="${relativePath}">`,
          content.trim(),
          `</instructions>`,
        ].join("\n");

        blocks.push({
          type: "text",
          text: wrappedText,
          cache_control: { type: "ephemeral" },
        });
      } catch {
        // 跳过无法读取的文件
        continue;
      }
    }
  } catch {
    // discoverGuidanceFiles 失败时返回空
  }

  return blocks;
}

/**
 * 构建 skills 列表块
 * 逆向: fwR ~515-535
 *
 * 从 SkillService 获取可用技能, 格式化为 bullet 列表
 */
async function buildSkillsBlock(
  opts: ContextBlocksOptions,
): Promise<SystemPromptBlock | null> {
  try {
    const skills = await opts.skillService.getSkills();
    if (skills.length === 0) return null;

    const skillLines = skills.map(
      (skill) => `- ${skill.name}: ${skill.description ?? "No description"}`,
    );

    const text = [
      "Available skills:",
      ...skillLines,
    ].join("\n");

    return { type: "text", text };
  } catch {
    return null;
  }
}

/**
 * 构建自定义系统提示词块
 * 从 config.settings.systemPrompt 读取
 */
function buildCustomSystemPromptBlock(
  opts: ContextBlocksOptions,
): SystemPromptBlock | null {
  const config = opts.configService.getConfig();
  const customPrompt = config.settings.systemPrompt;

  if (!customPrompt || customPrompt.trim().length === 0) return null;

  return {
    type: "text",
    text: customPrompt.trim(),
  };
}

/**
 * 构建仓库信息块
 * 逆向: fwR ~540-547
 *
 * 包含 git remote URL 和当前分支信息
 */
async function buildRepositoryBlock(
  opts: ContextBlocksOptions,
): Promise<SystemPromptBlock | null> {
  if (!opts.gitService) return null;

  try {
    const lines: string[] = [];

    const remoteUrl = await opts.gitService.getRemoteUrl();
    if (remoteUrl) {
      lines.push(`Repository remote: ${remoteUrl}`);
    }

    const branch = await opts.gitService.getCurrentBranch();
    if (branch) {
      lines.push(`Current branch: ${branch}`);
    }

    if (lines.length === 0) return null;

    return {
      type: "text",
      text: lines.join("\n"),
    };
  } catch {
    return null;
  }
}
```

### 2. system-prompt.ts -- 最终提示词组装

```ts
import type { SystemPromptBlock } from "@flitter/llm";
import type { ToolDefinition } from "@flitter/llm";

// ─── 基础角色提示词 ──────────────────────────────────────

/**
 * 核心 agent 身份与行为指令
 * 逆向: LO ~600-650
 *
 * 包含:
 * - Agent 身份与能力描述
 * - 代码编辑指南
 * - 安全指令
 * - 通信风格
 */
export const BASE_ROLE_PROMPT = `You are an interactive CLI-based coding assistant. You help users with software engineering tasks including reading, writing, and editing code, running commands, searching codebases, and managing files.

Your strengths:
- Searching for code, configurations, and patterns across large codebases
- Analyzing multiple files to understand system architecture
- Investigating complex questions that require exploring many files
- Performing multi-step research tasks
- Writing, editing, and refactoring code across multiple files
- Running shell commands and interpreting their output

Guidelines:
- For file searches: Use Grep or Glob when you need to search broadly. Use Read when you know the specific file path.
- For analysis: Start broad and narrow down. Use multiple search strategies if the first doesn't yield results.
- Be thorough: Check multiple locations, consider different naming conventions, look for related files.
- NEVER create files unless they are absolutely necessary for achieving your goal. ALWAYS prefer editing an existing file to creating a new one.
- NEVER proactively create documentation files (*.md) or README files. Only create documentation files if explicitly requested.
- In your final response always share relevant file names and code snippets. Any file paths you return in your response MUST be absolute. Do NOT use relative paths.

Safety:
- NEVER execute destructive commands without explicit user confirmation.
- Be cautious with git operations that modify history (push --force, reset --hard, etc.)
- Do not commit changes unless the user explicitly asks.

Communication:
- Be concise and direct.
- When uncertain, ask for clarification rather than guessing.
- Show your reasoning for complex decisions.`;

// ─── 子代理约束提示词 ────────────────────────────────────

/**
 * 子代理模式附加约束
 * 逆向: LO ~780-816
 */
export const SUB_AGENT_MODIFIER = `You are operating as a sub-agent with a focused scope. Additional constraints:
- Do NOT spawn additional sub-agents or use the Task tool.
- Stay focused on the specific task assigned to you.
- Do not make assumptions about the broader context beyond your assigned task.
- Complete your work and return results promptly.`;

// ─── 选项类型 ────────────────────────────────────────────

export interface BuildSystemPromptOptions {
  /** 当前可用的工具定义列表 */
  toolDefinitions: ToolDefinition[];
  /** 预先收集的上下文块 (来自 collectContextBlocks) */
  contextBlocks: SystemPromptBlock[];
  /** 是否为子代理模式 */
  isSubAgent?: boolean;
}

// ─── 主函数 ──────────────────────────────────────────────

/**
 * 组装最终系统提示词
 * 逆向: LO (buildSystemPrompt, tool-execution-engine.js 591-816)
 *
 * 组装顺序:
 * 1. 基础角色提示词 (硬编码)
 * 2. 工具指令块 (格式化工具列表)
 * 3. 上下文块 (环境/guidance/skills/仓库)
 * 4. 子代理修饰 (如果 isSubAgent=true)
 *
 * 返回多段 SystemPromptBlock, 每段可独立设置 cache_control
 * 以便 LLM provider 在段级别缓存 (避免因工具列表变化导致缓存失效)
 *
 * @param opts - 构建选项
 * @returns 系统提示词块数组
 */
export function buildSystemPrompt(
  opts: BuildSystemPromptOptions,
): SystemPromptBlock[] {
  const blocks: SystemPromptBlock[] = [];

  // ─── 1. 基础角色提示词 ──────────────────────────────
  blocks.push({
    type: "text",
    text: BASE_ROLE_PROMPT,
    cache_control: { type: "ephemeral" },
  });

  // ─── 2. 工具指令块 ─────────────────────────────────
  const toolBlock = buildToolInstructionsBlock(opts.toolDefinitions);
  if (toolBlock) blocks.push(toolBlock);

  // ─── 3. 上下文块 ────────────────────────────────────
  blocks.push(...opts.contextBlocks);

  // ─── 4. 子代理修饰 ─────────────────────────────────
  if (opts.isSubAgent) {
    blocks.push({
      type: "text",
      text: SUB_AGENT_MODIFIER,
    });
  }

  return blocks;
}

// ─── 内部函数 ────────────────────────────────────────────

/**
 * 格式化工具列表为指令块
 * 逆向: LO ~660-720
 *
 * 生成每个工具的名称和描述列表,
 * 供 LLM 了解可用工具
 */
function buildToolInstructionsBlock(
  toolDefinitions: ToolDefinition[],
): SystemPromptBlock | null {
  if (toolDefinitions.length === 0) return null;

  const toolLines = toolDefinitions.map(
    (tool) => `- **${tool.name}**: ${tool.description}`,
  );

  const text = [
    "You have access to the following tools:",
    "",
    ...toolLines,
    "",
    "Use the appropriate tool for each task. You can call multiple tools in a single response when they are independent of each other.",
  ].join("\n");

  return {
    type: "text",
    text,
  };
}
```
