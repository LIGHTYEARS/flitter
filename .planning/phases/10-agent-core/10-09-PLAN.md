### 1. events.ts -- Agent 事件类型

```ts
import type { StreamDelta } from "@flitter/llm";
import type { ToolResult } from "../tools/types";

// ─── 推理状态 ────────────────────────────────────────────

/**
 * 推理状态
 * - idle: 空闲, 等待用户输入或操作
 * - running: 推理中 (LLM 流式输出或工具执行)
 * - cancelled: 用户取消了推理
 */
export type InferenceState = "idle" | "running" | "cancelled";

// ─── Agent 事件 ──────────────────────────────────────────

/**
 * Agent 事件: ThreadWorker 向外层 (TUI) 发出的事件流
 * 逆向: ov 内部事件分发
 *
 * 使用判别联合 (discriminated union), type 字段区分事件种类
 */
export type AgentEvent =
  | InferenceStartEvent
  | InferenceDeltaEvent
  | InferenceCompleteEvent
  | InferenceErrorEvent
  | ToolStartEvent
  | ToolDataEvent
  | ToolCompleteEvent
  | TurnCompleteEvent
  | CompactionStartEvent
  | CompactionCompleteEvent;

/** LLM 推理开始 */
export interface InferenceStartEvent {
  type: "inference:start";
}

/** LLM 流式增量输出 */
export interface InferenceDeltaEvent {
  type: "inference:delta";
  delta: StreamDelta;
}

/** LLM 推理完成 (单轮流式结束) */
export interface InferenceCompleteEvent {
  type: "inference:complete";
  usage?: { inputTokens: number; outputTokens: number };
}

/** LLM 推理错误 */
export interface InferenceErrorEvent {
  type: "inference:error";
  error: Error;
}

/** 工具开始执行 */
export interface ToolStartEvent {
  type: "tool:start";
  toolUseId: string;
  toolName: string;
}

/** 工具数据事件 (中间结果或最终结果) */
export interface ToolDataEvent {
  type: "tool:data";
  toolUseId: string;
  data: ToolResult;
}

/** 工具执行完成 */
export interface ToolCompleteEvent {
  type: "tool:complete";
  toolUseId: string;
}

/** 完整 turn 完成 (无更多 tool_use, agent 回复结束) */
export interface TurnCompleteEvent {
  type: "turn:complete";
}

/** 上下文压缩开始 */
export interface CompactionStartEvent {
  type: "compaction:start";
}

/** 上下文压缩完成 */
export interface CompactionCompleteEvent {
  type: "compaction:complete";
}
```

### 2. thread-worker.ts -- ThreadWorker 状态机

```ts
import type { LLMProvider, StreamParams, StreamDelta, SystemPromptBlock } from "@flitter/llm";
import type { ThreadStore, ConfigService, ContextManager } from "@flitter/data";
import type { Config, Settings } from "@flitter/schemas";
import type { AssistantContentBlock, Message } from "@flitter/schemas";
import type { BehaviorSubject, Subject, Subscription } from "@flitter/util";
import type { ToolOrchestrator, OrchestratorCallbacks, ToolUseItem } from "../tools/orchestrator";
import type { PermissionEngine } from "../permissions/engine";
import type { ToolRegistry } from "../tools/registry";
import type { ToolResult, ToolContext } from "../tools/types";
import type { AgentEvent, InferenceState } from "./events";

// ─── 工具审批响应 ────────────────────────────────────────

/**
 * 用户对工具审批请求的响应
 */
export interface ToolApprovalResponse {
  /** 是否批准 */
  approved: boolean;
  /** 如果批准, 是否记住此决策 (用于自动批准同类工具) */
  remember?: boolean;
}

// ─── ThreadWorker 选项 ───────────────────────────────────

export interface ThreadWorkerOptions {
  /** 线程存储 (读写当前线程的消息历史) */
  threadStore: ThreadStore;
  /** LLM 提供者 (流式推理) */
  provider: LLMProvider;
  /** 工具执行引擎 */
  toolOrchestrator: ToolOrchestrator;
  /** 权限引擎 */
  permissionEngine: PermissionEngine;
  /** 系统提示词构建回调 (异步, 每次推理前调用) */
  buildSystemPrompt: () => Promise<SystemPromptBlock[]>;
  /** 上下文管理器 (消息压缩) */
  contextManager: ContextManager;
  /** 配置服务 */
  configService: ConfigService;
  /** 工具注册表 */
  toolRegistry: ToolRegistry;
}

// ─── ThreadWorker 类 ────────────────────────────────────

/**
 * ThreadWorker: Agent 核心状态机
 * 逆向: ov (tool-execution-engine.js 2450-2876)
 *
 * 职责:
 * 1. 管理推理状态 (idle/running/cancelled)
 * 2. 驱动推理循环: 构建提示词 → LLM stream → 工具执行 → 递归推理
 * 3. 发出 AgentEvent 供 TUI 层消费
 * 4. 实现 OrchestratorCallbacks 供 ToolOrchestrator 回调
 * 5. 处理取消/重试/用户输入/审批
 *
 * 状态机流:
 * ┌─────────────────────────────────────────────────┐
 * │                                                 │
 * │  idle ──runInference()──→ running               │
 * │   ↑                         │                   │
 * │   │                    ┌────┴────┐              │
 * │   │                    ↓         ↓              │
 * │   │              (no tools)  (has tools)        │
 * │   │                    │         │              │
 * │   │              turn:complete   │              │
 * │   │                    │    ToolOrchestrator    │
 * │   │                    │         │              │
 * │   │                    │    recursive           │
 * │   │                    │    runInference()      │
 * │   │                    ↓         │              │
 * │   ├────────────────── idle ←─────┘              │
 * │   │                                             │
 * │   │  cancelInference()                          │
 * │   │     running → cancelled                     │
 * │   │                                             │
 * │   │  retry()                                    │
 * │   ├──── cancelled → idle → runInference()       │
 * │   │                                             │
 * │   │  error                                      │
 * │   ├──── running → idle + inference:error        │
 * └─────────────────────────────────────────────────┘
 */
export class ThreadWorker {
  // ─── Observable 状态 ──────────────────────────────

  /** 推理状态 (BehaviorSubject, 初始 "idle") */
  readonly inferenceState$: BehaviorSubject<InferenceState>;

  /** Agent 事件流 */
  readonly events$: Subject<AgentEvent>;

  // ─── 内部状态 ──────────────────────────────────────

  private readonly opts: ThreadWorkerOptions;
  private abortController: AbortController | null = null;
  private subscriptions: Subscription[] = [];
  private disposed = false;

  constructor(opts: ThreadWorkerOptions) {
    this.opts = opts;
    // 由调用方注入已创建的 BehaviorSubject 和 Subject, 或在此处创建
    // 这里假设从 @flitter/util 导入的 BehaviorSubject/Subject 构造函数
    // 实际创建方式依赖 @flitter/util 的 API
    this.inferenceState$ = null as any; // 实际实现: new BehaviorSubject<InferenceState>("idle")
    this.events$ = null as any; // 实际实现: new Subject<AgentEvent>()
  }

  // ─── 公共方法 ──────────────────────────────────────

  /**
   * 执行完整推理循环
   * 逆向: ov.runInference (~2520-2650)
   *
   * 流程:
   * 1. inferenceState → "running"
   * 2. 检查上下文压缩 (ContextManager.checkAndCompact)
   * 3. 构建系统提示词 (buildSystemPrompt 回调)
   * 4. 获取工具定义 (ToolRegistry.getToolDefinitions)
   * 5. 构建 StreamParams 并调用 provider.stream()
   * 6. 迭代 AsyncIterable<StreamDelta>:
   *    - 更新 ThreadStore 中的 assistant 消息内容
   *    - 发出 inference:delta 事件
   * 7. 流式完成后:
   *    - 发出 inference:complete 事件
   *    - 检查 assistant 消息中的 tool_use 块
   *    - 如果有 tool_use:
   *      a. 提取 ToolUseItem[]
   *      b. 调用 ToolOrchestrator.executeToolsWithPlan()
   *      c. 等待所有工具完成
   *      d. 递归调用 runInference() (多轮推理)
   *    - 如果没有 tool_use:
   *      a. 发出 turn:complete 事件
   * 8. inferenceState → "idle"
   *
   * 错误处理:
   * - catch: inferenceState → "idle", 发出 inference:error 事件
   */
  async runInference(): Promise<void> {
    if (this.disposed) return;

    try {
      this.abortController = new AbortController();
      const signal = this.abortController.signal;

      this.inferenceState$.next("running");
      this.events$.next({ type: "inference:start" });

      // ─── Step 1: 检查上下文压缩 ───────────────
      await this.checkCompaction();

      // ─── Step 2: 构建系统提示词 ───────────────
      const systemPrompt = await this.opts.buildSystemPrompt();

      // ─── Step 3: 获取工具定义 ──────────────────
      const config = this.opts.configService.getConfig();
      const toolDefs = this.opts.toolRegistry.getToolDefinitions(config.settings);

      // ─── Step 4: 构建 StreamParams ─────────────
      const messages = this.opts.threadStore.getMessages();
      const streamParams: StreamParams = {
        model: config.settings.model ?? "default",
        messages,
        systemPrompt,
        tools: toolDefs,
        config,
        signal,
      };

      // ─── Step 5: 流式推理 ──────────────────────
      let lastDelta: StreamDelta | null = null;
      const stream = this.opts.provider.stream(streamParams);

      for await (const delta of stream) {
        if (signal.aborted) break;

        lastDelta = delta;

        // 更新 ThreadStore 中的 assistant 消息
        this.opts.threadStore.updateAssistantMessage(delta.content);

        // 发出 delta 事件
        this.events$.next({ type: "inference:delta", delta });
      }

      if (signal.aborted) {
        this.inferenceState$.next("cancelled");
        return;
      }

      // ─── Step 6: 流式完成 ──────────────────────
      this.events$.next({
        type: "inference:complete",
        usage: lastDelta?.usage,
      });

      // ─── Step 7: 检查 tool_use ─────────────────
      const toolUses = this.extractToolUses();

      if (toolUses.length > 0) {
        // 有 tool_use: 交给 ToolOrchestrator 执行
        await this.opts.toolOrchestrator.executeToolsWithPlan(toolUses);

        // 递归推理 (多轮)
        await this.runInference();
      } else {
        // 无 tool_use: turn 完成
        this.events$.next({ type: "turn:complete" });
        this.inferenceState$.next("idle");
      }
    } catch (error) {
      this.inferenceState$.next("idle");
      this.events$.next({
        type: "inference:error",
        error: error instanceof Error ? error : new Error(String(error)),
      });
    }
  }

  /**
   * 取消当前推理
   * 逆向: ov.cancelInference (~2660-2680)
   *
   * 流程:
   * 1. 中止 AbortController
   * 2. 取消 ToolOrchestrator 中运行的工具
   * 3. inferenceState → "cancelled"
   */
  cancelInference(): void {
    if (this.abortController) {
      this.abortController.abort();
      this.abortController = null;
    }
    this.opts.toolOrchestrator.cancelAll();
    this.inferenceState$.next("cancelled");
  }

  /**
   * 重试上次失败/取消的推理
   * 逆向: ov.retry (~2690-2710)
   *
   * 流程:
   * 1. 如果当前状态为 "cancelled", 先转为 "idle"
   * 2. 调用 runInference()
   */
  async retry(): Promise<void> {
    const currentState = this.inferenceState$.getValue();
    if (currentState === "cancelled") {
      this.inferenceState$.next("idle");
    }
    await this.runInference();
  }

  /**
   * 用户为被阻塞的工具提供输入
   *
   * 当工具执行需要用户输入时 (如 Bash 交互式命令),
   * 工具会阻塞等待。此方法传递用户输入以恢复执行。
   *
   * @param toolUseId - 被阻塞工具的 ID
   * @param input - 用户输入的文本
   */
  async userProvideInput(toolUseId: string, input: string): Promise<void> {
    // 将用户输入传递给 ToolOrchestrator 中被阻塞的工具
    // 具体实现: 通过 ToolOrchestrator 的回调机制通知工具
    // 工具在 ToolContext.userInput 中接收输入
    void toolUseId;
    void input;
    // TODO: 实现 — 需要 ToolOrchestrator 支持 blocked 工具的输入通道
  }

  /**
   * 用户响应工具审批请求
   *
   * 当 PermissionEngine 决策为 "ask" 时, 工具执行被阻塞
   * 等待用户批准/拒绝。此方法传递用户决策。
   *
   * @param toolUseId - 等待审批的工具 ID
   * @param response - 用户的审批响应
   */
  async userRespondToApproval(
    toolUseId: string,
    response: ToolApprovalResponse,
  ): Promise<void> {
    void toolUseId;
    void response;
    // TODO: 实现 — 需要与 PermissionEngine 的审批流集成
  }

  /**
   * 销毁 ThreadWorker, 释放所有资源
   *
   * 1. 取消正在进行的推理
   * 2. 取消所有订阅
   * 3. 销毁 ToolOrchestrator
   */
  dispose(): void {
    if (this.disposed) return;
    this.disposed = true;

    // 取消推理
    if (this.abortController) {
      this.abortController.abort();
      this.abortController = null;
    }

    // 销毁 ToolOrchestrator
    this.opts.toolOrchestrator.dispose();

    // 取消所有订阅
    for (const sub of this.subscriptions) {
      sub.unsubscribe();
    }
    this.subscriptions = [];
  }

  // ─── 内部方法 ──────────────────────────────────────

  /**
   * 检查并执行上下文压缩
   * 当消息历史超过阈值时, 通过 ContextManager 压缩
   */
  private async checkCompaction(): Promise<void> {
    const shouldCompact = this.opts.contextManager.shouldCompact(
      this.opts.threadStore.getSnapshot(),
    );

    if (shouldCompact) {
      this.events$.next({ type: "compaction:start" });

      await this.opts.contextManager.checkAndCompact(
        this.opts.threadStore.getSnapshot(),
      );

      this.events$.next({ type: "compaction:complete" });
    }
  }

  /**
   * 从最新 assistant 消息中提取 tool_use 块
   * 逆向: ov ~2600-2610 中的 tool_use 检查
   *
   * @returns ToolUseItem[] — 工具调用列表 (空数组表示无工具调用)
   */
  private extractToolUses(): ToolUseItem[] {
    const snapshot = this.opts.threadStore.getSnapshot();
    const messages = snapshot.messages;

    if (messages.length === 0) return [];

    const lastMessage = messages[messages.length - 1];
    if (lastMessage.role !== "assistant") return [];

    const toolUses: ToolUseItem[] = [];

    for (const block of lastMessage.content) {
      if (block.type === "tool_use") {
        toolUses.push({
          id: block.id,
          name: block.name,
          input: block.input as Record<string, unknown>,
        });
      }
    }

    return toolUses;
  }
}
```
