### headless.ts — Headless JSON 流模式

```typescript
import type { ServiceContainer } from "@flitter/flitter";
import type { CliContext } from "../context";
import type { AgentEvent } from "@flitter/agent-core";
import { createInterface } from "node:readline";

/**
 * Headless JSON 流模式
 *
 * 输入: stdin JSON Lines — 每行 { role: "user", content: "..." }
 * 输出: stdout JSON Lines — 每行 AgentEvent
 *
 * 逆向: SB() stream-json 分支 in cli-entrypoint.js
 */
export async function runHeadlessMode(
  container: ServiceContainer,
  context: CliContext,
): Promise<void> {
  const threadId = crypto.randomUUID();
  const worker = container.createThreadWorker(threadId);

  // 订阅 AgentEvent → JSON Lines 输出到 stdout
  const sub = worker.agentEvents$.subscribe((event: AgentEvent) => {
    writeJsonLine(process.stdout, event);
  });

  try {
    // 如果有初始消息 (命令行参数)
    if (context.userMessage) {
      await worker.submitUserMessage(context.userMessage);
      await worker.waitForIdle();
    }

    // 持续读 stdin JSON Lines
    const rl = createInterface({ input: process.stdin });
    for await (const line of rl) {
      const trimmed = line.trim();
      if (!trimmed) continue;
      try {
        const msg = JSON.parse(trimmed);
        if (msg.role === "user" && typeof msg.content === "string") {
          await worker.submitUserMessage(msg.content);
          await worker.waitForIdle();
        }
      } catch {
        process.stderr.write(`Warning: invalid JSON line: ${trimmed}\n`);
      }
    }
  } finally {
    sub.unsubscribe();
    await container.asyncDispose();
  }
}

function writeJsonLine(stream: NodeJS.WritableStream, data: unknown): void {
  stream.write(JSON.stringify(data) + "\n");
}
```

### execute.ts — 非交互式执行模式

```typescript
/**
 * 非交互式执行模式
 *
 * 逆向: SB() execute 分支 in cli-entrypoint.js
 */
export async function runExecuteMode(
  container: ServiceContainer,
  context: CliContext,
): Promise<void> {
  // 获取用户消息
  let userMessage = context.userMessage;
  if (!userMessage) {
    // stdin pipe mode: 读取全部 stdin 内容
    userMessage = await readStdin();
  }
  if (!userMessage) {
    process.stderr.write("Error: no input provided\n");
    process.exitCode = 1;
    return;
  }

  const threadId = crypto.randomUUID();
  const worker = container.createThreadWorker(threadId);

  try {
    // 可选: stream-json 模式
    let sub: Subscription | null = null;
    if (context.streamJson) {
      sub = worker.agentEvents$.subscribe((event) => {
        writeJsonLine(process.stdout, event);
      });
    }

    await worker.submitUserMessage(userMessage);
    await worker.waitForIdle();

    // 非 stream-json: 输出最终文本
    if (!context.streamJson) {
      const thread = container.threadStore.getThreadSnapshot(threadId);
      const lastAssistant = thread?.messages
        .filter(m => m.role === "assistant")
        .pop();
      if (lastAssistant) {
        const text = extractText(lastAssistant);
        process.stdout.write(text + "\n");
      }
    }

    sub?.unsubscribe();
  } finally {
    await container.asyncDispose();
  }
}
```
