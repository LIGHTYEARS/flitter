## 参考实现

### api-key.ts — API Key 认证

```typescript
import type { SecretStorage } from "@flitter/flitter";
import { createInterface } from "node:readline/promises";

const API_KEY_PREFIXES = ["sk-", "flitter-"];

/**
 * 检查是否已有 API Key
 */
export async function hasApiKey(secrets: SecretStorage, ampURL: string): Promise<boolean> {
  const key = await secrets.get("apiKey", ampURL);
  return Boolean(key);
}

/**
 * 验证 API Key 格式
 */
export function validateApiKey(key: string): boolean {
  if (!key || key.trim().length === 0) return false;
  return API_KEY_PREFIXES.some(prefix => key.startsWith(prefix));
}

/**
 * 从环境变量获取 API Key
 */
export function getApiKeyFromEnv(): string | undefined {
  return process.env.FLITTER_API_KEY;
}

/**
 * 交互式输入 API Key
 */
export async function promptApiKey(): Promise<string | undefined> {
  const rl = createInterface({ input: process.stdin, output: process.stderr });
  try {
    const key = await rl.question("Enter your API key: ");
    return key.trim() || undefined;
  } finally {
    rl.close();
  }
}

/**
 * 存储 API Key
 */
export async function storeApiKey(secrets: SecretStorage, ampURL: string, key: string): Promise<void> {
  await secrets.set("apiKey", key, ampURL);
}
```

### oauth.ts — OAuth PKCE 流程

```typescript
import { createServer } from "node:http";
import { generatePKCE } from "@flitter/llm";

const CALLBACK_PORT_START = 49152;
const CALLBACK_PORT_RETRIES = 10;

export interface OAuthResult {
  success: boolean;
  token?: string;
  error?: string;
}

/**
 * 执行 OAuth PKCE 认证流程
 *
 * 1. 生成 PKCE challenge
 * 2. 打开浏览器到授权 URL
 * 3. 启动本地 HTTP 服务器接收回调
 * 4. 验证 state, 交换 code → token
 * 5. 关闭服务器, 返回 token
 */
export async function performOAuth(ampURL: string): Promise<OAuthResult> {
  const { codeVerifier, codeChallenge } = await generatePKCE();
  const state = crypto.randomUUID();

  // 启动回调服务器
  const { server, port } = await startCallbackServer(CALLBACK_PORT_START);
  const redirectUri = `http://localhost:${port}/callback`;

  // 构建授权 URL
  const authUrl = buildAuthUrl(ampURL, {
    codeChallenge,
    state,
    redirectUri,
  });

  // 打开浏览器
  await openBrowser(authUrl);
  process.stderr.write(`Opening browser for authentication...\n`);
  process.stderr.write(`If browser doesn't open, visit: ${authUrl}\n`);

  // 等待回调
  const result = await waitForCallback(server, state);

  // 关闭服务器
  server.close();

  if (!result.code) return { success: false, error: result.error || "No code received" };

  // 交换 code → token
  const token = await exchangeCodeForToken(ampURL, result.code, codeVerifier, redirectUri);
  return { success: true, token };
}

async function startCallbackServer(startPort: number): Promise<{ server: http.Server; port: number }> {
  for (let port = startPort; port < startPort + CALLBACK_PORT_RETRIES; port++) {
    try {
      const server = createServer();
      await new Promise<void>((resolve, reject) => {
        server.listen(port, "127.0.0.1", resolve);
        server.on("error", reject);
      });
      return { server, port };
    } catch { continue; }
  }
  throw new Error(`Failed to bind callback server on ports ${startPort}-${startPort + CALLBACK_PORT_RETRIES - 1}`);
}
```
