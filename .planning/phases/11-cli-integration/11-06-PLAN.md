### container.ts — ServiceContainer 创建

```typescript
import type { ConfigService, ThreadStore, ThreadPersistence, SkillService,
  GuidanceLoader, ContextManager, FileSettingsStorage } from "@flitter/data";
import type { MCPServerManager } from "@flitter/llm";
import type { ToolRegistry, ToolOrchestrator, PermissionEngine,
  ThreadWorker, ThreadWorkerOptions } from "@flitter/agent-core";
import { Logger } from "@flitter/util";

const log = Logger.create("container");

/** 秘密存储接口 (API Key, OAuth Token 等) */
export interface SecretStorage {
  get(key: string, scope?: string): Promise<string | undefined>;
  set(key: string, value: string, scope?: string): Promise<void>;
  delete(key: string, scope?: string): Promise<void>;
}

/** 容器创建选项 */
export interface ContainerOptions {
  /** Flitter API 服务器 URL */
  ampURL: string;
  /** 用户设置存储 */
  settings: FileSettingsStorage;
  /** 秘密存储 (API Key 等) */
  secrets: SecretStorage;
  /** 是否延迟认证 (TUI 模式下先启动再登录) */
  deferAuth?: boolean;
  /** 工作目录 */
  workspaceRoot: string;
  /** 数据目录 (thread 持久化路径, 可选) */
  dataDir?: string;
  /** 用户主目录 */
  homeDir?: string;
  /** 用户配置目录 (~/.config/flitter) */
  configDir?: string;
}

/** 服务容器 — 包含所有已初始化的服务实例 */
export interface ServiceContainer {
  configService: ConfigService;
  toolRegistry: ToolRegistry;
  toolOrchestrator: ToolOrchestrator;
  permissionEngine: PermissionEngine;
  mcpServerManager: MCPServerManager;
  skillService: SkillService;
  threadStore: ThreadStore;
  threadPersistence: ThreadPersistence | null;
  guidanceLoader: GuidanceLoader;
  contextManager: ContextManager;
  secrets: SecretStorage;
  settings: FileSettingsStorage;

  /** 创建 ThreadWorker 实例 (工厂) */
  createThreadWorker(threadId: string, opts?: Partial<ThreadWorkerOptions>): ThreadWorker;

  /** 异步清理所有资源 */
  asyncDispose(): Promise<void>;
}

/**
 * 创建服务容器 — 组装所有服务并连接依赖
 *
 * 逆向: X3() in claude-config-system.js:1097-1324
 */
export async function createContainer(opts: ContainerOptions): Promise<ServiceContainer> {
  log.info("Initializing service container...", { ampURL: opts.ampURL });
  const disposables: Array<{ dispose(): void | Promise<void> }> = [];

  try {
    // 1. ConfigService
    const configService = createConfigService(opts);
    disposables.push(configService);

    // 2. ToolRegistry + 注册内置工具
    const toolRegistry = createToolRegistry();
    registerBuiltinTools(toolRegistry, opts);

    // 3. PermissionEngine
    const permissionEngine = createPermissionEngine(configService);

    // 4. ToolOrchestrator
    const toolOrchestrator = createToolOrchestrator(toolRegistry, permissionEngine, configService);
    disposables.push(toolOrchestrator);

    // 5. MCPServerManager
    const mcpServerManager = createMCPServerManager(configService);
    disposables.push(mcpServerManager);

    // 6. SkillService
    const skillService = createSkillService(configService);

    // 7. GuidanceLoader
    const guidanceLoader = createGuidanceLoader(opts);

    // 8. ThreadStore + ThreadPersistence
    const threadStore = createThreadStore(configService);
    disposables.push(threadStore);
    const threadPersistence = opts.dataDir ? createThreadPersistence(threadStore, opts) : null;
    if (threadPersistence) disposables.push(threadPersistence);

    // 9. ContextManager
    const contextManager = createContextManager(configService);

    log.info("Service container initialized successfully.");

    return {
      configService, toolRegistry, toolOrchestrator, permissionEngine,
      mcpServerManager, skillService, threadStore, threadPersistence,
      guidanceLoader, contextManager,
      secrets: opts.secrets, settings: opts.settings,

      createThreadWorker(threadId, workerOpts) {
        return createWorker(threadId, { /* 注入所有回调 */ ...workerOpts });
      },

      async asyncDispose() {
        log.info("Disposing service container...");
        // 逆向 X3 的 asyncDispose: 反序清理
        for (let i = disposables.length - 1; i >= 0; i--) {
          try { await disposables[i].dispose(); }
          catch (err) { log.warn("Dispose error", { error: err }); }
        }
        log.info("Service container disposed.");
      },
    };
  } catch (err) {
    // Partial failure: 清理已创建的服务
    log.error("Container initialization failed, cleaning up...", { error: err });
    for (let i = disposables.length - 1; i >= 0; i--) {
      try { await disposables[i].dispose(); } catch {}
    }
    throw err;
  }
}
```

### factory.ts — 各服务工厂函数

// 各 create* 函数封装创建逻辑
// 保持与各包的耦合集中在此文件

export function createConfigService(opts: ContainerOptions): ConfigService {
  return ConfigService.create({
    storage: opts.settings,
    workspaceRoot: opts.workspaceRoot,
    homeDir: opts.homeDir,
    configDir: opts.configDir,
  });
}

export function createToolRegistry(): ToolRegistry {
  return new ToolRegistry();
}

export function registerBuiltinTools(registry: ToolRegistry, opts: ContainerOptions): void {
  registry.register(new ReadTool());
  registry.register(new WriteTool());
  registry.register(new EditTool());
  registry.register(new BashTool());
  registry.register(new GrepTool());
  registry.register(new GlobTool());
  registry.register(new FuzzyFindTool());
}
