### main.ts — CLI 主入口

```typescript
import { createProgram, resolveCliContext } from "./program";
import { createContainer } from "@flitter/flitter";
import { launchInteractiveMode } from "./modes/interactive";
import { runExecuteMode } from "./modes/execute";
import { runHeadlessMode } from "./modes/headless";
import { handleLogin, handleLogout } from "./commands/auth";
import { handleUpdateCommand } from "./commands/update";
import { Logger } from "@flitter/util";

const log = Logger.create("cli");

/**
 * Flitter CLI 主入口
 *
 * 逆向: aF0() in cli-entrypoint.js:1013-1031
 *
 * 流程:
 * 1. 全局错误处理注册
 * 2. 初始化日志
 * 3. 解析命令行 (Commander.js)
 * 4. 路由到子命令或默认动作
 * 5. 创建 ServiceContainer
 * 6. 执行 (interactive / execute / headless)
 * 7. 清理 (asyncDispose)
 */
export async function main(): Promise<void> {
  // 1. 全局错误处理
  process.on("unhandledRejection", (err) => {
    log.error("Unhandled rejection", { error: err });
    process.exitCode = 2;
  });

  // 2. SIGINT/SIGTERM 信号处理
  let disposing = false;
  const handleSignal = async () => {
    if (disposing) return; // 防止重入
    disposing = true;
    log.info("Signal received, shutting down...");
    process.exitCode = 130; // Standard SIGINT exit code
  };
  process.on("SIGINT", handleSignal);
  process.on("SIGTERM", handleSignal);

  try {
    // 3. 版本和日志
    const version = getVersion();
    log.info("Starting Flitter CLI", { version });

    // 4. 创建 Commander 程序
    const program = createProgram(version);

    // 5. 注册子命令 action handlers
    registerCommandHandlers(program);

    // 6. 解析并执行
    await program.parseAsync(process.argv);

  } catch (err) {
    if (err instanceof Error) {
      process.stderr.write(`Error: ${err.message}\n`);
      log.error("CLI error", { error: err });
    }
    process.exitCode = process.exitCode || 1;
  }
}

function getVersion(): string {
  // 从 package.json 读取版本
  try {
    return require("../../package.json").version;
  } catch {
    return "0.0.0-dev";
  }
}
```
