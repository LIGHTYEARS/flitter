### inherited-widget.ts — InheritedWidget 抽象类

```typescript
import type { Widget, Element } from "./element.js";
import { InheritedElement } from "./inherited-element.js";

/**
 * InheritedWidget — 上下文数据注入的核心 Widget。
 *
 * 逆向: tui-widget-framework.js 中 Element.dependOnInheritedWidgetOfExactType
 * 消费侧通过 context.dependOnInheritedWidgetOfExactType(MyWidget) 订阅数据变更。
 *
 * @example
 * ```ts
 * class MyTheme extends InheritedWidget {
 *   readonly color: string;
 *   constructor(opts: { color: string; child: Widget }) {
 *     super({ child: opts.child });
 *     this.color = opts.color;
 *   }
 *   updateShouldNotify(old: MyTheme): boolean {
 *     return this.color !== old.color;
 *   }
 * }
 * ```
 */
export abstract class InheritedWidget implements Widget {
  readonly key: undefined;
  readonly child: Widget;

  constructor(opts: { child: Widget; key?: undefined }) {
    this.child = opts.child;
    this.key = opts.key;
  }

  canUpdate(other: Widget): boolean {
    return other.constructor === this.constructor && other.key === this.key;
  }

  createElement(): InheritedElement {
    return new InheritedElement(this);
  }

  abstract updateShouldNotify(oldWidget: InheritedWidget): boolean;
}
```

### inherited-element.ts — InheritedElement

```typescript
import { Element } from "./element.js";
import type { InheritedWidget } from "./inherited-widget.js";

/**
 * InheritedElement — 维护 dependent 集合，数据变更时通知依赖方。
 *
 * 逆向: tui-widget-framework.js addDependent/removeDependent
 */
export class InheritedElement extends Element {
  private _dependents: Set<Element> = new Set();

  constructor(widget: InheritedWidget) {
    super(widget);
  }

  override get widget(): InheritedWidget {
    return super.widget as InheritedWidget;
  }

  addDependent(dependent: Element): void {
    this._dependents.add(dependent);
  }

  removeDependent(dependent: Element): void {
    this._dependents.delete(dependent);
  }

  override mount(parent?: Element): void {
    super.mount(parent);
    const childElement = this.widget.child.createElement();
    this.addChild(childElement);
    childElement.mount(this);
  }

  override update(newWidget: InheritedWidget): void {
    const oldWidget = this.widget;
    super.update(newWidget);
    if (newWidget.updateShouldNotify(oldWidget)) {
      this._notifyDependents();
    }
    // 更新子 Widget
    if (this.children.length > 0) {
      const child = this.children[0]!;
      if (child.widget.canUpdate(newWidget.child)) {
        child.update(newWidget.child);
      }
    }
  }

  override performRebuild(): void {
    super.performRebuild();
  }

  private _notifyDependents(): void {
    for (const dependent of this._dependents) {
      dependent.markNeedsRebuild();
    }
  }
}
```

### element.ts — patch: 新增 _inheritedDependencies 和 dependOnInheritedWidgetOfExactType

```typescript
// 在 Element 类中新增:

/** InheritedElement 依赖集合，unmount 时自动清除 */
protected _inheritedDependencies: Set<Element> = new Set();

/**
 * 向上查找指定 Widget 类型的 InheritedElement 并注册依赖。
 *
 * 逆向: tui-widget-framework.js:1753-1766
 */
dependOnInheritedWidgetOfExactType(widgetType: Function): Element | null {
  let current = this._parent;
  while (current) {
    if (current.widget.constructor === widgetType) {
      if ("addDependent" in current && "removeDependent" in current) {
        const inherited = current as Element & {
          addDependent(e: Element): void;
          removeDependent(e: Element): void;
        };
        inherited.addDependent(this);
        this._inheritedDependencies.add(inherited);
      }
      return current;
    }
    current = current.parent ?? undefined;
  }
  return null;
}

// 修改 unmount() — 追加清除逻辑:
override unmount(): void {
  for (const dep of this._inheritedDependencies) {
    if ("removeDependent" in dep) {
      (dep as Element & { removeDependent(e: Element): void }).removeDependent(this);
    }
  }
  this._inheritedDependencies.clear();
  // ... existing unmount logic
}
```
