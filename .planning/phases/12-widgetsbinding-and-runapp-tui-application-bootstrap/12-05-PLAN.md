### tui-controller.ts

```typescript
import { InputParser, type KeyEvent, type MouseEvent, type PasteEvent } from "../vt/input-parser.js";
import { Screen } from "../screen/screen.js";
import { Renderer } from "../screen/renderer.js";

/**
 * 终端能力信息。
 */
export interface TerminalCapabilities {
  emojiWidth: boolean;
  syncOutput: boolean;
  kittyKeyboard: boolean;
  colorPaletteNotifications: boolean;
  xtversion: string | null;
}

/**
 * 终端尺寸。
 */
export interface TerminalSize {
  width: number;
  height: number;
}

/**
 * 能力变化事件。
 */
export interface CapabilityEvent {
  capabilities: TerminalCapabilities;
}

/**
 * TuiController — 终端控制器。
 *
 * 逆向: XXT in clipboard-and-input.js:511-620
 *
 * 管理终端 raw mode、alt screen、stdin 读取、鼠标/键盘/粘贴事件分发、
 * 终端能力检测、信号处理 (SIGWINCH, SIGINT, SIGTERM, SIGTSTP)。
 */
export class TuiController {
  private parser: InputParser | null = null;
  private initialized = false;
  private inAltScreen = false;
  private screen: Screen;
  private renderer: Renderer;
  private capabilities: TerminalCapabilities | null = null;
  private capabilityResolve: (() => void) | null = null;
  private capabilityTimeout: ReturnType<typeof setTimeout> | null = null;

  private keyHandlers: ((event: KeyEvent) => void)[] = [];
  private mouseHandlers: ((event: MouseEvent) => void)[] = [];
  private resizeHandlers: ((event: TerminalSize) => void)[] = [];
  private pasteHandlers: ((event: PasteEvent) => void)[] = [];
  private capabilityHandlers: ((event: CapabilityEvent) => void)[] = [];

  private boundHandleResize = this.handleResize.bind(this);
  private boundCleanup = this.cleanup.bind(this);

  constructor() {
    this.screen = new Screen(80, 24);
    this.renderer = new Renderer();
  }

  init(): void {
    if (this.initialized) throw Error("TUI is already initialized");
    try {
      this.parser = new InputParser();
      this.parser.onKey((event) => this.handleKeyEvent(event));
      this.parser.onMouse((event) => {
        for (const handler of this.mouseHandlers) handler(event);
      });
      this.parser.onPaste((event) => {
        for (const handler of this.pasteHandlers) handler(event);
      });

      if (process.stdin.isTTY) {
        process.stdin.setRawMode(true);
        process.stdin.resume();
        process.stdin.setEncoding("utf8");
      }
      process.stdin.on("data", (data: Buffer | string) => {
        const buf = typeof data === "string" ? Buffer.from(data) : data;
        this.parser?.feed(buf);
      });

      process.on("SIGWINCH", this.boundHandleResize);
      if (process.stdout.isTTY) {
        process.stdout.on("resize", this.boundHandleResize);
      }

      this.updateTerminalSize();
      this.screen.resize(this.getSize().width, this.getSize().height);
      this.setupCleanupHandlers();
      this.initialized = true;
      this.enableMouse();
      this.enableBracketedPaste();
    } catch (error) {
      this.deinit();
      throw error;
    }
  }

  async deinit(): Promise<void> {
    if (this.initialized) {
      let seq = "";
      seq += this.renderer.disableMouse();
      seq += this.renderer.disableBracketedPaste();
      seq += this.renderer.showCursor();
      if (this.inAltScreen) {
        seq += this.renderer.exitAltScreen();
        this.inAltScreen = false;
      }
      process.stdout.write(seq);
    }
    if (this.capabilityTimeout) clearTimeout(this.capabilityTimeout);
    this.keyHandlers.length = 0;
    this.mouseHandlers.length = 0;
    this.resizeHandlers.length = 0;
    this.pasteHandlers.length = 0;
    this.capabilityHandlers.length = 0;
    process.removeListener("SIGWINCH", this.boundHandleResize);
    process.stdout.removeListener("resize", this.boundHandleResize);
    process.removeListener("SIGINT", this.boundCleanup);
    process.removeListener("SIGTERM", this.boundCleanup);
    this.parser = null;
    this.capabilities = null;
    this.initialized = false;
  }

  enterAltScreen(): void {
    if (!this.inAltScreen) {
      process.stdout.write(this.renderer.enterAltScreen());
      this.inAltScreen = true;
    }
  }

  getSize(): TerminalSize {
    return {
      width: process.stdout.columns ?? 80,
      height: process.stdout.rows ?? 24,
    };
  }

  getScreen(): Screen { return this.screen; }
  getCapabilities(): TerminalCapabilities | null { return this.capabilities; }

  onKey(handler: (event: KeyEvent) => void): void { this.keyHandlers.push(handler); }
  offKey(handler: (event: KeyEvent) => void): void {
    const idx = this.keyHandlers.indexOf(handler);
    if (idx !== -1) this.keyHandlers.splice(idx, 1);
  }
  onMouse(handler: (event: MouseEvent) => void): void { this.mouseHandlers.push(handler); }
  onResize(handler: (event: TerminalSize) => void): void { this.resizeHandlers.push(handler); }
  onPaste(handler: (event: PasteEvent) => void): void { this.pasteHandlers.push(handler); }
  onCapabilities(handler: (event: CapabilityEvent) => void): void { this.capabilityHandlers.push(handler); }

  render(): void {
    const diff = this.screen.renderDiff(this.renderer);
    if (diff) process.stdout.write(diff);
  }

  async waitForCapabilities(timeout: number): Promise<void> {
    return new Promise<void>((resolve) => {
      if (this.capabilities) { resolve(); return; }
      this.capabilityResolve = resolve;
      this.capabilityTimeout = setTimeout(() => {
        this.capabilities ??= this.defaultCapabilities();
        resolve();
      }, timeout);
    });
  }

  handleSuspend(): void {
    this.deinit();
    process.kill(process.pid, "SIGTSTP");
  }

  // ... private helpers: handleResize, handleKeyEvent, updateTerminalSize,
  // setupCleanupHandlers, enableMouse, enableBracketedPaste, cleanup, defaultCapabilities
}
```
