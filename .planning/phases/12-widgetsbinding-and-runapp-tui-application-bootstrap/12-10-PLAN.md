## 参考实现

### theme-controller.ts

```typescript
import { InheritedWidget } from "@flitter/tui";
import type { Widget, Element } from "@flitter/tui";

/**
 * 主题数据。
 */
export interface ThemeData {
  name: string;
  primary: string;
  secondary: string;
  surface: string;
  background: string;
  error: string;
  text: string;
  mutedText: string;
  border: string;
  accent: string;
  success: string;
  warning: string;
}

/**
 * ThemeController — 通过 InheritedWidget 注入主题。
 *
 * @example
 * ```ts
 * const theme = ThemeController.of(context);
 * new Text({ text: "hello", style: new TextStyle({ color: theme.primary }) });
 * ```
 */
export class ThemeController extends InheritedWidget {
  readonly data: ThemeData;

  constructor(opts: { data: ThemeData; child: Widget }) {
    super({ child: opts.child });
    this.data = opts.data;
  }

  updateShouldNotify(oldWidget: ThemeController): boolean {
    return this.data !== oldWidget.data;
  }

  static of(context: Element): ThemeData {
    const element = context.dependOnInheritedWidgetOfExactType(ThemeController);
    if (!element) throw Error("ThemeController not found in ancestor tree");
    return (element.widget as ThemeController).data;
  }
}
```

### config-provider.ts

```typescript
import { InheritedWidget } from "@flitter/tui";
import type { Widget, Element } from "@flitter/tui";

/**
 * ConfigProvider — 通过 InheritedWidget 注入配置服务。
 */
export class ConfigProvider extends InheritedWidget {
  readonly configService: unknown; // 实际类型从 @flitter/data 导入

  constructor(opts: { configService: unknown; child: Widget }) {
    super({ child: opts.child });
    this.configService = opts.configService;
  }

  updateShouldNotify(oldWidget: ConfigProvider): boolean {
    return this.configService !== oldWidget.configService;
  }

  static of(context: Element): unknown {
    const element = context.dependOnInheritedWidgetOfExactType(ConfigProvider);
    if (!element) throw Error("ConfigProvider not found in ancestor tree");
    return (element.widget as ConfigProvider).configService;
  }
}
```
