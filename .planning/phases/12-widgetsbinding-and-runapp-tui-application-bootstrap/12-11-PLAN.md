### app-widget.ts

```typescript
import { StatefulWidget, type State } from "@flitter/tui";
import type { Widget, Element } from "@flitter/tui";
import { ThemeController, type ThemeData } from "./theme-controller.js";
import { ConfigProvider } from "./config-provider.js";

/**
 * AppWidget 配置。
 */
export interface AppWidgetConfig {
  themeData: ThemeData;
  configService: unknown;
  child: Widget;
}

/**
 * AppWidget — 应用根 Widget。
 *
 * 构建 ThemeController → ConfigProvider → child 的 InheritedWidget 嵌套。
 */
export class AppWidget extends StatefulWidget {
  readonly config: AppWidgetConfig;

  constructor(config: AppWidgetConfig) {
    super();
    this.config = config;
  }

  createState(): AppWidgetState {
    return new AppWidgetState();
  }
}

/**
 * AppWidgetState — 应用级状态。
 */
export class AppWidgetState implements State<AppWidget> {
  widget!: AppWidget;
  context!: { element: Element };
  private _mounted = false;

  get mounted(): boolean { return this._mounted; }

  initState(): void {}
  didUpdateWidget(_oldWidget: AppWidget): void {}

  dispose(): void {}

  setState(fn?: () => void): void {
    if (!this._mounted) throw Error("setState() called after dispose()");
    if (fn) fn();
    if ("markNeedsBuild" in this.context.element) {
      (this.context.element as Element & { markNeedsBuild(): void }).markNeedsBuild();
    }
  }

  build(): Widget {
    return new ThemeController({
      data: this.widget.config.themeData,
      child: new ConfigProvider({
        configService: this.widget.config.configService,
        child: this.widget.config.child,
      }),
    });
  }
}
```
