### input-field.ts

```typescript
import { StatefulWidget, type State } from "@flitter/tui";
import type { Widget, Element } from "@flitter/tui";
import { TextEditingController } from "@flitter/tui";
import { FocusNode, type KeyEventResult } from "@flitter/tui";
import { FocusManager } from "@flitter/tui";
import type { KeyEvent, PasteEvent } from "@flitter/tui";

/**
 * InputField 配置。
 */
export interface InputFieldConfig {
  onSubmit: (text: string) => void;
  placeholder?: string;
}

/**
 * InputField — 文本输入框 Widget。
 *
 * 集成 FocusNode + TextEditingController，处理键盘和粘贴输入。
 * Enter 键触发 onSubmit 回调。
 */
export class InputField extends StatefulWidget {
  readonly config: InputFieldConfig;

  constructor(config: InputFieldConfig) {
    super();
    this.config = config;
  }

  createState(): InputFieldState {
    return new InputFieldState();
  }
}

export class InputFieldState implements State<InputField> {
  widget!: InputField;
  context!: { element: Element };
  private _mounted = false;
  private _focusNode!: FocusNode;
  private _controller!: TextEditingController;

  get mounted(): boolean { return this._mounted; }

  initState(): void {
    this._controller = new TextEditingController();
    this._focusNode = new FocusNode({
      debugLabel: "InputField",
      onKey: (event) => this._handleKeyEvent(event),
      onPaste: (event) => this._handlePasteEvent(event),
    });
    FocusManager.instance.registerNode(this._focusNode);
    this._focusNode.requestFocus();
  }

  didUpdateWidget(_oldWidget: InputField): void {}

  dispose(): void {
    FocusManager.instance.unregisterNode(this._focusNode);
    this._focusNode.dispose();
    this._controller.dispose();
    this._mounted = false;
  }

  setState(fn?: () => void): void {
    if (!this._mounted) throw Error("setState() called after dispose()");
    if (fn) fn();
  }

  build(): Widget {
    // 返回 Text widget 显示当前输入
    // 实际实现会渲染光标和输入文本
    return { key: undefined, canUpdate: () => true, createElement: () => this.context.element } as Widget;
  }

  private _handleKeyEvent(event: KeyEvent): KeyEventResult {
    if (event.key === "Enter" && !event.shiftKey) {
      const text = this._controller.text;
      if (text.trim()) {
        this._controller.clear();
        this.widget.config.onSubmit(text);
      }
      return "handled";
    }
    if (event.key === "Backspace") {
      this._controller.deleteBackward();
      this.setState();
      return "handled";
    }
    if (event.key.length === 1 && !event.ctrlKey && !event.metaKey) {
      this._controller.insertText(event.key);
      this.setState();
      return "handled";
    }
    return "ignored";
  }

  private _handlePasteEvent(event: PasteEvent): KeyEventResult {
    this._controller.insertText(event.text);
    this.setState();
    return "handled";
  }
}
```

### conversation-view.ts

```typescript
import { StatefulWidget, type State } from "@flitter/tui";
import type { Widget, Element } from "@flitter/tui";

/**
 * 消息类型。
 */
export interface Message {
  role: "user" | "assistant" | "system";
  content: string;
}

/**
 * ConversationView 配置。
 */
export interface ConversationViewConfig {
  messages: Message[];
}

/**
 * ConversationView — 消息列表显示 Widget。
 *
 * 最小实现: 渲染消息列表，支持滚动。
 */
export class ConversationView extends StatefulWidget {
  readonly config: ConversationViewConfig;

  constructor(config: ConversationViewConfig) {
    super();
    this.config = config;
  }

  createState(): ConversationViewState {
    return new ConversationViewState();
  }
}

export class ConversationViewState implements State<ConversationView> {
  widget!: ConversationView;
  context!: { element: Element };
  private _mounted = false;

  get mounted(): boolean { return this._mounted; }
  initState(): void {}
  didUpdateWidget(_oldWidget: ConversationView): void {}
  dispose(): void { this._mounted = false; }
  setState(fn?: () => void): void {
    if (!this._mounted) throw Error("setState() called after dispose()");
    if (fn) fn();
  }

  build(): Widget {
    // 最小实现: 渲染消息文本列表
    return { key: undefined, canUpdate: () => true, createElement: () => this.context.element } as Widget;
  }
}
```
