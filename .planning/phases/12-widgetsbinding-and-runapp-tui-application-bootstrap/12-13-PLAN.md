### interactive.ts — 关键变更

```typescript
// 删除以下 stub:
// - function _runApp()
// - const ThemeController = { ... }
// - const ConfigProvider = { ... }
// - class AppWidget { ... }
// - class ThreadStateWidget { ... }

// 新增导入:
import { runApp } from "@flitter/tui";
import { AppWidget } from "../widgets/app-widget.js";
import { ThreadStateWidget } from "../widgets/thread-state-widget.js";
import { InputField } from "../widgets/input-field.js";
import { ConversationView } from "../widgets/conversation-view.js";
import { defaultThemeData } from "../widgets/theme-controller.js";

// 修改 launchInteractiveMode:
export async function launchInteractiveMode(
  args: InteractiveModeArgs,
): Promise<void> {
  // ... 现有 ServiceContainer 设置保持不变 ...
  // ... ThreadWorker 初始化保持不变 ...
  // ... Provider 解析保持不变 ...

  // 非交互分支保持不变
  if (args.prompt && !args.stdin) {
    // ... 直接运行推理 ...
    return;
  }

  // 交互分支: 替换 stub
  const threadStore = container.resolve("threadStore");
  const threadWorker = container.resolve("threadWorker");

  try {
    await runApp(
      new AppWidget({
        themeData: defaultThemeData,
        configService: container.resolve("configService"),
        child: new ThreadStateWidget({
          threadStore,
          threadWorker,
          child: new InputField({
            onSubmit: (text) => {
              threadWorker.runInference({ userMessage: text });
            },
          }),
        }),
      }),
      {
        onRootElementMounted: (rootElement) => {
          container.bind("rootElement", rootElement);
        },
      },
    );
  } finally {
    container.dispose();
  }
}
```
