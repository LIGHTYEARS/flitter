### interactive.e2e.test.ts

```typescript
import { describe, it, expect, mock } from "bun:test";

/**
 * Interactive Mode E2E — CLI 入口到 Widget 挂载。
 */
describe("launchInteractiveMode E2E", () => {
  it("should not contain any stub code", async () => {
    // 读取 interactive.ts 源码，确认不含 _runApp
    const source = await Bun.file("packages/cli/src/modes/interactive.ts").text();
    expect(source).not.toContain("function _runApp");
    expect(source).not.toContain("const ThemeController =");
    expect(source).not.toContain("class AppWidget");
  });

  it("should import from @flitter/tui", async () => {
    const source = await Bun.file("packages/cli/src/modes/interactive.ts").text();
    expect(source).toContain('from "@flitter/tui"');
    expect(source).toContain("runApp");
  });

  it("should pass typecheck", async () => {
    // 使用 Bun.spawn 运行 tsc --noEmit
    const proc = Bun.spawn(["tsc", "--noEmit"], { cwd: process.cwd() });
    const code = await proc.exited;
    expect(code).toBe(0);
  });
});
```
