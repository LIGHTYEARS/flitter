1. **RenderScrollable** extends `RenderBox` (single-child):
   ```typescript
   export class RenderScrollable extends RenderBox {
     private _scrollController: ScrollController;
     private _onScrollChange: (() => void) | null = null;

     constructor(scrollController: ScrollController) { ... }

     set scrollController(value: ScrollController) { /* detach old, attach new */ }

     attach(): void {
       super.attach();
       this._onScrollChange = () => this.markNeedsPaint();
       // Listen to controller offset changes
     }

     detach(): void {
       // Remove listener
       super.detach();
     }

     performLayout(): void {
       // 1. Layout child with unbounded height
       const childConstraints = this.constraints.copyWith({
         minHeight: 0,
         maxHeight: Infinity,
       });
       if (this.child) {
         this.child.layout(childConstraints, { parentUsesSize: true });
       }
       // 2. Own size = constrained viewport size
       this.size = { width: this.constraints.maxWidth, height: this.constraints.maxHeight };
       // 3. Update scroll extent
       const childHeight = this.child?.size.height ?? 0;
       const viewportHeight = this.size.height;
       this._scrollController.updateMaxScrollExtent(
         Math.max(0, childHeight - viewportHeight)
       );
     }

     paint(screen: Screen, offsetX: number, offsetY: number): void {
       if (!this.child) return;
       const scrollOffset = Math.floor(this._scrollController.offset);
       // Paint child with scroll offset applied, clip to viewport
       this.child.paint(screen, offsetX, offsetY - scrollOffset);
       // Note: The Screen's setCell bounds checking handles clipping naturally
     }
   }
   ```
