3. **Rewrite build() for visual fidelity:**
```typescript
build(context: BuildContext): Widget {
  const text = this._controller.text;
  const isEmpty = !text;
  const isFocused = this._focusNode.hasFocus;

  // Border color: primary (#7aa2f7) when focused, border (#3b4261) when not
  const borderColor = isFocused ? "#7aa2f7" : "#3b4261";

  // Content: placeholder or actual text with cursor
  let contentWidget: Widget;
  if (isEmpty) {
    // Placeholder text in mutedText (#565f89)
    contentWidget = new RichText({
      text: new TextSpan({ text: "Type a message...", style: { color: "#565f89" } }),
    });
  } else {
    // Text with cursor at current position (inverse video)
    const cursorPos = this._controller.cursorPosition;
    const before = text.slice(0, cursorPos);
    const cursorChar = text[cursorPos] || " ";
    const after = text.slice(cursorPos + 1);
    contentWidget = new RichText({
      text: new TextSpan({ children: [
        new TextSpan({ text: before, style: { color: "#a9b1d6" } }),
        new TextSpan({ text: cursorChar, style: { color: "#16161e", backgroundColor: "#a9b1d6" } }),  // inverse
        new TextSpan({ text: after, style: { color: "#a9b1d6" } }),
      ]}),
    });
  }

  // Calculate height: line count clamped to 1-5
  const lineCount = Math.min(5, Math.max(1, (text.match(/\n/g) || []).length + 1));

  // Bordered container with padding
  // Top border: ┌──...──┐ in borderColor
  // Content with 1-col left/right padding
  // Bottom border: └──...──┘ in borderColor
  return new Column({ children: [
    new RichText({ text: new TextSpan({ text: `┌${"─".repeat(/* width-2, from MediaQuery */78)}┐`, style: { color: borderColor } }) }),
    new Padding({
      padding: EdgeInsets.symmetric({ horizontal: 1 }),
      child: new SizedBox({ height: lineCount, child: contentWidget }),
    }),
    new RichText({ text: new TextSpan({ text: `└${"─".repeat(78)}┘`, style: { color: borderColor } }) }),
  ]});
}
```

Note: The exact border width should come from MediaQuery terminal width. If MediaQuery is not accessible in build context, use a reasonable default (80 cols) and plan to wire MediaQuery in the integration plan.
