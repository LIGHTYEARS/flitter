```typescript
export class ThreadStateWidgetState extends State<ThreadStateWidget> {
  private _threadSub: Subscription | null = null;
  private _eventSub: Subscription | null = null;
  private _messages: Message[] = [];
  private _inferenceState: "idle" | "running" = "idle";
  private _error: Error | null = null;
  private _scrollController: ScrollController;

  constructor() {
    super();
    this._scrollController = new ScrollController();
  }

  initState(): void {
    super.initState();
    const { threadStore, threadWorker, threadId } = this.widget.config;

    // Subscribe to thread snapshot changes (per D-10)
    const thread$ = threadStore.observeThread(threadId);
    if (thread$) {
      this._threadSub = thread$.subscribe((snapshot) => {
        this.setState(() => {
          this._messages = snapshot.messages.map((m) => ({
            role: m.role as "user" | "assistant" | "system",
            content: typeof m.content === "string" ? m.content
              : m.content.filter((b) => b.type === "text").map((b) => b.text).join(""),
          }));
        });
        // Auto-scroll to bottom on new messages
        if (this._scrollController.followMode) {
          this._scrollController.scrollToBottom();
        }
      });
    }

    // Subscribe to worker events for streaming/errors (per D-11)
    this._eventSub = threadWorker.events$.subscribe((event) => {
      switch (event.type) {
        case "inference:start":
          this.setState(() => { this._inferenceState = "running"; });
          break;
        case "inference:error":
          this.setState(() => { this._error = event.error; this._inferenceState = "idle"; });
          break;
        case "turn:complete":
          this.setState(() => { this._inferenceState = "idle"; this._error = null; });
          break;
      }
    });
  }

  dispose(): void {
    this._threadSub?.unsubscribe();
    this._eventSub?.unsubscribe();
    this._scrollController.dispose();
    super.dispose();
  }

  build(context: BuildContext): Widget {
    const { onSubmit, threadId, modelName, tokenCount } = this.widget.config;

    // Per UI-SPEC layout: Column with Expanded>Scrollable>ConversationView + separators + StatusBar + InputField
    return new Column({ children: [
      // ConversationView (takes all remaining space)
      new Expanded({
        child: new Scrollable({
          controller: this._scrollController,
          child: new ConversationView({
            messages: this._messages,
            inferenceState: this._inferenceState,
            error: this._error,
          }),
        }),
      }),
      // Separator line (1 row)
      new SizedBox({ height: 1, child: new Text({ data: "─".repeat(80) }) }),
      // Status bar (1 row)
      new StatusBar({
        modelName: modelName ?? "unknown",
        tokenCount: tokenCount ?? 0,
        threadId,
      }),
      // Separator line (1 row)
      new SizedBox({ height: 1, child: new Text({ data: "─".repeat(80) }) }),
      // Input field (1-5 rows)
      new InputField({ onSubmit }),
    ]});
  }
}
```

   ```typescript
   async function resolveThread(
     container: ServiceContainer,
     context: CliContext & { threadId?: string; continueThread?: boolean },
   ): Promise<string> {
     // Check --continue flag (resume existing thread)
     if (context.threadId) {
       log.info("Resuming thread", { threadId: context.threadId });
       return context.threadId;
     }

     // If --continue without threadId, find most recent thread
     if (context.continueThread) {
       const threads = container.threadStore.listThreads?.() ?? [];
       if (threads.length > 0) {
         const latest = threads[0]; // Most recent first
         log.info("Continuing most recent thread", { threadId: latest.id });
         return latest.id;
       }
     }

     // Create new thread with welcome message
     const id = crypto.randomUUID();
     container.threadStore.setCachedThread({
       id,
       v: 0,
       messages: [],
       relationships: [],
     } as unknown as ThreadSnapshot);
     log.info("Created new thread", { threadId: id });
     return id;
   }
   ```
