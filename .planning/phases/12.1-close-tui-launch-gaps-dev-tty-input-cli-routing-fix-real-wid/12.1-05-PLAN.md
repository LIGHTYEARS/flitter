```typescript
import { describe, it, before, after } from "node:test";
import assert from "node:assert/strict";
import { execSync, spawn } from "node:child_process";

describe("TUI Launch E2E", () => {
  const SESSION = "flitter-e2e-test";

  before(() => {
    // Kill any existing test session
    try { execSync(`tmux kill-session -t ${SESSION} 2>/dev/null`); } catch {}
  });

  after(() => {
    // Cleanup
    try { execSync(`tmux kill-session -t ${SESSION} 2>/dev/null`); } catch {}
  });

  it("should launch TUI with visible conversation UI", async () => {
    // Start flitter in a tmux session
    execSync(`tmux new-session -d -s ${SESSION} -x 80 -y 24 "npx tsx packages/cli/src/main.ts 2>/dev/null; sleep 1"`);

    // Wait for TUI to initialize
    await new Promise((r) => setTimeout(r, 3000));

    // Capture terminal output
    const captured = execSync(`tmux capture-pane -t ${SESSION} -p`).toString();

    // Verify UI elements are present:
    // 1. Separator lines (horizontal box-drawing characters)
    assert.ok(captured.includes("\u2500"), "Should have separator lines");

    // 2. Empty state message or input area
    const hasEmptyState = captured.includes("No messages") || captured.includes("Type a message");
    assert.ok(hasEmptyState, "Should show empty state or placeholder");

    // 3. Box drawing border characters for InputField
    const hasInputBorder = captured.includes("\u250c") || captured.includes("\u2514");
    assert.ok(hasInputBorder, "Should have InputField border");

    // Send Ctrl+C to exit
    execSync(`tmux send-keys -t ${SESSION} C-c`);
    await new Promise((r) => setTimeout(r, 1000));
  });

  it("should restore terminal after Ctrl+C exit", async () => {
    // Start and immediately exit
    execSync(`tmux new-session -d -s ${SESSION}-exit -x 80 -y 24 "npx tsx packages/cli/src/main.ts 2>/dev/null"`);
    await new Promise((r) => setTimeout(r, 2000));

    // Send Ctrl+C
    execSync(`tmux send-keys -t ${SESSION}-exit C-c`);
    await new Promise((r) => setTimeout(r, 1000));

    // Capture output after exit
    const captured = execSync(`tmux capture-pane -t ${SESSION}-exit -p 2>/dev/null`).toString();

    // After alt screen exit, the TUI content should be gone
    try { execSync(`tmux kill-session -t ${SESSION}-exit 2>/dev/null`); } catch {}
  });
});
```
