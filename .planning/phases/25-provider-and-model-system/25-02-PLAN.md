```typescript
import { z } from 'zod';
import { log } from '../utils/logger';

/**
 * Settings schema — Zod strictObject matching AMP's sKR.
 * 
 * Phase 25 scope: provider/model settings only.
 * Phase 26 will add: agent.deepReasoningEffort, internal.oracleReasoningEffort.
 * Phase 28 will add: internal.compactionThresholdPercent.
 * 
 * Verbatim from AMP source (sKR = X.strictObject({...})):
 */
export const settingsSchema = z.strictObject({
  'anthropic.speed': z.enum(['standard', 'fast']).optional(),
  'anthropic.provider': z.enum(['anthropic', 'vertex']).optional(),
  'anthropic.temperature': z.number().optional(),
  'anthropic.thinking.enabled': z.boolean().optional(),
  'anthropic.interleavedThinking.enabled': z.boolean().optional(),
  'anthropic.effort': z.enum(['low', 'medium', 'high', 'max']).optional(),
  'openai.speed': z.enum(['standard', 'fast']).optional(),
  'internal.model': z.union([z.string(), z.record(z.string(), z.string())]).optional(),
  'gemini.thinkingLevel': z.enum(['minimal', 'low', 'medium', 'high']).optional(),
});

/** Type inferred from the settings schema. */
export type Settings = z.infer<typeof settingsSchema>;

/** Known settings keys — matches AMP's xPR (Phase 25 subset). */
export const SETTINGS_KEYS = [
  'anthropic.effort',
  'anthropic.interleavedThinking.enabled',
  'anthropic.provider',
  'anthropic.speed',
  'anthropic.temperature',
  'anthropic.thinking.enabled',
  'gemini.thinkingLevel',
  'internal.model',
  'openai.speed',
] as const;

/** Dot-notation key type for settings. */
export type SettingsKey = keyof Settings;

/**
 * Hierarchical configuration service with Zod schema validation.
 * Matches AMP's configService pattern: dot-notation keys, typed get/set,
 * strict validation that rejects unknown keys.
 * 
 * Per D-12, resolution priority:
 * 1. Runtime overrides (set() calls)
 * 2. User config file (~/.flitter-cli/config.json "settings" field)
 * 3. Defaults (all optional, undefined = use provider defaults)
 */
export class ConfigService {
  private settings: Settings;

  constructor(initialSettings?: Record<string, unknown>) {
    if (initialSettings) {
      const result = settingsSchema.safeParse(initialSettings);
      if (result.success) {
        this.settings = result.data;
      } else {
        log.warn(`ConfigService: invalid settings, falling back to defaults: ${result.error.message}`);
        this.settings = {};
      }
    } else {
      this.settings = {};
    }
  }

  /**
   * Get a settings value by dot-notation key.
   * Returns undefined if key is not set.
   */
  get<K extends SettingsKey>(key: K): Settings[K] {
    return this.settings[key];
  }

  /**
   * Set a settings value. Validates the value against the schema.
   * Returns true if accepted, false if validation failed.
   */
  set<K extends SettingsKey>(key: K, value: Settings[K]): boolean {
    const testObj = { ...this.settings, [key]: value };
    const result = settingsSchema.safeParse(testObj);
    if (result.success) {
      this.settings = result.data;
      return true;
    }
    log.warn(`ConfigService.set: invalid value for '${key}': ${result.error.message}`);
    return false;
  }

  /**
   * Get the full validated settings object (read-only snapshot).
   */
  getAll(): Readonly<Settings> {
    return { ...this.settings };
  }

  /**
   * Merge new settings into existing. Invalid keys/values are skipped with a warning.
   */
  merge(partial: Record<string, unknown>): void {
    const merged = { ...this.settings, ...partial };
    const result = settingsSchema.safeParse(merged);
    if (result.success) {
      this.settings = result.data;
    } else {
      log.warn(`ConfigService.merge: some settings invalid, keeping previous: ${result.error.message}`);
    }
  }
}
```

This class:
- Uses `z.strictObject()` matching AMP's `sKR` (unknown keys rejected)
- Has `get(key)` and `set(key, value)` with typed access (per D-13)
- Warns + falls back on invalid config (per D-12: "never crash on bad config")
- Keeps Phase 26/28 keys commented out for future — only Phase 25 keys in schema now

```typescript
/**
 * Get the default model name for a provider, using the model catalog.
 * For AMP providers, looks up the registry's defaultModel key in MODEL_CATALOG.

export function getDefaultModel(providerId: string): string {
  // Check AMP provider registry first
  const entry = PROVIDER_REGISTRY[providerId as AmpProviderId];
  if (entry) {
    const catalogEntry = MODEL_CATALOG[entry.defaultModel as ModelKey];
    if (catalogEntry) return catalogEntry.name;

  // Fall back to legacy defaults for flitter-cli extension providers
  return DEFAULT_MODELS[providerId] ?? 'claude-sonnet-4-20250514';
}
```
