```typescript
editPreviousMessage: () => {
  const appState = this._appState; // or however AppState is accessed
  const session = appState.session;
  const items = session.items;

  // Find the last user_message in the conversation
  let lastUserMsgIndex = -1;
  let lastUserMsgText = '';
  for (let i = items.length - 1; i >= 0; i--) {
    if (items[i].type === 'user_message') {
      lastUserMsgIndex = i;
      lastUserMsgText = (items[i] as any).text ?? '';
      break;
    }
  }

  if (lastUserMsgIndex < 0 || !lastUserMsgText) return;

  // Only allow editing when not streaming/processing
  if (session.lifecycle !== 'idle') return;

  // Truncate everything after (and including) the user message
  // truncateAfter keeps items[0..index], so we truncate at index-1 to remove the user msg too
  if (lastUserMsgIndex > 0) {
    session.truncateAfter(lastUserMsgIndex - 1);
  } else {
    // If it's the only item, clear everything (or leave system prompt)
    session.truncateAfter(-1); // This would be a no-op, handle edge case
  }

  // Populate the InputArea with the message text
  // Find the InputArea controller reference and set its text
  // This likely uses appState.inputController or a direct reference
  const inputController = this._inputController; // or however it's accessed
  if (inputController) {
    inputController.text = lastUserMsgText;
  }
},
```
