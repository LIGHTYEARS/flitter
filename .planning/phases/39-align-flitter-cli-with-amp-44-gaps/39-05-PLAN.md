**Step 1: Save current mode to thread handle before switching away**

In `thread-pool.ts`, modify `switchThread()` to accept a callback for saving the current mode:

```typescript
switchThread(threadID: string, options?: { currentAgentMode?: string | null }): void {
  const handle = this.threadHandleMap.get(threadID);
  if (!handle) throw new Error(`ThreadPool.switchThread: thread ${threadID} not found`);

  // Save current agent mode to the outgoing thread handle
  if (options?.currentAgentMode !== undefined && this.activeThreadContextID) {
    const outgoing = this.threadHandleMap.get(this.activeThreadContextID);
    if (outgoing) {
      outgoing.agentMode = options.currentAgentMode;
    }
  }

  this.activateThreadWithNavigation(handle, true);
  log.info(`[thread-pool] switchThread: -> ${threadID}`);
}
```

```typescript
import { log } from '../utils/logger';
import type { ThreadPool } from './thread-pool';
import type { HandoffState } from './types';

export interface HandoffServiceDeps {
  threadPool: ThreadPool;
  getHandoffState: () => HandoffState;
  setHandoffState: (state: Partial<HandoffState>) => void;
  getActiveThreadID: () => string | null;
  submitPrompt: (text: string) => Promise<void>;
  notifyListeners: () => void;
}

export class HandoffService {
  private _deps: HandoffServiceDeps;
  private _countdownTimer: ReturnType<typeof setInterval> | null = null;

  constructor(deps: HandoffServiceDeps) {
    this._deps = deps;
  }

  /** Build system prompt for handoff including source thread context. */
  buildSystemPrompt(sourceThreadID: string, goal: string): string {
    const threadPool = this._deps.threadPool;
    const sourceHandle = threadPool.threadHandleMap.get(sourceThreadID);
    if (!sourceHandle) return goal;

    const items = sourceHandle.session.items;
    const contextSummary = items
      .filter(item => item.type === 'user_message' || item.type === 'assistant_message')
      .slice(-5) // Last 5 messages for context
      .map(item => {
        const role = item.type === 'user_message' ? 'User' : 'Assistant';
        const text = (item as any).text?.slice(0, 200) ?? '';
        return `${role}: ${text}`;
      })
      .join('\n');

    return `Handoff from thread ${sourceThreadID}:\n\nContext:\n${contextSummary}\n\nGoal: ${goal}`;
  }

  enterHandoffMode(): void { /* Move logic from AppState */ }
  exitHandoffMode(): void { /* Move logic from AppState */ }
  async submitHandoff(goal: string): Promise<void> { /* Move logic from AppState */ }
  abortHandoffConfirmation(): void { /* Move logic from AppState */ }

  /** Check if the source thread of an active handoff is still active. */
  followHandoffIfSourceActive(sourceThreadID: string): boolean {
    const threadPool = this._deps.threadPool;
    const worker = threadPool.getOrCreateWorker(sourceThreadID);
    return worker.isRunning;
  }

  dispose(): void {
    if (this._countdownTimer) {
      clearInterval(this._countdownTimer);
      this._countdownTimer = null;
    }
  }
}
```
