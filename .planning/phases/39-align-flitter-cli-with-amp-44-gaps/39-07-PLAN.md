```typescript
registry.registerCommand({
  id: 'thread.setVisibility',
  label: 'Set Thread Visibility',
  category: 'thread',
  action: () => {
    const threadPool = (appState as any).threadPool;
    const activeID = threadPool?.activeThreadContextID;
    if (!activeID) return;
    const handle = threadPool.threadHandleMap.get(activeID);
    if (!handle) return;
    // Cycle through visibility modes
    const modes: ThreadVisibility[] = ['visible', 'hidden', 'archived'];
    const currentIndex = modes.indexOf(handle.visibility);
    const nextVisibility = modes[(currentIndex + 1) % modes.length];
    threadPool.setThreadVisibility(activeID, nextVisibility);
  },
});
```

```typescript
/**
 * Merge source thread into target thread.
 * Sets source status to 'merging', copies messages, then 'merged'.
 * Matches AMP's thread merging lifecycle.
 *
 * @param sourceThreadID - Thread to merge from (will be marked 'merged')
 * @param targetThreadID - Thread to merge into (receives messages)
 */
mergeThread(sourceThreadID: string, targetThreadID: string): void {
  const source = this.threadHandleMap.get(sourceThreadID);
  const target = this.threadHandleMap.get(targetThreadID);
  if (!source || !target) {
    log.warn(`[thread-pool] mergeThread: source or target not found`);
    return;
  }

  // Set source to 'merging' status
  source.status = 'merging';
  this._notifyListeners();

  // Copy user and assistant messages from source to target session.
  // Iterate source items and use the target session's typed methods
  // to add each message in order.
  const sourceItems = source.session.items;
  for (const item of sourceItems) {
    if (item.type === 'user_message') {
      // Use the session's addUserMessage to properly append
      const text = (item as { text?: string }).text ?? '';
      if (text) {
        target.session.addUserMessage(text);
      }
    } else if (item.type === 'assistant_message') {
      // For assistant messages, use startAssistantTurn + appendText
      // pattern if addAssistantMessage doesn't exist directly.
      // Read session.ts to find the correct method. Options:
      //   1. target.session.addAssistantMessage(text) — if method exists
      //   2. target.session.startAssistantTurn(); target.session.appendText(text); target.session.finishAssistantTurn(); — if turn-based API
      //   3. target.session.addItem({ type: 'assistant_message', ... }) — if generic addItem exists
      // Check which API is available by reading session.ts in read_first.
      // The executor MUST use whichever typed method SessionState exposes
      // for adding assistant-originated content. Do NOT cast or use internal
      // arrays directly (_items is private).
      const text = (item as { text?: string }).text ?? '';
      if (text) {
        // Preferred: target.session.addAssistantMessage(text)
        // Fallback: use the streaming API:
        //   target.session.startAssistantTurn();
        //   target.session.appendText(text);
        //   target.session.completeAssistantTurn();
        // The executor should check which methods exist on SessionState
        // and use the appropriate one.
        try {
          if (typeof (target.session as any).addAssistantMessage === 'function') {
            (target.session as any).addAssistantMessage(text);
          } else {
            // Fallback: use streaming turn API
            target.session.startAssistantTurn?.();
            target.session.appendText?.(text);
            target.session.completeAssistantTurn?.();
          }
        } catch (e) {
          log.warn(`[thread-pool] mergeThread: failed to copy assistant message: ${e}`);
        }
      }
    }
    // Skip tool_call, info_message, and other non-conversation items
    // during merge — they are implementation artifacts, not conversation content
  }

  // Register merge relationship
  this.addRelationship('fork', sourceThreadID, targetThreadID);

  // Mark source as 'merged'
  source.status = 'merged';
  this._notifyListeners();

  log.info(`[thread-pool] mergeThread: ${sourceThreadID} -> ${targetThreadID}`);
}

/**
 * Get thread status.
 */
getThreadStatus(threadID: string): ThreadStatus {
  const handle = this.threadHandleMap.get(threadID);
  return handle?.status ?? null;
}
```
